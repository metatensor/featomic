13/// A stupid calculator implementation used to test the API, and API binding to
14/// C/Python/etc.
15///
16/// The calculator has two features: one containing the atom index +
17/// `self.delta`, and the other one containing `x + y + z`.
18#[doc(hidden)]
19#[derive(Debug, Clone)]
20#[derive(serde::Deserialize, serde::Serialize, schemars::JsonSchema)]
21pub struct DummyCalculator {
22    /// Spherical cutoff to use for atomic environments
23    pub cutoff: f64,
24    /// Delta added to the atom index in the first feature
25    pub delta: isize,
26    /// Unused name parameter, to test passing string values
27    pub name: String,
28}
29
30impl CalculatorBase for DummyCalculator {
31    fn name(&self) -> String {
32        // abusing the name as description
33        format!("dummy test calculator with cutoff: {} - delta: {} - name: {}",
34            self.cutoff, self.delta, self.name
35        )
36    }
37
38    fn parameters(&self) -> String {
39        serde_json::to_string(self).expect("failed to serialize to JSON")
40    }
41
42    fn cutoffs(&self) -> &[f64] {
43        std::slice::from_ref(&self.cutoff)
44    }
45
46    fn keys(&self, systems: &mut [Box<dyn System>]) -> Result<Labels, Error> {
47        return CenterTypesKeys.keys(systems);
48    }
49
50    fn sample_names(&self) -> Vec<&str> {
51        AtomCenteredSamples::sample_names()
52    }
53
54    fn samples(&self, keys: &Labels, systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error> {
55        assert_eq!(keys.names(), ["center_type"]);
56        let mut samples = Vec::new();
57        for [center_type] in keys.iter_fixed_size() {
58            let builder = AtomCenteredSamples {
59                cutoff: self.cutoff,
60                center_type: AtomicTypeFilter::Single(center_type.i32()),
61                neighbor_type: AtomicTypeFilter::Any,
62                self_pairs: false,
63            };
64
65            samples.push(builder.samples(systems)?);
66        }
67
68        return Ok(samples);
69    }
70
71    fn supports_gradient(&self, parameter: &str) -> bool {
72        match parameter {
73            "positions" => true,
74            _ => false,
75        }
76    }
77
78    fn positions_gradient_samples(&self, keys: &Labels, samples: &[Labels], systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error> {
79        debug_assert_eq!(keys.count(), samples.len());
80        let mut gradient_samples = Vec::new();
81        for ([center_type], samples) in keys.iter_fixed_size().zip(samples) {
82            let builder = AtomCenteredSamples{
83                cutoff: self.cutoff,
84                center_type: AtomicTypeFilter::Single(center_type.i32()),
85                neighbor_type: AtomicTypeFilter::Any,
86                self_pairs: false,
87            };
88
89            gradient_samples.push(builder.gradients_for(systems, samples)?);
90        }
91
92        return Ok(gradient_samples);
93    }
94
95    fn components(&self, keys: &Labels) -> Vec<Vec<Labels>> {
96        return vec![Vec::new(); keys.count()];
97    }
98
99    fn property_names(&self) -> Vec<&str> {
100        vec!["index_delta", "x_y_z"]
101    }
102
103    fn properties(&self, keys: &Labels) -> Vec<Labels> {
104        let mut properties = LabelsBuilder::new(self.property_names());
105        properties.add(&[1, 0]);
106        properties.add(&[0, 1]);
107        let properties = properties.finish();
108
109        return vec![properties; keys.count()];
110    }
111
112    #[time_graph::instrument(name = "DummyCalculator::compute")]
113    fn compute(&mut self, systems: &mut [Box<dyn System>], descriptor: &mut TensorMap) -> Result<(), Error> {
114        if self.name.contains("log-test-info:") {
115            info!("{}", self.name);
116        } else if self.name.contains("log-test-warn:") {
117            warn!("{}", self.name);
118        }
119
120        for (key, mut block) in descriptor {
121            let center_type = key[0].i32();
122
123            let block_data = block.data_mut();
124            let array = block_data.values.to_array_mut();
125
126            for (sample_i, [system, atom]) in block_data.samples.iter_fixed_size().enumerate() {
127                let system_i = system.usize();
128                let atom_i = atom.usize();
129
130                debug_assert_eq!(systems[system_i].types()?[atom_i], center_type);
131
132                for (property_i, property) in block_data.properties.iter().enumerate() {
133                    if property[0].i32() == 1 {
134                        array[[sample_i, property_i]] = atom_i as f64 + self.delta as f64;
135                    } else if property[1].i32() == 1 {
136                        let system = &mut *systems[system_i];
137                        system.compute_neighbors(self.cutoff)?;
138
139                        let positions = system.positions()?;
140                        let cell = system.cell()?.matrix();
141                        let mut sum = positions[atom_i][0] + positions[atom_i][1] + positions[atom_i][2];
142                        for pair in system.pairs()? {
143                            // this code just check for consistency in the
144                            // neighbor list
145                            let shift = pair.cell_shift_indices[0] as f64 * Vector3D::from(cell[0])
146                                + pair.cell_shift_indices[1] as f64 * Vector3D::from(cell[1])
147                                + pair.cell_shift_indices[2] as f64 * Vector3D::from(cell[2]);
148                            let from_shift = positions[pair.second] - positions[pair.first] + shift;
149                            if !approx::relative_eq!(from_shift, pair.vector, max_relative=1e-6) {
150                                return Err(Error::InvalidParameter(format!(
151                                    "system implementation returned inconsistent neighbors list:\
152                                    pair.vector is {:?}, but the cell shift give {:?} for atoms {}-{}",
153                                    pair.vector, from_shift, pair.first, pair.second
154                                )));
155                            }
156
157                            if !approx::relative_eq!(pair.vector.norm(), pair.distance, max_relative=1e-6) {
158                                return Err(Error::InvalidParameter(format!(
159                                    "system implementation returned inconsistent neighbors list:\
160                                    pair.vector norm is {}, but pair.distance is {} for atoms {}-{}",
161                                    pair.vector.norm(), pair.distance, pair.first, pair.second
162                                )));
163                            }
164
165                            let pairs_by_center = system.pairs_containing(pair.first)?;
166                            if !pairs_by_center.iter().any(|p| p == pair) {
167                                return Err(Error::InvalidParameter(format!(
168                                    "system implementation returned inconsistent neighbors list:\
169                                    pairs_containing({}) does not contains a pair for atoms {}-{}",
170                                    pair.first, pair.first, pair.second
171                                )));
172                            }
173
174                            let pairs_by_center = system.pairs_containing(pair.second)?;
175                            if !pairs_by_center.iter().any(|p| p == pair) {
176                                return Err(Error::InvalidParameter(format!(
177                                    "system implementation returned inconsistent neighbors list:\
178                                    pairs_containing({}) does not contains a pair for atoms {}-{}",
179                                    pair.second, pair.first, pair.second
180                                )));
181                            }
182                            // end of neighbors list consistency check
183
184                            // actual values calculation
185                            if pair.first == atom_i {
186                                sum += positions[pair.second][0] + positions[pair.second][1] + positions[pair.second][2];
187                            }
188
189                            if pair.second == atom_i {
190                                sum += positions[pair.first][0] + positions[pair.first][1] + positions[pair.first][2];
191                            }
192                        }
193
194                        array[[sample_i, property_i]] = sum;
195                    }
196                }
197            }
198
199            if let Some(mut gradient) = block.gradient_mut("positions") {
200                let gradient = gradient.data_mut();
201                let array = gradient.values.to_array_mut();
202
203                for gradient_sample_i in 0..array.shape()[0] {
204                    for (property_i, property) in gradient.properties.iter().enumerate() {
205                        if property[0].i32() == 1 {
206                            array[[gradient_sample_i, 0, property_i]] = 0.0;
207                            array[[gradient_sample_i, 1, property_i]] = 0.0;
208                            array[[gradient_sample_i, 2, property_i]] = 0.0;
209                        } else if property[1].i32() == 1 {
210                            array[[gradient_sample_i, 0, property_i]] = 1.0;
211                            array[[gradient_sample_i, 1, property_i]] = 1.0;
212                            array[[gradient_sample_i, 2, property_i]] = 1.0;
213                        }
214                    }
215                }
216            }
217        }
218
219        return Ok(());
220    }
221}

254    #[test]
255    fn values() {
256        let mut calculator = Calculator::from(Box::new(DummyCalculator{
257            cutoff: 1.0,
258            delta: 9,
259            name: String::new(),
260        }) as Box<dyn CalculatorBase>);
261
262        let mut systems = test_systems(&["water"]);
263        let descriptor = calculator.compute(&mut systems, Default::default()).unwrap();
264
265        let keys = descriptor.keys();
266        assert_eq!(keys.names(), ["center_type"]);
267        assert_eq!(keys.count(), 2);
268        assert_eq!(keys[0], [-42]);
269        assert_eq!(keys[1], [1]);
270
271        let o_block = &descriptor.block_by_id(0);
272        let values = o_block.values().to_array();
273        assert_eq!(values.shape(), [1, 2]);
274        assert_eq!(values.slice(s![0, ..]), aview1(&[9.0, -1.1778999999999997]));
275
276        let h_block = &descriptor.block_by_id(1);
277        let values = h_block.values().to_array();
278        assert_eq!(values.shape(), [2, 2]);
279        assert_eq!(values.slice(s![0, ..]), aview1(&[10.0, 0.16649999999999998]));
280        assert_eq!(values.slice(s![1, ..]), aview1(&[11.0, -1.3443999999999998]));
281    }
