20/// Parameters for SOAP power spectrum calculator.
21///
22/// In the SOAP power spectrum, each sample represents rotationally-averaged
23/// atomic density correlations, built on top of the spherical expansion. Each
24/// sample is a vector indexed by `n1, n2, l`, where `n1` and `n2` are radial
25/// basis indexes and `l` is the angular index:
26///
27/// `< n1 n2 l | X_i > = \sum_m < n1 l m | X_i > < n2 l m | X_i >`
28///
29/// where the `< n l m | X_i >` are the spherical expansion coefficients.
30///
31/// See [this review article](https://doi.org/10.1063/1.5090481) for more
32/// information on the SOAP representations.
33#[derive(Debug, Clone)]
34#[derive(serde::Deserialize, serde::Serialize, schemars::JsonSchema)]
35pub struct PowerSpectrumParameters {
36    /// Definition of the atomic environment within a cutoff, and how
37    /// neighboring atoms enter and leave the environment.
38    pub cutoff: Cutoff,
39    /// Definition of the density arising from atoms in the local environment.
40    pub density: Density,
41    /// Definition of the basis functions used to expand the atomic density
42    pub basis: SphericalExpansionBasis<SoapRadialBasis>,
43}
44
45/// Calculator implementing the Smooth Overlap of Atomic Position (SOAP) power
46/// spectrum representation of atomistic systems.
47pub struct SoapPowerSpectrum {
48    parameters: PowerSpectrumParameters,
49    spherical_expansion: Calculator,
50}

58impl SoapPowerSpectrum {
59    pub fn new(parameters: PowerSpectrumParameters) -> Result<SoapPowerSpectrum, Error> {
60        let expansion_parameters = SphericalExpansionParameters {
61            cutoff: parameters.cutoff,
62            density: parameters.density,
63            basis: parameters.basis.clone(),
64        };
65
66        let spherical_expansion = SphericalExpansion::new(expansion_parameters)?;
67
68        return Ok(SoapPowerSpectrum {
69            parameters: parameters,
70            spherical_expansion: Calculator::from(
71                Box::new(spherical_expansion) as Box<dyn CalculatorBase>
72            ),
73        });
74    }

76    /// Construct a `TensorMap` containing the set of samples/properties we want
77    /// the spherical expansion calculator to compute.
78    ///
79    /// For each block, samples will contain the same set of samples as the
80    /// power spectrum, even if a neighbor type might not be around, since
81    /// that simplifies the accumulation loops quite a lot.
82    fn selected_spx_labels(&self, descriptor: &TensorMap) -> TensorMap {
83        assert_eq!(descriptor.keys().names(), ["center_type", "neighbor_1_type", "neighbor_2_type"]);
84
85        // first, go over the requested power spectrum properties and group them
86        // depending on the neighbor_type
87        let mut requested_by_key = HashMap::new();
88        let mut requested_o3_lambda = BTreeSet::new();
89        for (&[center, neighbor_1, neighbor_2], block) in descriptor.keys().iter_fixed_size().zip(descriptor.blocks()) {
90            for &[l, n1, n2] in block.properties().iter_fixed_size() {
91                requested_o3_lambda.insert(l.usize());
92
93                let (_, properties) = requested_by_key
94                    .entry([l, 1.into(), center, neighbor_1])
95                    .or_insert_with(|| (BTreeSet::new(), BTreeSet::new()));
96                properties.insert([n1]);
97
98                let (_, properties) = requested_by_key
99                    .entry([l, 1.into(), center, neighbor_2])
100                    .or_insert_with(|| (BTreeSet::new(), BTreeSet::new()));
101                properties.insert([n2]);
102            }
103        }
104
105        // make sure all the expected blocks are there, even if the power
106        // spectrum does not contain e.g. l=3 at all. The corresponding blocks
107        // will have an empty set of properties
108        for &[center, neighbor_1, neighbor_2] in descriptor.keys().iter_fixed_size() {
109            for &l in &requested_o3_lambda {
110                requested_by_key
111                    .entry([l.into(), 1.into(), center, neighbor_1])
112                    .or_insert_with(|| (BTreeSet::new(), BTreeSet::new()));
113
114                requested_by_key
115                    .entry([l.into(), 1.into(), center, neighbor_2])
116                    .or_insert_with(|| (BTreeSet::new(), BTreeSet::new()));
117            }
118        }
119
120        // Then, loop over the requested power spectrum, and accumulate the
121        // samples we want to compute.
122        for (&[_, _, requested_center, requested_neighbor], (samples, _)) in &mut requested_by_key {
123            for (key, block) in descriptor {
124                let center = key[0];
125                let neighbor_1 = key[1];
126                let neighbor_2 = key[2];
127
128                if center != requested_center {
129                    continue;
130                }
131
132                if !(requested_neighbor == neighbor_1 || requested_neighbor == neighbor_2) {
133                    continue;
134                }
135
136                for &sample in block.samples().iter_fixed_size::<2>() {
137                    samples.insert(sample);
138                }
139            }
140        }
141
142        let mut keys_builder = LabelsBuilder::new(vec!["o3_lambda", "o3_sigma", "center_type", "neighbor_type"]);
143        let mut blocks = Vec::new();
144        for (key, (samples, properties)) in requested_by_key {
145            keys_builder.add(&key);
146
147            let mut samples_builder = LabelsBuilder::new(vec!["system", "atom"]);
148            for entry in samples {
149                samples_builder.add(&entry);
150            }
151            let samples = samples_builder.finish();
152
153            let mut properties_builder = LabelsBuilder::new(vec!["n"]);
154            for entry in properties {
155                properties_builder.add(&entry);
156            }
157            let properties = properties_builder.finish();
158
159            blocks.push(TensorBlock::new(
160                EmptyArray::new(vec![samples.count(), properties.count()]),
161                &samples,
162                &[],
163                &properties,
164            ).expect("invalid TensorBlock"));
165        }
166
167        // if the user selected only a subset of l entries, make sure there are
168        // empty blocks for the corresponding keys in the spherical expansion
169        // selection
170        let mut missing_keys = BTreeSet::new();
171        for &[center, neighbor_1, neighbor_2] in descriptor.keys().iter_fixed_size() {
172            for o3_lambda in self.parameters.basis.angular_channels() {
173                if !requested_o3_lambda.contains(&o3_lambda) {
174                    missing_keys.insert([o3_lambda.into(), 1.into(), center, neighbor_1]);
175                    missing_keys.insert([o3_lambda.into(), 1.into(), center, neighbor_2]);
176                }
177            }
178        }
179        for key in missing_keys {
180            keys_builder.add(&key);
181
182            let samples = Labels::empty(vec!["system", "atom"]);
183            let properties = Labels::empty(vec!["n"]);
184            blocks.push(TensorBlock::new(
185                EmptyArray::new(vec![samples.count(), properties.count()]),
186                &samples,
187                &[],
188                &properties,
189            ).expect("invalid TensorBlock"));
190        }
191
192        return TensorMap::new(keys_builder.finish(), blocks).expect("invalid TensorMap")
193    }

195    /// Pre-compute the correspondance between samples of the spherical
196    /// expansion & the power spectrum, both for values and gradients.
197    ///
198    /// For example, the key `center, neighbor_1, neighbor_2 = 1, 6, 8` will
199    /// have a very different set of samples from `c, n_1, n_2 = 1, 6, 6`; but
200    /// both will use the spherical expansion `center, neighbor = 1, 6`.
201    ///
202    /// This function returns the list of spherical expansion sample indexes
203    /// corresponding to the requested samples in `descriptor` for each block.
204    fn samples_mapping(
205        descriptor: &TensorMap,
206        spherical_expansion: &TensorMap
207    ) -> HashMap<Vec<LabelValue>, SamplesMapping> {
208        let mut mapping = HashMap::new();
209        for (key, block) in descriptor {
210            let center_type = key[0];
211            let neighbor_1_type = key[1];
212            let neighbor_2_type = key[2];
213
214            let block_data = block.data();
215            if block_data.properties.count() == 0 {
216                // no properties to compute, we don't really care about sample
217                // mapping and we can not compute the real one (there is no l to
218                // find the corresponding spx block), so we'll create a dummy
219                // sample mapping / gradient sample mapping
220                let mut values_mapping = Vec::new();
221                for i in 0..block_data.samples.count() {
222                    values_mapping.push((i, i));
223                }
224
225                let mut gradient_mapping = Vec::new();
226                if let Some(gradient) = block.gradient("positions") {
227                    let gradient = gradient.data();
228                    for i in 0..gradient.samples.count() {
229                        gradient_mapping.push((Some(i), Some(i)));
230                    }
231                }
232
233                mapping.insert(key.to_vec(), SamplesMapping {
234                    values: values_mapping,
235                    gradients: gradient_mapping,
236                });
237                continue;
238            }
239
240            let mut values_mapping = Vec::new();
241
242            // the spherical expansion samples are the same for all
243            // `o3_lambda` values, so we only need to compute it for
244            // the first one.
245            let first_l = block_data.properties[0][0];
246
247            let block_id_1 = spherical_expansion.keys().position(&[
248                first_l, 1.into(), center_type, neighbor_1_type
249            ]).expect("missing block in spherical expansion");
250            let spx_block_1 = &spherical_expansion.block_by_id(block_id_1);
251            let spx_samples_1 = spx_block_1.samples();
252
253            let block_id_2 = spherical_expansion.keys().position(&[
254                first_l, 1.into(), center_type, neighbor_2_type
255            ]).expect("missing block in spherical expansion");
256            let spx_block_2 = &spherical_expansion.block_by_id(block_id_2);
257            let spx_samples_2 = spx_block_2.samples();
258
259            values_mapping.reserve(block_data.samples.count());
260            for sample in &*block_data.samples {
261                let sample_1 = spx_samples_1.position(sample).expect("missing spherical expansion sample");
262                let sample_2 = spx_samples_2.position(sample).expect("missing spherical expansion sample");
263                values_mapping.push((sample_1, sample_2));
264            }
265
266            let mut gradient_mapping = Vec::new();
267            if let Some(gradient) = block.gradient("positions") {
268                let spx_gradient_1 = spx_block_1.gradient("positions").expect("missing spherical expansion gradients");
269                let spx_gradient_2 = spx_block_2.gradient("positions").expect("missing spherical expansion gradients");
270
271                let gradient_samples = gradient.samples();
272                gradient_mapping.reserve(gradient_samples.count());
273
274                let spx_gradient_1_samples = spx_gradient_1.samples();
275                let spx_gradient_2_samples = spx_gradient_2.samples();
276
277                for &[sample, system, atom] in gradient_samples.iter_fixed_size() {
278                    // The "sample" dimension in the power spectrum gradient
279                    // samples do not necessarily matches the "sample" dimension
280                    // in the spherical expansion gradient samples. We use the
281                    // sample mapping for the values to create what would be the
282                    // right gradient sample for spx, and then lookup its
283                    // position in the spx gradient samples
284                    let (spx_1_sample, spx_2_sample) = values_mapping[sample.usize()];
285
286                    let mapping_1 = spx_gradient_1_samples.position(
287                        &[spx_1_sample.into(), system, atom]
288                    );
289                    let mapping_2 = spx_gradient_2_samples.position(
290                        &[spx_2_sample.into(), system, atom]
291                    );
292
293                    // at least one of the spx block should contribute to the
294                    // gradients
295                    debug_assert!(mapping_1.is_some() || mapping_2.is_some());
296
297                    gradient_mapping.push((mapping_1, mapping_2));
298                }
299            }
300
301            mapping.insert(key.to_vec(), SamplesMapping {
302                values: values_mapping,
303                gradients: gradient_mapping
304            });
305        }
306
307        return mapping;
308    }

310    /// Get the list of spherical expansion to combine when computing a single
311    /// block (associated with the given key) of the power spectrum.
312    fn spx_properties_to_combine<'a>(
313        key: &[LabelValue],
314        properties: &Labels,
315        spherical_expansion: &HashMap<&[LabelValue], SphericalExpansionBlock<'a>>,
316    ) -> Vec<SpxPropertiesToCombine<'a>> {
317        let center_type = key[0];
318        let neighbor_1_type = key[1];
319        let neighbor_2_type = key[2];
320
321        return properties.par_iter().map(|property| {
322            let l = property[0];
323            let n1 = property[1];
324            let n2 = property[2];
325
326            let key_1: &[_] = &[l, 1.into(), center_type, neighbor_1_type];
327            let block_1 = spherical_expansion.get(&key_1)
328            .expect("missing first neighbor type block in spherical expansion");
329
330            let key_2: &[_] = &[l, 1.into(), center_type, neighbor_2_type];
331            let block_2 = spherical_expansion.get(&key_2)
332                .expect("missing first neighbor type block in spherical expansion");
333
334            // both blocks should had the same number of m components
335            debug_assert_eq!(block_1.values.shape()[1], block_2.values.shape()[1]);
336
337            let property_1 = block_1.properties.position(&[n1]).expect("missing n1");
338            let property_2 = block_2.properties.position(&[n2]).expect("missing n2");
339
340            let o3_lambda = l.usize();
341
342            // For consistency with a full Clebsch-Gordan product we need to add
343            // a `-1^l / sqrt(2 l + 1)` factor to the power spectrum invariants
344            let normalization = if o3_lambda % 2 == 0 {
345                f64::sqrt((2 * o3_lambda + 1) as f64)
346            } else {
347                -f64::sqrt((2 * o3_lambda + 1) as f64)
348            };
349
350            SpxPropertiesToCombine {
351                o3_lambda,
352                normalization,
353                property_1,
354                property_2,
355                spx_1: block_1.clone(),
356                spx_2: block_2.clone(),
357            }
358        }).collect();
359    }
360}

363/// Data about the two spherical expansion block that will get combined to
364/// produce a single (l, n1, n2) property in a single power spectrum block
365struct SpxPropertiesToCombine<'a> {
366    /// value of l
367    o3_lambda: usize,
368    /// normalization factor $-1^l * \sqrt{2 l + 1}$
369    normalization: f64,
370    /// position of n1 in the first spherical expansion properties
371    property_1: usize,
372    /// position of n2 in the second spherical expansion properties
373    property_2: usize,
374    /// first spherical expansion block
375    spx_1: SphericalExpansionBlock<'a>,
376    /// second spherical expansion block
377    spx_2: SphericalExpansionBlock<'a>,
378}
379
380/// Data from a single spherical expansion block
381#[derive(Debug, Clone)]
382struct SphericalExpansionBlock<'a> {
383    properties: Labels,
384    /// spherical expansion values
385    values: &'a ndarray::ArrayD<f64>,
386    /// spherical expansion position gradients
387    positions_gradients: Option<&'a ndarray::ArrayD<f64>>,
388    /// spherical expansion cell gradients
389    cell_gradients: Option<&'a ndarray::ArrayD<f64>>,
390    /// spherical expansion strain gradients
391    strain_gradients: Option<&'a ndarray::ArrayD<f64>>,
392}
393
394/// Indexes of the spherical expansion samples/rows corresponding to each power
395/// spectrum row.
396struct SamplesMapping {
397    /// Mapping for the values: if the row `i` of the power spectrum is a
398    /// combination of the rows `j` and `k` of two spherical expansion blocks,
399    /// then this vector will contain `(j, k)` at index `i`
400    values: Vec<(usize, usize)>,
401    /// Mapping for the gradients, with a similar layout as the `values`
402    ///
403    /// Some samples might not be defined in both of the spherical expansion
404    /// blocks being considered, for examples when dealing with two different
405    /// neighbor types, only one the sample corresponding to the right
406    /// neighbor type will be `Some`.
407    gradients: Vec<(Option<usize>, Option<usize>)>,
408}

436    fn samples(&self, keys: &metatensor::Labels, systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error> {
437        assert_eq!(keys.names(), ["center_type", "neighbor_1_type", "neighbor_2_type"]);
438        let mut result = Vec::new();
439        for [center_type, neighbor_1_type, neighbor_2_type] in keys.iter_fixed_size() {
440
441            let builder = AtomCenteredSamples {
442                cutoff: self.parameters.cutoff.radius,
443                center_type: AtomicTypeFilter::Single(center_type.i32()),
444                // we only want center with both neighbor types present
445                neighbor_type: AtomicTypeFilter::AllOf(
446                    [
447                        neighbor_1_type.i32(),
448                        neighbor_2_type.i32()
449                    ].iter().copied().collect()
450                ),
451                self_pairs: true,
452            };
453
454            result.push(builder.samples(systems)?);
455        }
456
457        return Ok(result);
458    }

460    fn positions_gradient_samples(&self, keys: &Labels, samples: &[Labels], systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error> {
461        assert_eq!(keys.names(), ["center_type", "neighbor_1_type", "neighbor_2_type"]);
462        assert_eq!(keys.count(), samples.len());
463
464        let mut gradient_samples = Vec::new();
465        for ([center_type, neighbor_1_type, neighbor_2_type], samples) in keys.iter_fixed_size().zip(samples) {
466            let builder = AtomCenteredSamples {
467                cutoff: self.parameters.cutoff.radius,
468                center_type: AtomicTypeFilter::Single(center_type.i32()),
469                // gradients samples should contain either neighbor types
470                neighbor_type: AtomicTypeFilter::OneOf(vec![
471                    neighbor_1_type.i32(),
472                    neighbor_2_type.i32()
473                ]),
474                self_pairs: true,
475            };
476
477            gradient_samples.push(builder.gradients_for(systems, samples)?);
478        }
479
480        return Ok(gradient_samples);
481    }

498    fn properties(&self, keys: &metatensor::Labels) -> Vec<Labels> {
499        match self.parameters.basis {
500            SphericalExpansionBasis::TensorProduct(ref basis) => {
501                let mut properties = LabelsBuilder::new(self.property_names());
502                for l in 0..=basis.max_angular {
503                    for n1 in 0..basis.radial.size() {
504                        for n2 in 0..basis.radial.size() {
505                            properties.add(&[l, n1, n2]);
506                        }
507                    }
508                }
509
510                return vec![properties.finish(); keys.count()];
511            }
512            SphericalExpansionBasis::Explicit(ref basis) => {
513                let mut properties = LabelsBuilder::new(self.property_names());
514                for (&l, radial) in &*basis.by_angular {
515                    for n1 in 0..radial.size() {
516                        for n2 in 0..radial.size() {
517                            properties.add(&[l, n1, n2]);
518                        }
519                    }
520                }
521
522                return vec![properties.finish(); keys.count()];
523            },
524        }

527    #[time_graph::instrument(name = "SoapPowerSpectrum::compute")]
528    #[allow(clippy::too_many_lines)]
529    fn compute(&mut self, systems: &mut [Box<dyn System>], descriptor: &mut TensorMap) -> Result<(), Error> {
530        assert!(descriptor.keys().count() > 0);
531
532        let mut gradients = Vec::new();
533        if descriptor.block_by_id(0).gradient("positions").is_some() {
534            gradients.push("positions");
535        }
536        if descriptor.block_by_id(0).gradient("cell").is_some() {
537            gradients.push("cell");
538        }
539        if descriptor.block_by_id(0).gradient("strain").is_some() {
540            gradients.push("strain");
541        }
542
543        let selected = self.selected_spx_labels(descriptor);
544
545        let options = CalculationOptions {
546            gradients: &gradients,
547            selected_samples: LabelsSelection::Predefined(&selected),
548            selected_properties: LabelsSelection::Predefined(&selected),
549            selected_keys: Some(selected.keys()),
550            ..Default::default()
551        };
552
553        let spherical_expansion = self.spherical_expansion.compute(
554            systems,
555            options,
556        ).expect("failed to compute spherical expansion");
557        let samples_mapping = SoapPowerSpectrum::samples_mapping(descriptor, &spherical_expansion);
558
559        let spherical_expansion = spherical_expansion.iter().map(|(key, block)| {
560            let spx_block = SphericalExpansionBlock {
561                properties: block.properties(),
562                values: block.values().to_array(),
563                positions_gradients: block.gradient("positions").map(|g| g.values().to_array()),
564                cell_gradients: block.gradient("cell").map(|g| g.values().to_array()),
565                strain_gradients: block.gradient("strain").map(|g| g.values().to_array()),
566            };
567
568            (key, spx_block)
569        }).collect();
570
571        for (key, mut block) in descriptor {
572            let neighbor_1_type = key[1];
573            let neighbor_2_type = key[2];
574
575            let mut block_data = block.data_mut();
576            let properties_to_combine = SoapPowerSpectrum::spx_properties_to_combine(
577                key,
578                &block_data.properties,
579                &spherical_expansion,
580            );
581
582            let mapping = samples_mapping.get(key).expect("missing sample mapping");
583
584            block_data.values.as_array_mut()
585                .axis_iter_mut(ndarray::Axis(0))
586                .into_par_iter()
587                .zip_eq(&mapping.values)
588                .for_each(|(mut values, &(spx_sample_1, spx_sample_2))| {
589                    for (property_i, spx) in properties_to_combine.iter().enumerate() {
590                        let SpxPropertiesToCombine { spx_1, spx_2, ..} = spx;
591
592                        let mut sum = 0.0;
593
594                        for m in 0..(2 * spx.o3_lambda + 1) {
595                            // unsafe is required to remove the bound checking
596                            // in release mode (`uget` still checks bounds in
597                            // debug mode)
598                            unsafe {
599                                sum += spx_1.values.uget([spx_sample_1, m, spx.property_1])
600                                     * spx_2.values.uget([spx_sample_2, m, spx.property_2]);
601                            }
602                        }
603
604                        if neighbor_1_type != neighbor_2_type {
605                            // We only store values for `neighbor_1_type < neighbor_2_type`
606                            // because the values are the same for pairs
607                            // `neighbor_1_type <-> neighbor_2_type` and
608                            // `neighbor_2_type <-> neighbor_1_type`.
609                            // To ensure the final kernels are correct, we have
610                            // to multiply the corresponding values.
611                            sum *= std::f64::consts::SQRT_2;
612                        }
613
614                        unsafe {
615                            *values.uget_mut(property_i) = sum / spx.normalization;
616                        }
617                    }
618                });
619
620            // gradients with respect to the atomic positions
621            if let Some(mut gradient) = block.gradient_mut("positions") {
622                let gradient = gradient.data_mut();
623
624                gradient.values.to_array_mut()
625                    .axis_iter_mut(ndarray::Axis(0))
626                    .into_par_iter()
627                    .zip_eq(gradient.samples.par_iter())
628                    .zip_eq(&mapping.gradients)
629                    .for_each(|((mut values, gradient_sample), &(spx_grad_sample_1, spx_grad_sample_2))| {
630                        for (property_i, spx) in properties_to_combine.iter().enumerate() {
631                            let SpxPropertiesToCombine { spx_1, spx_2, ..} = spx;
632
633                            let spx_1_gradient = spx_1.positions_gradients.expect("missing spherical expansion gradients");
634                            let spx_2_gradient = spx_2.positions_gradients.expect("missing spherical expansion gradients");
635
636                            let sample_i = gradient_sample[0].usize();
637                            let (spx_sample_1, spx_sample_2) = mapping.values[sample_i];
638
639                            let mut sum = [0.0, 0.0, 0.0];
640                            if let Some(grad_sample_1) = spx_grad_sample_1 {
641                                for m in 0..(2 * spx.o3_lambda + 1) {
642                                    // SAFETY: see same loop for values
643                                    unsafe {
644                                        let value_2 = spx_2.values.uget([spx_sample_2, m, spx.property_2]);
645                                        for d in 0..3 {
646                                            sum[d] += value_2 * spx_1_gradient.uget([grad_sample_1, d, m, spx.property_1]);
647                                        }
648                                    }
649                                }
650                            }
651
652                            if let Some(grad_sample_2) = spx_grad_sample_2 {
653                                for m in 0..(2 * spx.o3_lambda + 1) {
654                                    // SAFETY: see same loop for values
655                                    unsafe {
656                                        let value_1 = spx_1.values.uget([spx_sample_1, m, spx.property_1]);
657                                        for d in 0..3 {
658                                            sum[d] += value_1 * spx_2_gradient.uget([grad_sample_2, d, m, spx.property_2]);
659                                        }
660                                    }
661                                }
662                            }
663
664                            if neighbor_1_type != neighbor_2_type {
665                                // see above
666                                for d in 0..3 {
667                                    sum[d] *= std::f64::consts::SQRT_2;
668                                }
669                            }
670
671                            for d in 0..3 {
672                                unsafe {
673                                    *values.uget_mut([d, property_i]) = sum[d] / spx.normalization;
674                                }
675                            }
676                        }
677                    });
678            }
679
680            // gradients with respect to the strain/cell
681            for parameter in ["cell", "strain"] {
682                if let Some(mut gradient) = block.gradient_mut(parameter) {
683                    let gradient = gradient.data_mut();
684
685                    gradient.values.to_array_mut()
686                        .axis_iter_mut(ndarray::Axis(0))
687                        .into_par_iter()
688                        .zip_eq(gradient.samples.par_iter())
689                        .for_each(|(mut values, gradient_sample)| {
690                            for (property_i, spx) in properties_to_combine.iter().enumerate() {
691                                let SpxPropertiesToCombine { spx_1, spx_2, ..} = spx;
692
693                                let (spx_1_gradient, spx_2_gradient) = if parameter == "cell" {
694                                    let spx_1_gradient = spx_1.cell_gradients.expect("missing spherical expansion gradients");
695                                    let spx_2_gradient = spx_2.cell_gradients.expect("missing spherical expansion gradients");
696
697                                    (spx_1_gradient, spx_2_gradient)
698                                } else if parameter == "strain" {
699                                    let spx_1_gradient = spx_1.strain_gradients.expect("missing spherical expansion gradients");
700                                    let spx_2_gradient = spx_2.strain_gradients.expect("missing spherical expansion gradients");
701
702                                    (spx_1_gradient, spx_2_gradient)
703                                } else {
704                                    panic!("invalid gradient parameter: {}", parameter);
705                                };
706
707
708                                let sample_i = gradient_sample[0].usize();
709                                let (spx_sample_1, spx_sample_2) = mapping.values[sample_i];
710
711                                let mut sum = [
712                                    [0.0, 0.0, 0.0],
713                                    [0.0, 0.0, 0.0],
714                                    [0.0, 0.0, 0.0],
715                                ];
716
717                                // accumulate \grad SPX_1 * SPX_2
718                                for m in 0..(2 * spx.o3_lambda + 1) {
719                                    // SAFETY: see same loop for values
720                                    unsafe {
721                                        let value_2 = spx_2.values.uget([spx_sample_2, m, spx.property_2]);
722                                        for xyz_1 in 0..3 {
723                                            for xyz_2 in 0..3 {
724                                                sum[xyz_1][xyz_2] += value_2 * spx_1_gradient.uget([spx_sample_1, xyz_1, xyz_2, m, spx.property_1]);
725                                            }
726                                        }
727                                    }
728                                }
729
730                                // accumulate \grad SPX_2 * SPX_1
731                                for m in 0..(2 * spx.o3_lambda + 1) {
732                                    // SAFETY: see same loop for values
733                                    unsafe {
734                                        let value_1 = spx_1.values.uget([spx_sample_1, m, spx.property_1]);
735                                        for xyz_1 in 0..3 {
736                                            for xyz_2 in 0..3 {
737                                                sum[xyz_1][xyz_2] += value_1 * spx_2_gradient.uget([spx_sample_2, xyz_1, xyz_2, m, spx.property_2]);
738                                            }
739                                        }
740                                    }
741                                }
742
743                                if neighbor_1_type != neighbor_2_type {
744                                    // see above
745                                    for xyz_1 in 0..3 {
746                                        for xyz_2 in 0..3 {
747                                            sum[xyz_1][xyz_2] *= std::f64::consts::SQRT_2;
748                                        }
749                                    }
750                                }
751
752                                // store accumulated gradient in output
753                                for xyz_1 in 0..3 {
754                                    for xyz_2 in 0..3 {
755                                        unsafe {
756                                            *values.uget_mut([xyz_1, xyz_2, property_i]) = sum[xyz_1][xyz_2] / spx.normalization;
757                                        }
758                                    }
759                                }
760                            }
761                        });
762                }
763            }
764        }
765
766        Ok(())
767    }
768}
