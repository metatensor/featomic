11/// `LodeRadialIntegralSpline` allows to evaluate another radial integral
12/// implementation using [cubic Hermit spline][splines-wiki].
13///
14/// This can be much faster than using the actual radial integral
15/// implementation.
16///
17/// [splines-wiki]: https://en.wikipedia.org/wiki/Cubic_Hermite_spline
18pub struct LodeRadialIntegralSpline {
19    spline: Arc<HermitCubicSpline<ndarray::Ix1>>,
20    density: DensityKind,
21    center_contribution: Option<Array1<f64>>,
22}
23
24impl LodeRadialIntegralSpline {
25    /// Create a new `LodeRadialIntegralSpline` taking values from the given
26    /// `radial_integral`. Points are added to the spline until the requested
27    /// accuracy is reached. We consider that the accuracy is reached when
28    /// either the mean absolute error or the mean relative error gets below the
29    /// `accuracy` threshold.
30    #[time_graph::instrument(name = "LodeRadialIntegralSpline::with_accuracy")]
31    pub fn with_accuracy(
32        radial_integral: impl LodeRadialIntegral,
33        density: DensityKind,
34        k_cutoff: f64,
35        accuracy: f64,
36        with_center_contribution: bool,
37    ) -> Result<LodeRadialIntegralSpline, Error> {
38        let size = radial_integral.size();
39        let spline_parameters = SplineParameters {
40            start: 0.0,
41            stop: k_cutoff,
42            shape: vec![size],
43        };
44
45        let spline = HermitCubicSpline::with_accuracy(
46            accuracy,
47            spline_parameters,
48            |x| {
49                let mut values = Array1::from_elem(size, 0.0);
50                let mut derivatives = Array1::from_elem(size, 0.0);
51                radial_integral.compute(x, values.view_mut(), Some(derivatives.view_mut()));
52                (values, derivatives)
53            },
54        )?;
55
56        let mut center_contribution = None;
57        if with_center_contribution {
58            center_contribution = Some(radial_integral.get_center_contribution(density)?);
59        }
60
61        return Ok(LodeRadialIntegralSpline {
62            spline: Arc::new(spline),
63            density: density,
64            center_contribution: center_contribution,
65        });
66    }
67
68    /// Create a new `LodeRadialIntegralSpline` with user-defined spline points.
69    ///
70    /// The density/`tabulated.center_contribution` are assumed to match each
71    /// other
72    pub fn from_tabulated(tabulated: LodeTabulated, density: DensityKind) -> LodeRadialIntegralSpline {
73        return LodeRadialIntegralSpline {
74            spline: tabulated.spline,
75            density: density,
76            center_contribution: tabulated.center_contribution,
77        };
78    }
79}
80
81impl LodeRadialIntegral for LodeRadialIntegralSpline {
82    fn size(&self) -> usize {
83        self.spline.points[0].values.shape()[0]
84    }
85
86    #[time_graph::instrument(name = "SplinedRadialIntegral::compute")]
87    fn compute(&self, x: f64, values: ArrayViewMut1<f64>, gradients: Option<ArrayViewMut1<f64>>) {
88        self.spline.compute(x, values, gradients);
89    }
90
91    fn get_center_contribution(&self, density: DensityKind) -> Result<Array1<f64>, Error> {
92        if density != self.density {
93            return Err(Error::InvalidParameter("mismatched atomic density in splined LODE radial integral".into()));
94        }
95
96        if self.center_contribution.is_none() {
97            return Err(Error::InvalidParameter(
98                "`center_contribution` must be defined for the Tabulated radial \
99                basis used with the L=0 angular channel".into()
100            ));
101        }
102
103        return Ok(self.center_contribution.clone().expect("just checked"));
104    }
105}

107#[cfg(test)]
108mod tests {
109    use approx::assert_relative_eq;
110
111    use crate::calculators::LodeRadialBasis;

113    use super::*;
114    use super::super::LodeRadialIntegralGto;

116    #[test]
117    fn high_accuracy() {
118        // Check that even with high accuracy and large domain MAX_SPLINE_SIZE is enough
119        let basis = LodeRadialBasis::Gto { max_radial: 15, radius: 5.0 };
120        let gto = LodeRadialIntegralGto::new(&basis, 3).unwrap();
121
122        let accuracy = 5e-10;
123        let k_cutoff = 10.0;
124        let density = DensityKind::SmearedPowerLaw { smearing: 0.5, exponent: 1 };
125
126        // this test only check that this code runs without crashing
127        LodeRadialIntegralSpline::with_accuracy(gto, density, k_cutoff, accuracy, true).unwrap();
128    }
129
130    #[test]
131    fn finite_difference() {
132        let radial_size = 8;
133        let basis = LodeRadialBasis::Gto { max_radial: (radial_size - 1), radius: 5.0 };
134        let gto = LodeRadialIntegralGto::new(&basis, 3).unwrap();
135
136        let accuracy = 1e-2;
137        let k_cutoff = 10.0;
138        let density = DensityKind::SmearedPowerLaw { smearing: 0.5, exponent: 1 };
139
140        // even with very bad accuracy, we want the gradients of the spline to match the
141        // values produces by the spline, and not necessarily the actual GTO gradients.
142        let spline = LodeRadialIntegralSpline::with_accuracy(gto, density, k_cutoff, accuracy, false).unwrap();
143
144        let rij = 3.4;
145        let delta = 1e-9;
146
147        let mut values = Array1::from_elem(radial_size, 0.0);
148        let mut values_delta = Array1::from_elem(radial_size, 0.0);
149        let mut gradients = Array1::from_elem(radial_size, 0.0);
150        spline.compute(rij, values.view_mut(), Some(gradients.view_mut()));
151        spline.compute(rij + delta, values_delta.view_mut(), None);
152
153        let finite_differences = (&values_delta - &values) / delta;
154        assert_relative_eq!(
155            finite_differences, gradients,
156            epsilon=delta, max_relative=5e-6
157        );
158    }
