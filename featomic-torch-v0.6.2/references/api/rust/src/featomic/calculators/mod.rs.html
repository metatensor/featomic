5/// The `CalculatorBase` trait is the interface shared by all calculator
6/// implementations; and used by [`crate::Calculator`] to run the calculation.
7///
8/// This should not be used directly by end users, who should use the facilities
9/// in [`crate::Calculator`] instead.
10///
11/// `std::panic::RefUnwindSafe` is a required super-trait to enable passing
12/// calculators across the C API.
13pub trait CalculatorBase: std::panic::RefUnwindSafe {
14    /// Get the name of this Calculator
15    fn name(&self) -> String;
16
17    /// Get the parameters used to create this Calculator as a JSON string
18    fn parameters(&self) -> String;
19
20    /// Get the all radial cutoffs used by this Calculator's neighbors lists
21    /// (which can be an empty list)
22    fn cutoffs(&self) -> &[f64];
23
24    /// Get the set of keys for this calculator and the given systems
25    fn keys(&self, systems: &mut [Box<dyn System>]) -> Result<Labels, Error>;
26
27    /// Get the names used for sample labels by this calculator
28    fn sample_names(&self) -> Vec<&str>;
29
30    /// Get the full list of samples this calculator would create for the given
31    /// systems. This function should return one set of samples for each key.
32    fn samples(&self, keys: &Labels, systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error>;
33
34    /// Can this calculator compute gradients with respect to the `parameter`?
35    /// Right now, `parameter` can be either `"positions"`, `"strain"` or
36    /// `"cell"`.
37    fn supports_gradient(&self, parameter: &str) -> bool;
38
39    /// Get the samples for gradients with respect to positions, corresponding
40    /// the given values samples.
41    ///
42    /// The `samples` slice contains one set of samples for each key.
43    ///
44    /// If the gradients with respect to positions are not available, this
45    /// function should return an error.
46    fn positions_gradient_samples(&self, keys: &Labels, samples: &[Labels], systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error>;
47
48    /// Get the components this calculator computes for each key.
49    fn components(&self, keys: &Labels) -> Vec<Vec<Labels>>;
50
51    /// Get the names used for property labels by this calculator
52    fn property_names(&self) -> Vec<&str>;
53
54    /// Get the properties this calculator computes for each key.
55    fn properties(&self, keys: &Labels) -> Vec<Labels>;
56
57    /// Actually run the calculation.
58    ///
59    /// This function is given a pre-allocated descriptor, filled with zeros.
60    /// The samples/properties in each blocks might not match the values
61    /// returned by [`CalculatorBase::samples`] and
62    /// [`CalculatorBase::properties`]: instead they will only contain the
63    /// values that where requested by the end user.
64    ///
65    /// Gradients (with respect to positions or cell) are allocated in each
66    /// block if they are supported according to
67    /// [`CalculatorBase::supports_gradient`], and the users requested them as
68    /// part of the calculation options.
69    fn compute(&mut self, systems: &mut [Box<dyn System>], descriptor: &mut TensorMap) -> Result<(), Error>;
70}
