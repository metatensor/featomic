18/// Pair of atoms coming from a neighbor list.
19// WARNING: any change to this definition MUST be reflected in featomic_pair_t as
20// well
21#[repr(C)]
22#[derive(Debug, Clone, Copy, PartialEq)]
23pub struct Pair {
24    /// index of the first atom in the pair
25    pub first: usize,
26    /// index of the second atom in the pair
27    pub second: usize,
28    /// distance between the two atoms
29    pub distance: f64,
30    /// vector from the first atom to the second atom, accounting for periodic
31    /// boundary conditions. This should be `position\[second\] -
32    /// position\[first\] + H * cell_shift` where `H` is the cell matrix.
33    pub vector: Vector3D,
34    /// How many cell shift where applied to the `second` atom to create this
35    /// pair.
36    pub cell_shift_indices: [i32; 3],
37}
38
39/// A `System` deals with the storage of atoms and related information, as well
40/// as the computation of neighbor lists.
41pub trait System: Send + Sync {
42    /// Get the unit cell for this system
43    fn cell(&self) -> Result<UnitCell, Error>;
44
45    /// Get the number of atoms in this system
46    fn size(&self) -> Result<usize, Error>;
47
48    /// Get the atomic types for all atoms in this system. The returned value
49    /// must be a slice of length `self.size()`, where each different atomic
50    /// type is identified with a different integer value. These values are
51    /// usually the atomic number, but don't have to.
52    fn types(&self) -> Result<&[i32], Error>;
53
54    /// Get the positions for all atoms in this system. The returned value must
55    /// be a slice of length `self.size()` containing the Cartesian coordinates
56    /// of all atoms in the system.
57    fn positions(&self) -> Result<&[Vector3D], Error>;
58
59    /// Compute the neighbor list according to the given cutoff, and store it
60    /// for later access with `pairs` or `pairs_around`.
61    fn compute_neighbors(&mut self, cutoff: f64) -> Result<(), Error>;
62
63    /// Get the list of pairs in this system. This list of pair should only
64    /// contain each pair once (and not twice as `i-j` and `j-i`), should not
65    /// contain self pairs (`i-i`); and should only contains pairs where the
66    /// distance between atoms is actually bellow the cutoff passed in the last
67    /// call to `compute_neighbors`. This function is only valid to call after a
68    /// call to `compute_neighbors`.
69    fn pairs(&self) -> Result<&[Pair], Error>;
70
71    /// Get the list of pairs in this system which include the atom at the given
72    /// index. The same restrictions on the list of pairs as `System::pairs`
73    /// applies, with the additional condition that the pair `i-j` should be
74    /// included both in the return of `pairs_containing(i)` and
75    /// `pairs_containing(j)`.
76    fn pairs_containing(&self, atom: usize) -> Result<&[Pair], Error>;
77}
