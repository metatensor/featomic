9/// A `SoapRadialIntegral` computes the SOAP radial integral for all radial
10/// basis functions and a single spherical harmonic `l` channel
11///
12/// See equations 5 to 8 of [this paper](https://doi.org/10.1063/5.0044689) for
13/// mor information on the radial integral.
14///
15/// `std::panic::RefUnwindSafe` is a required super-trait to enable passing
16/// radial integrals across the C API. `Send` is a required super-trait to
17/// enable passing radial integrals between threads.
18#[allow(clippy::doc_markdown)]
19pub trait SoapRadialIntegral: std::panic::RefUnwindSafe + Send {
20    /// Compute the radial integral for a single `distance` between two atoms
21    /// and store the resulting data in the `values` array. If `gradients` is
22    /// `Some`, also compute and store gradients there.
23    ///
24    /// The radial integral $I_{nl}$ is defined as "the non-spherical harmonics
25    /// part of the spherical expansion". Depending on the atomic density,
26    /// different expressions can be used.
27    ///
28    /// For a delta density, the radial integral is simply the radial basis
29    /// function $R_{nl}$ evaluated at the pair distance:
30    ///
31    /// $$ I_{nl}(r_{ij}) = R_{nl}(r_{ij}) $$
32    ///
33    /// For a Gaussian atomic density with a width of $\sigma$, the radial
34    /// integral reduces to:
35    ///
36    /// $$
37    /// I_{nl}(r_{ij}) = \frac{4\pi}{(\pi \sigma^2)^{3/4}} e^{-\frac{r_{ij}^2}{2\sigma^2}}
38    ///     \int_0^\infty \mathrm{d}r r^2 R_{nl}(r) e^{-\frac{r^2}{2\sigma^2}} i_l\left(\frac{rr_{ij}}{\sigma^2}\right)
39    /// $$
40    ///
41    /// where $i_l$ is the modified spherical Bessel function of the first kind
42    /// of order $l$.
43    ///
44    /// Finally, for an arbitrary spherically symmetric atomic density `g(r)`,
45    /// the radial integral is
46    ///
47    /// $$
48    /// I_{nl}(r_{ij}) = 2\pi \int_0^\infty \mathrm{d}r r^2 R_{nl}(r)
49    ///     \int_{-1}^1 \mathrm{d}u P_l(u) g(\sqrt{r^2+r_{ij}^2-2rr_{ij}u})
50    /// $$
51    ///
52    /// where $P_l$ is the l-th Legendre polynomial.
53    fn compute(&self, distance: f64, values: ArrayViewMut1<f64>, gradients: Option<ArrayViewMut1<f64>>);
54
55    /// Get how many basis functions are part of this integral. This is the
56    /// shape to use for the `values` and `gradients` parameters to `compute`.
57    fn size(&self) -> usize;
58}

66/// Store together a radial integral implementation and cached allocation for
67/// values/gradients.
68pub struct SoapRadialIntegralCache {
69    /// Implementation of the radial integral
70    implementation: Box<dyn SoapRadialIntegral>,
71    /// Cache for the radial integral values
72    pub(crate) values: Array1<f64>,
73    /// Cache for the radial integral gradient
74    pub(crate) gradients: Array1<f64>,
75}
76
77impl SoapRadialIntegralCache {
78    fn new(
79        o3_lambda: usize,
80        radial: &SoapRadialBasis,
81        density: DensityKind,
82        cutoff: f64,
83        spline_accuracy: Option<f64>,
84    ) -> Result<SoapRadialIntegralCache, Error> {
85        // We only support some specific combinations of density and basis
86        let implementation = match (density, radial) {
87            // Gaussian density + GTO basis
88            (DensityKind::Gaussian {..}, &SoapRadialBasis::Gto { .. }) => {
89                let gto = SoapRadialIntegralGto::new(cutoff, density, radial, o3_lambda)?;
90
91                if let Some(accuracy) = spline_accuracy {
92                    Box::new(SoapRadialIntegralSpline::with_accuracy(
93                        gto, cutoff, accuracy
94                    )?)
95                } else {
96                    Box::new(gto) as Box<dyn SoapRadialIntegral>
97                }
98            },
99            // Dirac density + tabulated basis (also used for
100            // tabulated radial integral with a different density)
101            (DensityKind::DiracDelta, SoapRadialBasis::Tabulated(tabulated)) => {
102                Box::new(SoapRadialIntegralSpline::from_tabulated(
103                    tabulated.clone()
104                )) as Box<dyn SoapRadialIntegral>
105            }
106            // Everything else is an error
107            _ => {
108                return Err(Error::InvalidParameter(
109                    "this combination of basis and density is not supported in SOAP".into()
110                ))
111            }
112        };
113
114        let size = implementation.size();
115        let values = Array1::from_elem(size, 0.0);
116        let gradients = Array1::from_elem(size, 0.0);
117
118        return Ok(SoapRadialIntegralCache {
119            implementation,
120            values,
121            gradients,
122        });
123    }
124
125    /// Run the calculation, the results are stored inside `self.values` and
126    /// `self.gradients`
127    pub fn compute(&mut self, distance: f64, do_gradients: bool) {
128        let gradient_view = if do_gradients {
129            Some(self.gradients.view_mut())
130        } else {
131            None
132        };
133
134        self.implementation.compute(distance, self.values.view_mut(), gradient_view);
135    }
136}

138/// Store all `SoapRadialIntegralCache` for different angular channels
139pub struct SoapRadialIntegralCacheByAngular {
140    pub(crate) by_angular: BTreeMap<usize, SoapRadialIntegralCache>,
141}
142
143impl SoapRadialIntegralCacheByAngular {
144    /// Create a new `SoapRadialIntegralCacheByAngular` for the given radial basis & density
145    pub fn new(
146        cutoff: f64,
147        density: DensityKind,
148        basis: &SphericalExpansionBasis<SoapRadialBasis>
149    ) -> Result<SoapRadialIntegralCacheByAngular, Error> {
150        match basis {
151            SphericalExpansionBasis::TensorProduct(basis) => {
152                let mut by_angular = BTreeMap::new();
153                for o3_lambda in 0..=basis.max_angular {
154                    let cache = SoapRadialIntegralCache::new(
155                        o3_lambda,
156                        &basis.radial,
157                        density,
158                        cutoff,
159                        basis.spline_accuracy
160                    )?;
161                    by_angular.insert(o3_lambda, cache);
162                }
163
164                return Ok(SoapRadialIntegralCacheByAngular { by_angular });
165            }
166            SphericalExpansionBasis::Explicit(basis) => {
167                let mut by_angular = BTreeMap::new();
168                for (&o3_lambda, radial) in &*basis.by_angular {
169                    let cache = SoapRadialIntegralCache::new(
170                        o3_lambda,
171                        radial,
172                        density,
173                        cutoff,
174                        basis.spline_accuracy
175                    )?;
176                    by_angular.insert(o3_lambda, cache);
177                }
178                return Ok(SoapRadialIntegralCacheByAngular {
179                    by_angular
180                });
181            }
182        }
183    }
184
185    /// Run the calculation, the results are accessible with `get`
186    pub fn compute(&mut self, distance: f64, do_gradients: bool) {
187        self.by_angular.iter_mut().for_each(|(_, cache)| cache.compute(distance, do_gradients));
188    }
189
190    /// Get one of the individual cache, corresponding to the `o3_lambda`
191    /// angular channel
192    pub fn get(&self, o3_lambda: usize) -> Option<&SoapRadialIntegralCache> {
193        self.by_angular.get(&o3_lambda)
194    }
195
196    pub(crate) fn get_mut(&mut self, o3_lambda: usize) -> Option<&mut SoapRadialIntegralCache> {
197        self.by_angular.get_mut(&o3_lambda)
198    }
