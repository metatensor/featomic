1use crate::Error;

3/// Definition of a local environment for SOAP calculations
4#[derive(Debug, Clone, Copy)]
5#[derive(serde::Deserialize, serde::Serialize, schemars::JsonSchema)]
6#[serde(deny_unknown_fields)]
7pub struct Cutoff {
8    /// Radius of the spherical cutoff to use for atomic environments
9    pub radius: f64,
10    /// Cutoff function used to smooth the behavior around the cutoff radius
11    pub smoothing: Smoothing,
12}
13
14/// Possible values for the smoothing cutoff function
15#[derive(Debug, Clone, Copy)]
16#[derive(serde::Deserialize, serde::Serialize, schemars::JsonSchema)]
17#[serde(deny_unknown_fields)]
18#[serde(tag = "type")]
19pub enum Smoothing {
20    /// Shifted cosine smoothing function
21    /// `f(r) = 1/2 * (1 + cos(Ï€ (r - cutoff + width) / width ))`
22    ShiftedCosine {
23        /// Width of the switching function
24        width: f64,
25    },
26    /// Step smoothing function (i.e. no smoothing). This is 1 inside the cutoff
27    /// and 0 outside, with a sharp step at the boundary.
28    Step,
29}
30
31impl Cutoff {
32    pub fn validate(&self) -> Result<(), Error> {
33        match self.smoothing {
34            Smoothing::Step => {},
35            Smoothing::ShiftedCosine { width } => {
36                if width <= 0.0 || !width.is_finite() {
37                    return Err(Error::InvalidParameter(format!(
38                        "expected positive width for shifted cosine cutoff function, got {}",
39                        width
40                    )));
41                }
42            }
43        }
44        return Ok(());
45    }
46
47    /// Evaluate the smoothing function at the distance `r`
48    pub fn smoothing(&self, r: f64) -> f64 {
49        match self.smoothing {
50            Smoothing::Step => {
51                if r >= self.radius { 0.0 } else { 1.0 }
52            },
53            Smoothing::ShiftedCosine { width } => {
54                if r <= (self.radius - width) {
55                    1.0
56                } else if r >= self.radius {
57                    0.0
58                } else {
59                    let s = std::f64::consts::PI * (r - self.radius + width) / width;
60                    0.5 * (1. + f64::cos(s))
61                }
62            }
63        }
64    }
65
66    /// Evaluate the gradient of the smoothing function at the distance `r`
67    pub fn smoothing_gradient(&self, r: f64) -> f64 {
68        match self.smoothing {
69            Smoothing::Step => 0.0,
70            Smoothing::ShiftedCosine { width } => {
71                if r <= (self.radius - width) || r >= self.radius {
72                    0.0
73                } else {
74                    let s = std::f64::consts::PI * (r - self.radius + width) / width;
75                    return -0.5 * std::f64::consts::PI * f64::sin(s) / width;
76                }
77            }
78        }
79    }
80}

83#[cfg(test)]
84mod tests {
85    use super::*;
86    #[test]
87    fn no_smoothing() {
88        let cutoff = Cutoff { radius: 4.0, smoothing: Smoothing::Step};
89
90        assert_eq!(cutoff.smoothing(2.0), 1.0);
91        assert_eq!(cutoff.smoothing(5.0), 0.0);
92
93        assert_eq!(cutoff.smoothing_gradient(2.0), 0.0);
94        assert_eq!(cutoff.smoothing_gradient(5.0), 0.0);
95    }
96
97    #[test]
98    fn shifted_cosine() {
99        let cutoff = Cutoff { radius: 4.0, smoothing: Smoothing::ShiftedCosine { width: 0.5 }};
100
101        assert_eq!(cutoff.smoothing(2.0), 1.0);
102        assert_eq!(cutoff.smoothing(3.5), 1.0);
103        assert_eq!(cutoff.smoothing(3.8), 0.34549150281252683);
104        assert_eq!(cutoff.smoothing(4.0), 0.0);
105        assert_eq!(cutoff.smoothing(5.0), 0.0);
106
107        assert_eq!(cutoff.smoothing_gradient(2.0), 0.0);
108        assert_eq!(cutoff.smoothing_gradient(3.5), 0.0);
109        assert_eq!(cutoff.smoothing_gradient(3.8), -2.987832164741557);
110        assert_eq!(cutoff.smoothing_gradient(4.0), 0.0);
111        assert_eq!(cutoff.smoothing_gradient(5.0), 0.0);
112    }
