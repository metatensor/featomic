11/// Implementation of `metatensor::Array` storing a view inside another array
12///
13/// This is relatively unsafe, and only viable for use inside this module.
14struct UnsafeArrayViewMut {
15    /// Shape of the sub-array
16    shape: Vec<usize>,
17    /// Pointer to the first element of the data. This point inside another
18    /// array that is ASSUMED to stay alive for as long as this one does.
19    ///
20    /// We can not use lifetimes to track this assumption, since metatensor
21    /// requires `'static` lifetimes
22    data: *mut f64,
23}
24
25// SAFETY: `UnsafeArrayViewMut` can be transferred from one thread to another
26unsafe impl Send for UnsafeArrayViewMut {}
27// SAFETY: `UnsafeArrayViewMut` is Sync since there is no interior mutability
28// (each array being a separate mutable view in the initial array)
29unsafe impl Sync for UnsafeArrayViewMut {}
30
31impl metatensor::Array for UnsafeArrayViewMut {
32    fn as_any(&self) -> &dyn std::any::Any {
33        self
34    }
35
36    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
37        self
38    }
39
40    fn create(&self, _: &[usize]) -> Box<dyn metatensor::Array> {
41        unimplemented!("invalid operation on UnsafeArrayViewMut");
42    }
43
44    fn copy(&self) -> Box<dyn metatensor::Array> {
45        unimplemented!("invalid operation on UnsafeArrayViewMut");
46    }
47
48    fn data(&mut self) -> &mut [f64] {
49        unimplemented!("invalid operation on UnsafeArrayViewMut");
50    }
51
52    fn shape(&self) -> &[usize] {
53        &self.shape
54    }
55
56    fn reshape(&mut self, _: &[usize]) {
57        unimplemented!("invalid operation on UnsafeArrayViewMut");
58    }
59
60    fn swap_axes(&mut self, _: usize, _: usize) {
61        unimplemented!("invalid operation on UnsafeArrayViewMut");
62    }
63
64    fn move_samples_from(
65        &mut self,
66        _: &dyn metatensor::Array,
67        _: &[metatensor::c_api::mts_sample_mapping_t],
68        _: std::ops::Range<usize>,
69    ) {
70        unimplemented!("invalid operation on UnsafeArrayViewMut");
71    }
72}

74/// Extract an array stored in the `TensorBlock` returned by `split_tensor_map_by_system`
75pub fn array_mut_for_system(array: metatensor::ArrayRefMut<'_>) -> ArrayViewMutD<'_, f64> {
76    let array = array.to_any_mut().downcast_mut::<UnsafeArrayViewMut>().expect("invalid array type");
77
78    // SAFETY: we checked that the arrays do not overlap when creating
79    // `UnsafeArrayViewMut` in split_by_system
80    return unsafe {
81        ArrayViewMutD::from_shape_ptr(array.shape.clone(), array.data)
82    };
83}

85/// View inside a `TensorMap` corresponding to one system
86pub struct TensorMapView<'a> {
87    // all arrays in this TensorMap are `UnsafeArrayViewMut` with the lifetime
88    // tracked by the marker
89    data: TensorMap,
90    marker: std::marker::PhantomData<&'a mut TensorMap>,
91}
92
93impl std::ops::Deref for TensorMapView<'_> {
94    type Target = TensorMap;
95
96    fn deref(&self) -> &Self::Target {
97        &self.data
98    }
99}
100
101impl std::ops::DerefMut for TensorMapView<'_> {
102    fn deref_mut(&mut self) -> &mut Self::Target {
103        &mut self.data
104    }
105}

108/// Split a descriptor into multiple descriptors, one by system. The resulting
109/// descriptors contain views inside the descriptor
110#[allow(clippy::too_many_lines)]
111pub fn split_tensor_map_by_system(descriptor: &mut TensorMap, n_systems: usize) -> Vec<TensorMapView<'_>> {
112    #[derive(Debug, Clone, Copy, PartialEq)]
113    struct GradientPosition {
114        positions: usize,
115        cell: usize,
116        strain: usize,
117    }
118
119    let mut descriptor_by_system = Vec::new();
120
121    let mut values_end = vec![0; descriptor.keys().count()];
122    let mut gradients_end = vec![GradientPosition { positions: 0, cell: 0, strain: 0 }; descriptor.keys().count()];
123    for system_i in 0..n_systems {
124        let blocks = descriptor.par_iter_mut()
125            .zip_eq(&mut values_end)
126            .zip_eq(&mut gradients_end)
127            .map(|(((_, mut block), system_end), system_end_grad)| {
128                let mut block_data = block.data_mut();
129
130                let mut samples = LabelsBuilder::new(block_data.samples.names());
131                let mut samples_mapping = BTreeMap::new();
132                let mut system_per_sample = vec![LabelValue::new(-1); block_data.samples.count()];
133
134                let system_start = *system_end;
135                for (sample_i, &[system, atom]) in block_data.samples.iter_fixed_size().enumerate().skip(system_start) {
136                    system_per_sample[sample_i] = system;
137
138                    if system.usize() == system_i {
139                        // this sample is part of to the current system
140                        samples.add(&[system, atom]);
141                        let new_sample = samples_mapping.len();
142                        samples_mapping.insert(sample_i, new_sample);
143
144                        *system_end += 1;
145                    } else if system.usize() > system_i {
146                        // found the next system
147                        break;
148                    } else {
149                        // system.usize() < system_i
150                        panic!("expected samples to be ordered by system, they are not");
151                    }
152                }
153
154                let mut shape = Vec::new();
155
156                let samples = samples.finish_assume_unique();
157                shape.push(samples.count());
158
159                for component in &*block_data.components {
160                    shape.push(component.count());
161                }
162
163                let n_properties = block_data.properties.count();
164                shape.push(n_properties);
165
166                let per_sample_size: usize = shape.iter().skip(1).product();
167                let data_ptr = unsafe {
168                    // SAFETY: this creates non-overlapping regions (from
169                    // `data_ptr` to `data_ptr + shape.product()`.
170                    //
171                    // `per_sample_size * system_start` skips all the data
172                    // associated with the previous systems.
173                    block_data.values.as_array_mut().as_mut_ptr().add(per_sample_size * system_start)
174                };
175
176                let values = UnsafeArrayViewMut {
177                    shape: shape,
178                    data: data_ptr,
179                };
180                let mut new_block = TensorBlock::new(
181                    values,
182                    &samples,
183                    &block_data.components,
184                    &block_data.properties,
185                ).expect("invalid TensorBlock");
186
187                for (parameter, mut gradient) in block.gradients_mut() {
188                    let gradient = gradient.data_mut();
189
190                    let system_end_grad = match parameter {
191                        "positions" => &mut system_end_grad.positions,
192                        "cell" => &mut system_end_grad.cell,
193                        "strain" => &mut system_end_grad.strain,
194                        other => panic!("unsupported gradient parameter {}", other)
195                    };
196                    let system_start_grad = *system_end_grad;
197
198                    let mut samples = LabelsBuilder::new(gradient.samples.names());
199                    for gradient_sample in gradient.samples.iter().skip(system_start_grad) {
200                        let sample_i = gradient_sample[0].usize();
201                        let system = system_per_sample[sample_i];
202                        if system.usize() == system_i {
203                            // this sample is part of to the current system
204                            let mut new_gradient_sample = gradient_sample.to_vec();
205                            new_gradient_sample[0] = samples_mapping[&sample_i].into();
206                            samples.add(&new_gradient_sample);
207
208                            *system_end_grad += 1;
209                        } else if system.usize() > system_i {
210                            // found the next system
211                            break;
212                        } else {
213                            // system.usize() < system_i
214                            panic!("expected samples to be ordered by system, they are not");
215                        }
216                    }
217
218                    let mut shape = Vec::new();
219
220                    let samples = samples.finish_assume_unique();
221                    shape.push(samples.count());
222
223                    for component in &*gradient.components {
224                        shape.push(component.count());
225                    }
226
227                    shape.push(n_properties);
228
229                    let per_sample_size: usize = shape.iter().skip(1).product();
230                    let data_ptr = unsafe {
231                        // SAFETY: same as the values above, this is creating
232                        // multiple non-overlapping regions in memory
233                        gradient.values.to_array_mut().as_mut_ptr().add(per_sample_size * system_start_grad)
234                    };
235
236                    let values = UnsafeArrayViewMut {
237                        shape: shape,
238                        data: data_ptr,
239                    };
240
241                    new_block.add_gradient(
242                        parameter,
243                        TensorBlock::new(
244                            values,
245                            &samples,
246                            &gradient.components,
247                            &gradient.properties,
248                        ).expect("created invalid gradients")
249                    ).expect("created invalid gradients");
250                }
251
252                return new_block;
253            }).collect();
254
255        let tensor = TensorMapView {
256            data: TensorMap::new(descriptor.keys().clone(), blocks).expect("invalid TensorMap"),
257            marker: std::marker::PhantomData
258        };
259
260        descriptor_by_system.push(tensor);
261    }
262
263    return descriptor_by_system;
264}
