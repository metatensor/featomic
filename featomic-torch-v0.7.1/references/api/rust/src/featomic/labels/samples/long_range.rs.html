7/// Samples builder for long-range (i.e. infinite cutoff), per atom, two bodies
8/// representation.
9///
10/// With this builder, all atoms are considered neighbors of all other atoms.
11pub struct LongRangeSamplesPerAtom {
12    /// Filter for the central atom type
13    pub center_type: AtomicTypeFilter,
14    /// Filter for the neighbor atom type
15    pub neighbor_type: AtomicTypeFilter,
16    /// Should the central atom be considered it's own neighbor?
17    pub self_pairs: bool,
18}
19
20impl SamplesBuilder for LongRangeSamplesPerAtom {
21    fn sample_names() -> Vec<&'static str> {
22        vec!["system", "atom"]
23    }
24
25    fn samples(&self, systems: &mut [Box<dyn System>]) -> Result<Labels, Error> {
26        assert!(self.self_pairs, "self.self_pairs = false is not implemented");
27
28        let mut builder = LabelsBuilder::new(Self::sample_names());
29        for (system_i, system) in systems.iter_mut().enumerate() {
30            let types = system.types()?;
31
32            // we want to take all atoms matching `center_type` iff
33            // there is at least one atom in the system matching
34            // `neighbor_type`
35            let mut has_matching_neighbor = false;
36            for &neighbor_type in types {
37                if self.neighbor_type.matches(neighbor_type) {
38                    has_matching_neighbor = true;
39                    break;
40                }
41            }
42
43            if has_matching_neighbor {
44                for (center_i, &center_type) in types.iter().enumerate() {
45                    if self.center_type.matches(center_type) {
46                        builder.add(&[system_i, center_i]);
47                    }
48                }
49            }
50        }
51
52        return Ok(builder.finish_assume_unique());
53    }
54
55    fn gradients_for(&self, systems: &mut [Box<dyn System>], samples: &Labels) -> Result<Labels, Error> {
56        assert_eq!(samples.names(), ["system", "atom"]);
57        let mut builder = LabelsBuilder::new(vec!["sample", "system", "atom"]);
58
59        for (sample_i, [system_i, center_i]) in samples.iter_fixed_size().enumerate() {
60            let system_i = system_i.usize();
61
62            let system = &mut systems[system_i];
63            for (neighbor_i, &neighbor_type) in system.types()?.iter().enumerate() {
64                if self.neighbor_type.matches(neighbor_type) || neighbor_i == center_i.usize() {
65                    builder.add(&[sample_i, system_i, neighbor_i]);
66                }
67            }
68        }
69
70        return Ok(builder.finish_assume_unique());
71    }
72}

74#[cfg(test)]
75mod tests {
76    use super::*;
77    use crate::systems::test_utils::test_systems;
78
79    #[test]
80    fn all_samples() {
81        let mut systems = test_systems(&["CH", "water"]);
82        let builder = LongRangeSamplesPerAtom {
83            center_type: AtomicTypeFilter::Single(1),
84            neighbor_type: AtomicTypeFilter::Any,
85            self_pairs: true,
86        };
87
88        let samples = builder.samples(&mut systems).unwrap();
89        assert_eq!(samples, Labels::new(
90            ["system", "atom"],
91            &[[0, 1], [1, 1], [1, 2]],
92        ));
93
94
95        let gradient_samples = builder.gradients_for(&mut systems, &samples).unwrap();
96        assert_eq!(gradient_samples, Labels::new(
97            ["sample", "system", "atom"],
98            &[
99                // gradients of atoms in CH
100                [0, 0, 0], [0, 0, 1],
101                // gradients of atoms in water
102                [1, 1, 0], [1, 1, 1], [1, 1, 2],
103                [2, 1, 0], [2, 1, 1], [2, 1, 2],
104            ]
105        ));
106    }
