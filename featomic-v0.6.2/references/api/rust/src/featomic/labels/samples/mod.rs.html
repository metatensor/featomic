7/// Atomic type filters to be used when building samples and gradient sample
8pub enum AtomicTypeFilter {
9    /// Any atomic type is fine
10    Any,
11    /// Only the given atomic type should match
12    Single(i32),
13    /// Any of the given atomic type is fine
14    OneOf(Vec<i32>),
15    /// All of the given atoms types must be present. This can only be used
16    /// for neighbor types selection.
17    AllOf(BTreeSet<i32>),
18}
19
20impl AtomicTypeFilter {
21    /// Check if a given type matches the filter
22    pub fn matches(&self, atomic_type: i32) -> bool {
23        match self {
24            AtomicTypeFilter::Any => true,
25            AtomicTypeFilter::Single(selected) => atomic_type == *selected,
26            AtomicTypeFilter::OneOf(selected) => selected.contains(&atomic_type),
27            AtomicTypeFilter::AllOf(_) => panic!("internal error: can not call `matches` on a `AtomicTypeFilter::AllOf`"),
28        }
29    }
30}

32/// Abstraction over the different kinds of samples used in featomic.
33///
34/// Different implementations of this trait correspond to different types of
35/// samples (for example one sample for each system; or one sample for each
36/// pair, etc.)
37///
38/// Each implementation must be able to generate samples from a list of systems,
39/// and can optionally implement samples for gradients calculation.
40pub trait SamplesBuilder {
41    /// Get the names used by the samples
42    fn sample_names() -> Vec<&'static str>;
43
44    /// Create `Labels` containing all the samples corresponding to the given
45    /// list of systems.
46    fn samples(&self, systems: &mut [Box<dyn System>]) -> Result<Labels, Error>;
47
48    /// Create a set of `Labels` containing the gradient samples corresponding
49    /// to the given `samples` in the given `systems`; and only these.
50    #[allow(unused_variables)]
51    fn gradients_for(&self, systems: &mut [Box<dyn System>], samples: &Labels) -> Result<Labels, Error>;
52}
