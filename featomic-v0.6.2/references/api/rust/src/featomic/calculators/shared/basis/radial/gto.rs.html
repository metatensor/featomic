2
3use crate::math::gamma;

5/// Use a radial basis similar to Gaussian-Type Orbitals.
6///
7/// The basis is defined as `R_n(r) ∝ r^n e^{- r^2 / (2 σ_n^2)}`, where `σ_n
8/// = cutoff * \sqrt{n} / n_max`
9#[derive(Debug, Clone, Copy)]
10pub struct GtoRadialBasis {
11    pub size: usize,
12    pub radius: f64,
13}
14
15impl GtoRadialBasis {
16    /// Get the overlap matrix between non-orthonormalized GTO basis function
17    pub fn overlap(&self) -> Array2<f64> {
18        let gaussian_widths = self.gaussian_widths();
19
20        let mut overlap = Array2::from_elem((self.size, self.size), 0.0);
21        for n1 in 0..self.size {
22            let sigma1 = gaussian_widths[n1];
23            let sigma1_sq = sigma1 * sigma1;
24            for n2 in n1..self.size {
25                let sigma2 = gaussian_widths[n2];
26                let sigma2_sq = sigma2 * sigma2;
27
28                let n1_n2_3_over_2 = 0.5 * (3.0 + n1 as f64 + n2 as f64);
29                let value =
30                    (0.5 / sigma1_sq + 0.5 / sigma2_sq).powf(-n1_n2_3_over_2)
31                    / (sigma1.powi(n1 as i32) * sigma2.powi(n2 as i32))
32                    * gamma(n1_n2_3_over_2)
33                    / ((sigma1 * sigma2).powf(1.5) * f64::sqrt(gamma(n1 as f64 + 1.5) * gamma(n2 as f64 + 1.5)));
34
35
36                overlap[(n2, n1)] = value;
37                overlap[(n1, n2)] = value;
38            }
39        }
40        return overlap;
41    }
42
43    /// Get the vector of GTO Gaussian width, i.e. `cutoff * max(√n, 1) / n_max`
44    pub fn gaussian_widths(&self) -> Vec<f64> {
45        return (0..self.size).map(|n| {
46            let n = n as f64;
47            let n_max = self.size as f64;
48            self.radius * f64::max(f64::sqrt(n), 1.0) / n_max
49        }).collect();
50    }
51
52    /// Get the matrix to orthonormalize the GTO basis
53    pub fn orthonormalization_matrix(&self) -> Array2<f64> {
54        let normalization = self.gaussian_widths().iter()
55            .zip(0..self.size)
56            .map(|(sigma, n)| f64::sqrt(2.0 / (sigma.powi(2 * n as i32 + 3) * gamma(n as f64 + 1.5))))
57            .collect::<Array1<_>>();
58
59        let overlap = self.overlap();
60        // compute overlap^-1/2 through its eigendecomposition
61        let mut eigen = crate::math::SymmetricEigen::new(overlap);
62        for n in 0..self.size {
63            if eigen.eigenvalues[n] <= f64::EPSILON {
64                panic!(
65                    "radial overlap matrix is singular, try with a lower \
66                    max_radial (current value is {})", self.size - 1
67                );
68            }
69            eigen.eigenvalues[n] = 1.0 / f64::sqrt(eigen.eigenvalues[n]);
70        }
71
72        let orthonormalization = eigen.recompose().dot(&Array2::from_diag(&normalization));
73
74        return orthonormalization;
75    }
76}

79#[cfg(test)]
80mod tests {
81    use approx::assert_ulps_eq;
82    use super::*;
83
84    #[test]
85    fn gto_overlap() {
86        // some basic sanity checks on the overlap matrix
87        let basis = GtoRadialBasis {
88            size: 8,
89            radius: 6.3,
90        };
91
92        let overlap = basis.overlap();
93
94        for i in 0..basis.size {
95            for j in 0..basis.size {
96                if i == j {
97                    assert_ulps_eq!(overlap[(i, j)], 1.0, max_ulps=10);
98                } else {
99                    assert!(overlap[(i, j)] > 0.0);
100                    assert!(overlap[(i, j)] < 1.0);
101                }
102            }
103        }
104    }
