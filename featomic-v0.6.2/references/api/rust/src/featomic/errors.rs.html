3#[non_exhaustive]
4#[derive(Debug)]
5pub enum Error {
6    /// Got an invalid parameter value in a function
7    InvalidParameter(String),
8    /// Error while serializing/deserializing data
9    Json(serde_json::Error),
10    /// Error due to C strings containing non-utf8 data
11    Utf8(Utf8Error),
12    /// Errors coming from metatensor
13    Metatensor(metatensor::Error),
14    /// Errors coming from external callbacks, typically inside the System
15    /// implementation
16    External {
17        status: i32,
18        message: String
19    },
20    /// Error used when a memory buffer is too small to fit the requested data,
21    /// usually in the C API.
22    BufferSize(String),
23    /// Error used for failed internal consistency check and panics, i.e. bugs
24    /// in featomic.
25    Internal(String),
26}
27
28impl std::fmt::Display for Error {
29    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
30        match self {
31            Error::InvalidParameter(e) => write!(f, "invalid parameter: {}", e),
32            Error::Json(e) => write!(f, "json error: {}", e),
33            Error::Utf8(e) => write!(f, "utf8 decoding error: {}", e),
34            Error::Metatensor(e) => write!(f, "metatensor error: {}", e),
35            Error::BufferSize(e) => write!(f, "buffer is not big enough: {}", e),
36            Error::External{status, message} => write!(f, "error from external code (status {}): {}", status, message),
37            Error::Internal(e) => {
38                write!(f, "internal featomic error")?;
39                if e.contains("assertion failed") {
40                    write!(f, " (this is likely a bug, please report it)")?;
41                }
42                write!(f, ": {}", e)
43            }
44        }
45    }
46}

48impl std::error::Error for Error {
49    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
50        match self {
51            Error::InvalidParameter(_) |
52            Error::Internal(_) |
53            Error::BufferSize(_) |
54            Error::External{..} => None,
55            Error::Metatensor(e) => Some(e),
56            Error::Json(e) => Some(e),
57            Error::Utf8(e) => Some(e),
58        }
59    }
60}

81// Box<dyn Any + Send + 'static> is the error type in std::panic::catch_unwind
82impl From<Box<dyn std::any::Any + Send + 'static>> for Error {
83    fn from(error: Box<dyn std::any::Any + Send + 'static>) -> Error {
84        let message = if let Some(message) = error.downcast_ref::<String>() {
85            message.clone()
86        } else if let Some(message) = error.downcast_ref::<&str>() {
87            (*message).to_owned()
88        } else {
89            panic!("panic message is not a string, something is very wrong")
90        };
91
92        Error::Internal(message)
93    }
