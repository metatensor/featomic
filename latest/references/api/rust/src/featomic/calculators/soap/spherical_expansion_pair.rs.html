21/// Parameters for spherical expansion calculator.
22///
23/// The spherical expansion is at the core of representations in the SOAP
24/// (Smooth Overlap of Atomic Positions) family. The core idea is to define
25/// atom-centered environments using a spherical cutoff; create an atomic
26/// density according to all neighbors in a given environment; and finally
27/// expand this density on a given set of basis functions. The parameters for
28/// each of these steps can be defined separately below.
29///
30/// See [this review article](https://doi.org/10.1063/1.5090481) for more
31/// information on the SOAP representation, and [this
32/// paper](https://doi.org/10.1063/5.0044689) for information on how it is
33/// implemented in featomic.
34#[derive(Debug, Clone)]
35#[derive(serde::Deserialize, serde::Serialize, schemars::JsonSchema)]
36#[serde(deny_unknown_fields)]
37pub struct SphericalExpansionParameters {
38    /// Definition of the atomic environment within a cutoff, and how
39    /// neighboring atoms enter and leave the environment.
40    pub cutoff: Cutoff,
41    /// Definition of the density arising from atoms in the local environment.
42    pub density: Density,
43    /// Definition of the basis functions used to expand the atomic density
44    pub basis: SphericalExpansionBasis<SoapRadialBasis>,
45}
46
47impl SphericalExpansionParameters {
48    /// Validate all the parameters
49    pub fn validate(&mut self) -> Result<(), Error> {
50        self.cutoff.validate()?;
51        if let Some(scaling) = self.density.scaling {
52            scaling.validate()?;
53        }
54
55        // try constructing a radial integral cache to catch any errors early
56        SoapRadialIntegralCacheByAngular::new(self.cutoff.radius, self.density.kind, &self.basis)?;
57
58        return Ok(());
59    }
60}

97/// Contribution of a single pair to the spherical expansion
98pub(super) struct PairContribution {
99    /// Values of the contribution. The `BTreeMap` contains one array for each
100    /// angular channel, and the shape of the arrays is (2 * L + 1, N)
101    pub values: BTreeMap<usize, ndarray::Array2<f64>>,
102    /// Gradients of the contribution w.r.t. the distance between the atoms in
103    /// the pair. shape of the arrays is (3, 2 * L + 1, N)
104    pub gradients: Option<BTreeMap<usize, ndarray::Array3<f64>>>,
105}
106
107impl PairContribution {
108    pub fn new(angular_channels: &[usize], radial_sizes: &[usize], do_gradients: bool) -> PairContribution {
109        let values = angular_channels.iter().zip(radial_sizes).map(|(&o3_lambda, &radial_size)| {
110            let array = ndarray::Array2::from_elem((2 * o3_lambda + 1, radial_size), 0.0);
111            (o3_lambda, array)
112        }).collect();
113
114        let gradients = if do_gradients {
115            Some(angular_channels.iter().zip(radial_sizes).map(|(&o3_lambda, &radial_size)| {
116                let array = ndarray::Array3::from_elem((3, 2 * o3_lambda + 1, radial_size), 0.0);
117                (o3_lambda, array)
118            }).collect())
119        } else {
120            None
121        };
122
123        return PairContribution { values, gradients }
124    }
125
126    /// Modify the values/gradients as required to construct the
127    /// values/gradients associated with pair j -> i from pair i -> j.
128    ///
129    /// `m_1_pow_l` should contain the values of `(-1)^l` up to `max_angular`
130    pub fn inverse_pair(&mut self, m_1_pow_l: &[f64]) {
131        // inverting the pair is equivalent to adding a (-1)^l factor to the
132        // pair contribution values, and -(-1)^l to the gradients
133        for (&o3_lambda, values) in &mut self.values {
134            let shape = values.shape();
135            debug_assert_eq!(shape[0], 2 * o3_lambda + 1);
136            let shape_n = shape[1];
137
138            let factor = m_1_pow_l[o3_lambda];
139            for m in 0..2 * o3_lambda + 1 {
140                for n in 0..shape_n {
141                    values[[m, n]] *= factor;
142                }
143            }
144        }
145
146        if let Some(ref mut gradients) = self.gradients {
147            for (&o3_lambda, gradients) in gradients.iter_mut() {
148                let shape = gradients.shape();
149                debug_assert_eq!(shape[0], 3);
150                debug_assert_eq!(shape[1], 2 * o3_lambda + 1);
151                let shape_n = shape[2];
152
153                let factor = -m_1_pow_l[o3_lambda];
154                for xyz in 0..3 {
155                    for m in 0..(2 * o3_lambda + 1) {
156                        for n in 0..shape_n {
157                            gradients[[xyz, m, n]] *= factor;
158                        }
159                    }
160                }
161            }
162        }
163    }
164}

167impl SphericalExpansionByPair {
168    pub fn new(mut parameters: SphericalExpansionParameters) -> Result<SphericalExpansionByPair, Error> {
169        parameters.validate()?;
170
171        let max_angular = parameters.basis.angular_channels().into_iter().max().expect("there should be at least one angular channel");
172        let m_1_pow_l = (0..=max_angular).map(|l| f64::powi(-1.0, l as i32))
173            .collect::<Vec<f64>>();
174
175        Ok(SphericalExpansionByPair {
176            parameters: parameters,
177            radial_integral: ThreadLocal::new(),
178            spherical_harmonics: ThreadLocal::new(),
179            m_1_pow_l: m_1_pow_l,
180        })
181    }
182
183    /// Access the spherical expansion parameters used by this calculator
184    pub fn parameters(&self) -> &SphericalExpansionParameters {
185        &self.parameters
186    }
187
188    /// Compute the product of radial scaling & cutoff smoothing functions
189    fn scaling_functions(&self, r: f64) -> f64 {
190        let mut scaling = 1.0;
191        if let Some(scaler) = self.parameters.density.scaling {
192            scaling = scaler.compute(r);
193        }
194        return scaling * self.parameters.cutoff.smoothing(r);
195    }
196
197    /// Compute the gradient of the product of radial scaling & cutoff smoothing functions
198    fn scaling_functions_gradient(&self, r: f64) -> f64 {
199        let mut scaling = 1.0;
200        let mut scaling_grad = 0.0;
201        if let Some(scaler) = self.parameters.density.scaling {
202            scaling = scaler.compute(r);
203            scaling_grad = scaler.gradient(r);
204        }
205
206        let cutoff = self.parameters.cutoff.smoothing(r);
207        let cutoff_grad = self.parameters.cutoff.smoothing_gradient(r);
208
209        return cutoff_grad * scaling + cutoff * scaling_grad;
210    }
211
212    /// Compute the self-contribution (contribution coming from an atom "seeing"
213    /// it's own density). This is equivalent to a normal pair contribution,
214    /// with a distance of 0.
215    ///
216    /// For now, the same density is used for all atoms, so this function can be
217    /// called only once and re-used for all atoms (see `do_self_contributions`
218    /// below).
219    ///
220    /// By symmetry, the self-contribution is only non-zero for `L=0`, and does
221    /// not contributes to the gradients.
222    pub(super) fn self_contribution(&self) -> ndarray::Array1<f64> {
223        let mut radial_integral = self.radial_integral.get_or(|| {
224            RefCell::new(SoapRadialIntegralCacheByAngular::new(
225                    self.parameters.cutoff.radius,
226                    self.parameters.density.kind,
227                    &self.parameters.basis
228                ).expect("invalid radial integral parameters")
229            )
230        }).borrow_mut();
231
232        let mut spherical_harmonics = self.spherical_harmonics.get_or(|| {
233            let max_angular = self.parameters.basis.angular_channels().into_iter().max().unwrap_or(0);
234            RefCell::new(SphericalHarmonicsCache::new(max_angular))
235        }).borrow_mut();
236
237        // Compute the three factors that appear in the center contribution.
238        // Note that this is simply the pair contribution for the special
239        // case where the pair distance is zero.
240        let radial_integral = radial_integral.get_mut(0)
241            .expect("self_contribution can't be done when o3_lambda=0 is missing");
242        radial_integral.compute(0.0, false);
243
244        spherical_harmonics.compute(Vector3D::new(0.0, 0.0, 1.0), false);
245        let f_scaling = self.scaling_functions(0.0);
246
247        let factor = self.parameters.density.center_atom_weight
248            * f_scaling
249            * spherical_harmonics.values[[0, 0]];
250
251        return factor * radial_integral.values.clone();
252    }
253
254    /// Accumulate the self contribution to the spherical expansion
255    /// coefficients.
256    ///
257    /// For the pair-by-pair spherical expansion, we use a special `pair_id`
258    /// (-1) to store the data associated with self-pairs.
259    fn do_self_contributions(&self, systems: &[Box<dyn System>], descriptor: &mut TensorMap) -> Result<(), Error> {
260        debug_assert_eq!(descriptor.keys().names(), ["o3_lambda", "o3_sigma", "first_atom_type", "second_atom_type"]);
261
262        if !self.parameters.basis.angular_channels().contains(&0) {
263            // o3_lambda is not part of the output, skip self contributions
264            return Ok(());
265        }
266
267        let self_contribution = self.self_contribution();
268
269        for (key, mut block) in descriptor {
270            let o3_lambda = key[0];
271            let first_atom_type = key[2];
272            let second_atom_type = key[3];
273
274            if o3_lambda != 0 || first_atom_type != second_atom_type {
275                // center contribution is non-zero only for l=0
276                continue;
277            }
278
279            let data = block.data_mut();
280            let array = data.values.to_array_mut();
281
282            // loop over all samples in this block, find self pairs (`i == j`
283            // and `shift == [0, 0, 0]`), and fill the data using
284            // `self_contribution`
285            for (sample_i, &[system, atom_1, atom_2, cell_a, cell_b, cell_c]) in data.samples.iter_fixed_size().enumerate() {
286                // it is possible that the samples from values.samples are not
287                // part of the systems (the user requested extra samples). In
288                // that case, we need to skip anything that does not exist, or
289                // with a different center atomic types
290                let is_self_pair = atom_1 == atom_2 && cell_a == 0 && cell_b == 0 && cell_c == 0;
291                if system.usize() >= systems.len() || !is_self_pair {
292                    continue;
293                }
294
295                let system = &systems[system.usize()];
296                let n_atoms = system.size()?;
297                let types = system.types()?;
298
299                if atom_1.usize() > n_atoms || atom_2.usize() > n_atoms {
300                    continue;
301                }
302
303                if types[atom_1.usize()] != first_atom_type || types[atom_2.usize()] != second_atom_type {
304                    continue;
305                }
306
307                for (property_i, &[n]) in data.properties.iter_fixed_size().enumerate() {
308                    array[[sample_i, 0, property_i]] = self_contribution[n.usize()];
309                }
310            }
311        }
312
313        return Ok(());
314    }

316    /// Compute the contribution of a single pair and store the corresponding
317    /// data inside the given descriptor.
318    ///
319    /// This will store data both for the spherical expansion with `pair.first`
320    /// as the central atom and `pair.second` as the neighbor, and for the
321    /// spherical expansion with `pair.second` as the central atom and
322    /// `pair.first` as the neighbor.
323    pub(super) fn compute_for_pair(
324        &self,
325        distance: f64,
326        mut direction: Vector3D,
327        do_gradients: GradientsOptions,
328        contribution: &mut PairContribution,
329    ) {
330        debug_assert!(distance >= 0.0);
331
332        // Deal with the possibility that two atoms are at the same
333        // position. While this is not usual, there is no reason to
334        // prevent the calculation of spherical expansion. The user will
335        // still get a warning about atoms being very close together
336        // when calculating the neighbor list.
337        if distance < 1e-6 {
338            direction = Vector3D::new(0.0, 0.0, 1.0);
339        }
340
341        let mut radial_integral = self.radial_integral.get_or(|| {
342            RefCell::new(SoapRadialIntegralCacheByAngular::new(
343                    self.parameters.cutoff.radius,
344                    self.parameters.density.kind,
345                    &self.parameters.basis
346                ).expect("invalid radial integral parameters")
347            )
348        }).borrow_mut();
349
350        let mut spherical_harmonics = self.spherical_harmonics.get_or(|| {
351            let max_angular = self.parameters.basis.angular_channels().into_iter().max().unwrap_or(0);
352            RefCell::new(SphericalHarmonicsCache::new(max_angular))
353        }).borrow_mut();
354
355        radial_integral.compute(distance, do_gradients.any());
356        spherical_harmonics.compute(direction, do_gradients.any());
357
358        let f_scaling = self.scaling_functions(distance);
359        let f_scaling_grad = self.scaling_functions_gradient(distance);
360
361        for o3_lambda in self.parameters.basis.angular_channels() {
362            let radial_basis_size = match self.parameters.basis {
363                SphericalExpansionBasis::TensorProduct(ref basis) => basis.radial.size(),
364                SphericalExpansionBasis::Explicit(ref basis) => {
365                    basis.by_angular.get(&o3_lambda).expect("missing o3_lambda").size()
366                },
367            };
368
369            let spherical_harmonics_grad = [
370                spherical_harmonics.gradients[0].angular_slice(o3_lambda),
371                spherical_harmonics.gradients[1].angular_slice(o3_lambda),
372                spherical_harmonics.gradients[2].angular_slice(o3_lambda),
373            ];
374            let spherical_harmonics = spherical_harmonics.values.angular_slice(o3_lambda);
375
376            let radial_integral_grad = &radial_integral.get(o3_lambda).expect("missing o3_lambda").gradients;
377            let radial_integral = &radial_integral.get(o3_lambda).expect("missing o3_lambda").values;
378
379            let values = contribution.values.get_mut(&o3_lambda).expect("missing o3_lambda");
380
381            // compute the full spherical expansion coefficients & gradients
382            for m in 0..(2 * o3_lambda + 1) {
383                let sph_value = spherical_harmonics[m];
384                for (n, ri_value) in radial_integral.iter().enumerate() {
385                    values[[m, n]] = f_scaling * sph_value * ri_value;
386                }
387            }
388
389            if let Some(ref mut gradients) = contribution.gradients {
390                let gradients = gradients.get_mut(&o3_lambda).expect("missing o3_lambda");
391
392                let dr_d_spatial = direction;
393
394                for m in 0..(2 * o3_lambda + 1) {
395                    let sph_value = spherical_harmonics[m];
396                    let sph_grad_x = spherical_harmonics_grad[0][m];
397                    let sph_grad_y = spherical_harmonics_grad[1][m];
398                    let sph_grad_z = spherical_harmonics_grad[2][m];
399
400                    for n in 0..radial_basis_size {
401                        let ri_value = radial_integral[n];
402                        let ri_grad = radial_integral_grad[n];
403
404                        gradients[[0, m, n]] =
405                            f_scaling_grad * dr_d_spatial[0] * ri_value * sph_value
406                            + f_scaling * ri_grad * dr_d_spatial[0] * sph_value
407                            + f_scaling * ri_value * sph_grad_x / distance;
408
409                        gradients[[1, m, n]] =
410                            f_scaling_grad * dr_d_spatial[1] * ri_value * sph_value
411                            + f_scaling * ri_grad * dr_d_spatial[1] * sph_value
412                            + f_scaling * ri_value * sph_grad_y / distance;
413
414                        gradients[[2, m, n]] =
415                            f_scaling_grad * dr_d_spatial[2] * ri_value * sph_value
416                            + f_scaling * ri_grad * dr_d_spatial[2] * sph_value
417                            + f_scaling * ri_value * sph_grad_z / distance;
418                    }
419                }
420            }
421        }
422    }

424    /// Accumulate a single pair `contribution` in the right block.
425    fn accumulate_in_block(
426        o3_lambda: usize,
427        mut block: TensorBlockRefMut,
428        sample: &[LabelValue],
429        contributions: &PairContribution,
430        do_gradients: GradientsOptions,
431        pair_vector: Vector3D,
432    ) {
433        let data = block.data_mut();
434        let array = data.values.to_array_mut();
435
436        let contribution_values = contributions.values.get(&o3_lambda).expect("missing o3_lambda");
437        let sample_i = data.samples.position(sample);
438        if let Some(sample_i) = sample_i {
439            for m in 0..(2 * o3_lambda + 1) {
440                for (property_i, [n]) in data.properties.iter_fixed_size().enumerate() {
441                    unsafe {
442                        let out = array.uget_mut([sample_i, m, property_i]);
443                        *out += *contribution_values.uget([m, n.usize()]);
444                    }
445                }
446            }
447
448            if let Some(ref contribution_gradients) = contributions.gradients {
449                let contribution_gradients = contribution_gradients.get(&o3_lambda).expect("missing o3_lambda");
450                if do_gradients.positions {
451                    let mut gradient = block.gradient_mut("positions").expect("missing positions gradients");
452                    let gradient = gradient.data_mut();
453
454                    let array = gradient.values.to_array_mut();
455                    debug_assert_eq!(gradient.samples.names(), ["sample", "system", "atom"]);
456
457                    // gradient of the pair contribution w.r.t. the position of
458                    // the first atom
459                    let first_grad_sample_i = gradient.samples.position(&[
460                        sample_i.into(), /* system */ sample[0], /* pair.first */ sample[1]
461                    ]).expect("missing first gradient sample");
462
463                    for xyz in 0..3 {
464                        for m in 0..(2 * o3_lambda + 1) {
465                            for (property_i, [n]) in gradient.properties.iter_fixed_size().enumerate() {
466                                unsafe {
467                                    let out = array.uget_mut([first_grad_sample_i, xyz, m, property_i]);
468                                    *out -= contribution_gradients.uget([xyz, m, n.usize()]);
469                                }
470                            }
471                        }
472                    }
473
474                    // gradient of the pair contribution w.r.t. the position of
475                    // the second atom
476                    let second_grad_sample_i = gradient.samples.position(&[
477                        sample_i.into(), /* system */ sample[0], /* pair.second */ sample[2]
478                    ]).expect("missing second gradient sample");
479
480                    for xyz in 0..3 {
481                        for m in 0..(2 * o3_lambda + 1) {
482                            for (property_i, [n]) in gradient.properties.iter_fixed_size().enumerate() {
483                                unsafe {
484                                    let out = array.uget_mut([second_grad_sample_i, xyz, m, property_i]);
485                                    *out += contribution_gradients.uget([xyz, m, n.usize()]);
486                                }
487                            }
488                        }
489                    }
490                }
491
492                if do_gradients.strain {
493                    let mut gradient = block.gradient_mut("strain").expect("missing strain gradients");
494                    let gradient = gradient.data_mut();
495
496                    debug_assert_eq!(gradient.samples.names(), ["sample"]);
497                    assert_eq!(gradient.samples[sample_i][0].usize(), sample_i);
498
499                    let array = gradient.values.to_array_mut();
500                    for xyz_1 in 0..3 {
501                        for xyz_2 in 0..3 {
502                            for m in 0..(2 * o3_lambda + 1) {
503                                for (property_i, [n]) in gradient.properties.iter_fixed_size().enumerate() {
504                                    unsafe {
505                                        let out = array.uget_mut([sample_i, xyz_1, xyz_2, m, property_i]);
506                                        *out += pair_vector[xyz_1] * contribution_gradients.uget([xyz_2, m, n.usize()]);
507                                    }
508                                }
509                            }
510                        }
511                    }
512                }
513
514                if do_gradients.cell {
515                    let mut gradient = block.gradient_mut("cell").expect("missing cell gradients");
516                    let gradient = gradient.data_mut();
517
518                    debug_assert_eq!(gradient.samples.names(), ["sample"]);
519                    assert_eq!(gradient.samples[sample_i][0].usize(), sample_i);
520
521                    let shifts = [
522                        sample[3].i32() as f64,
523                        sample[4].i32() as f64,
524                        sample[5].i32() as f64,
525                    ];
526
527                    let array = gradient.values.to_array_mut();
528                    for abc in 0..3 {
529                        for xyz in 0..3 {
530                            for m in 0..(2 * o3_lambda + 1) {
531                                for (property_i, [n]) in gradient.properties.iter_fixed_size().enumerate() {
532                                    unsafe {
533                                        let out = array.uget_mut([sample_i, abc, xyz, m, property_i]);
534                                        *out += shifts[abc] * contribution_gradients.uget([xyz, m, n.usize()]);
535                                    }
536                                }
537                            }
538                        }
539                    }
540                }
541            }
542        }
543    }
544}

547impl CalculatorBase for SphericalExpansionByPair {
548    fn name(&self) -> String {
549        "spherical expansion by pair".into()
550    }
551
552    fn parameters(&self) -> String {
553        serde_json::to_string(&self.parameters).expect("failed to serialize to JSON")
554    }
555
556    fn cutoffs(&self) -> &[f64] {
557        std::slice::from_ref(&self.parameters.cutoff.radius)
558    }
559
560    fn keys(&self, systems: &mut [Box<dyn System>]) -> Result<Labels, Error> {
561        // the atomic type part of the keys is the same for all l, and the same
562        // as what a FullNeighborList with `self_pairs=True` produces.
563        let full_neighbors_list_keys = FullNeighborList {
564            cutoff: self.parameters.cutoff.radius,
565            self_pairs: true,
566        }.keys(systems)?;
567
568        let mut keys = LabelsBuilder::new(vec![
569            "o3_lambda",
570            "o3_sigma",
571            "first_atom_type",
572            "second_atom_type"
573        ]);
574
575        for &[first_type, second_type] in full_neighbors_list_keys.iter_fixed_size() {
576            for o3_lambda in self.parameters.basis.angular_channels() {
577                keys.add(&[o3_lambda.into(), 1.into(), first_type, second_type]);
578            }
579        }
580
581        return Ok(keys.finish_assume_unique());
582    }
583
584    fn sample_names(&self) -> Vec<&str> {
585        return vec!["system", "first_atom", "second_atom", "cell_shift_a", "cell_shift_b", "cell_shift_c"];
586    }
587
588    fn samples(&self, keys: &Labels, systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error> {
589        // get all atomic types pairs in keys as a new set of Labels
590        let mut types_keys = BTreeSet::new();
591        for &[_, _, first_type, second_type] in keys.iter_fixed_size() {
592            types_keys.insert((first_type, second_type));
593        }
594        let mut builder = LabelsBuilder::new(vec!["first_atom_type", "second_atom_type"]);
595        for (first_type, second_type) in types_keys {
596            builder.add(&[first_type, second_type]);
597        }
598        let types_keys = builder.finish_assume_unique();
599
600        // for l=0, we want to include self pairs in the samples
601        let mut samples_by_types_l0: BTreeMap<_, Labels> = BTreeMap::new();
602        let full_neighbors_list_samples = FullNeighborList {
603            cutoff: self.parameters.cutoff.radius,
604            self_pairs: true,
605        }.samples(&types_keys, systems)?;
606
607        debug_assert_eq!(types_keys.count(), full_neighbors_list_samples.len());
608        for (&[first_type, second_type], samples) in types_keys.iter_fixed_size().zip(full_neighbors_list_samples) {
609            samples_by_types_l0.insert((first_type, second_type), samples);
610        }
611
612        // we only need to compute samples once for each l>0, so we compute them
613        // using FullNeighborList::samples, store them in a (center_type,
614        // neighbor_type) => Labels map and then re-use them from this map as
615        // needed.
616        let mut samples_by_types: BTreeMap<_, Labels> = BTreeMap::new();
617        let full_neighbors_list_samples = FullNeighborList {
618            cutoff: self.parameters.cutoff.radius,
619            self_pairs: false,
620        }.samples(&types_keys, systems)?;
621
622        debug_assert_eq!(types_keys.count(), full_neighbors_list_samples.len());
623        for (&[first_type, second_type], samples) in types_keys.iter_fixed_size().zip(full_neighbors_list_samples) {
624            samples_by_types.insert((first_type, second_type), samples);
625        }
626
627        let mut result = Vec::new();
628        for &[l, _, first_type, second_type] in keys.iter_fixed_size() {
629            let samples = if l.i32() == 0 {
630                samples_by_types_l0.get(&(first_type, second_type)).expect("missing samples for one pair of types")
631            } else {
632                samples_by_types.get(&(first_type, second_type)).expect("missing samples for one pair of types")
633            };
634
635            result.push(samples.clone());
636        }
637
638        return Ok(result);
639    }

648    fn positions_gradient_samples(&self, _: &Labels, samples: &[Labels], _: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error> {
649        let mut results = Vec::new();
650
651        for block_samples in samples {
652            let mut builder = LabelsBuilder::new(vec!["sample", "system", "atom"]);
653            for (sample_i, &[system_i, first, second, cell_a, cell_b, cell_c]) in block_samples.iter_fixed_size().enumerate() {
654                // self pairs do not contribute to gradients
655                if first == second && cell_a == 0 && cell_b == 0 && cell_c == 0 {
656                    continue;
657                }
658                builder.add(&[sample_i.into(), system_i, first]);
659                if first != second {
660                    builder.add(&[sample_i.into(), system_i, second]);
661                }
662            }
663
664            results.push(builder.finish_assume_unique());
665        }
666
667        return Ok(results);
668    }

670    fn components(&self, keys: &Labels) -> Vec<Vec<Labels>> {
671        assert_eq!(keys.names().len(), 4);
672        assert_eq!(keys.names()[0], "o3_lambda");
673
674        let mut result = Vec::new();
675        // only compute the components once for each `o3_lambda`,
676        // and re-use the results across the other keys.
677        let mut cache: BTreeMap<_, Vec<Labels>> = BTreeMap::new();
678        for &[o3_lambda, _, _, _] in keys.iter_fixed_size() {
679            let components = match cache.entry(o3_lambda) {
680                Entry::Occupied(entry) => entry.get().clone(),
681                Entry::Vacant(entry) => {
682                    let mut component = LabelsBuilder::new(vec!["o3_mu"]);
683                    for m in -o3_lambda.i32()..=o3_lambda.i32() {
684                        component.add(&[LabelValue::new(m)]);
685                    }
686
687                    let components = vec![component.finish_assume_unique()];
688                    entry.insert(components).clone()
689                }
690            };
691
692            result.push(components);
693        }
694
695        return result;
696    }

702    fn properties(&self, keys: &Labels) -> Vec<Labels> {
703        assert_eq!(keys.names(), ["o3_lambda", "o3_sigma", "first_atom_type", "second_atom_type"]);
704
705        match self.parameters.basis {
706            SphericalExpansionBasis::TensorProduct(ref basis) => {
707                let mut properties = LabelsBuilder::new(self.property_names());
708                for n in 0..basis.radial.size() {
709                    properties.add(&[n]);
710                }
711
712                return vec![properties.finish_assume_unique(); keys.count()];
713            }
714            SphericalExpansionBasis::Explicit(ref basis) => {
715                let mut result = Vec::new();
716                for [o3_lambda, _, _, _] in keys.iter_fixed_size() {
717                    let mut properties = LabelsBuilder::new(self.property_names());
718
719                    let radial = basis.by_angular.get(&o3_lambda.usize()).expect("missing o3_lambda");
720                    for n in 0..radial.size() {
721                        properties.add(&[n]);
722                    }
723
724                    result.push(properties.finish_assume_unique());
725                }
726                return result;
727            }
728        }

731    #[time_graph::instrument(name = "SphericalExpansionByPair::compute")]
732    fn compute(&mut self, systems: &mut [Box<dyn System>], descriptor: &mut TensorMap) -> Result<(), Error> {
733        assert_eq!(descriptor.keys().names(), ["o3_lambda", "o3_sigma", "first_atom_type", "second_atom_type"]);
734        assert!(descriptor.keys().count() > 0);
735
736        let do_gradients = GradientsOptions {
737            positions: descriptor.block_by_id(0).gradient("positions").is_some(),
738            strain: descriptor.block_by_id(0).gradient("strain").is_some(),
739            cell: descriptor.block_by_id(0).gradient("cell").is_some(),
740        };
741
742        self.do_self_contributions(systems, descriptor)?;
743
744        let keys = descriptor.keys().clone();
745
746        let radial_sizes = match self.parameters.basis {
747            SphericalExpansionBasis::TensorProduct(ref basis) => {
748                vec![basis.radial.size(); basis.max_angular + 1]
749            },
750            SphericalExpansionBasis::Explicit(ref basis) => {
751                basis.by_angular.values().map(|radial| radial.size()).collect()
752            },
753        };
754
755        let mut contribution = PairContribution::new(
756            &self.parameters.basis.angular_channels(),
757            &radial_sizes,
758            do_gradients.any(),
759        );
760
761        for (system_i, system) in systems.iter_mut().enumerate() {
762            system.compute_neighbors(self.parameters.cutoff.radius)?;
763            let types = system.types()?;
764
765            for pair in system.pairs()? {
766                let direction = pair.vector / pair.distance;
767                self.compute_for_pair(pair.distance, direction, do_gradients, &mut contribution);
768
769                let cell_shift_a = pair.cell_shift_indices[0];
770                let cell_shift_b = pair.cell_shift_indices[1];
771                let cell_shift_c = pair.cell_shift_indices[2];
772
773                let first_type = types[pair.first];
774                let second_type = types[pair.second];
775                for o3_lambda in self.parameters.basis.angular_channels() {
776                    let block_i = keys.position(&[
777                        o3_lambda.into(),
778                        1.into(),
779                        first_type.into(),
780                        second_type.into(),
781                    ]);
782
783                    if let Some(block_i) = block_i {
784                        let sample = &[
785                            LabelValue::from(system_i),
786                            LabelValue::from(pair.first),
787                            LabelValue::from(pair.second),
788                            LabelValue::from(cell_shift_a),
789                            LabelValue::from(cell_shift_b),
790                            LabelValue::from(cell_shift_c),
791                        ];
792
793                        SphericalExpansionByPair::accumulate_in_block(
794                            o3_lambda,
795                            descriptor.block_mut_by_id(block_i),
796                            sample,
797                            &contribution,
798                            do_gradients,
799                            pair.vector,
800                        );
801                    }
802                }
803
804                // also check for the block with a reversed pair
805                contribution.inverse_pair(&self.m_1_pow_l);
806
807                for o3_lambda in self.parameters.basis.angular_channels() {
808                    let block_i = keys.position(&[
809                        o3_lambda.into(),
810                        1.into(),
811                        second_type.into(),
812                        first_type.into(),
813                    ]);
814
815                    if let Some(block_i) = block_i {
816                        let sample = &[
817                            LabelValue::from(system_i),
818                            LabelValue::from(pair.second),
819                            LabelValue::from(pair.first),
820                            LabelValue::from(-cell_shift_a),
821                            LabelValue::from(-cell_shift_b),
822                            LabelValue::from(-cell_shift_c),
823                        ];
824
825                        SphericalExpansionByPair::accumulate_in_block(
826                            o3_lambda,
827                            descriptor.block_mut_by_id(block_i),
828                            sample,
829                            &contribution,
830                            do_gradients,
831                            -pair.vector,
832                        );
833                    }
834                }
835            }
836        }
837
838        Ok(())
839    }
