8/// The different kinds of radial basis supported by SOAP calculators
9#[derive(Debug, Clone)]
10#[derive(serde::Deserialize, serde::Serialize, schemars::JsonSchema)]
11#[serde(deny_unknown_fields)]
12#[serde(tag = "type")]
13pub enum SoapRadialBasis {
14    /// Use a radial basis similar to Gaussian-Type Orbitals.
15    ///
16    /// The basis is defined as `R_n(r) ∝ r^n e^{- r^2 / (2 σ_n^2)}`, where `σ_n
17    /// = cutoff * \sqrt{n} / n_max`
18    Gto {
19        /// Maximal value of `n` to include in the radial basis function
20        /// definition. The overall basis will have `max_radial + 1` basis
21        /// functions, indexed from `0` to `max_radial` (inclusive).
22        max_radial: usize,
23
24        #[doc(hidden)]
25        ///
26        radius: Option<f64>,
27    },
28    /// Use pre-tabulated radial basis.
29    ///
30    /// This enables the calculation of the overall radial integral using
31    /// user-defined splines.
32    ///
33    /// The easiest way to create such a tabulated basis is the corresponding
34    /// functions in featomic's Python API.
35    #[schemars(with = "tabulated::TabulatedSerde")]
36    Tabulated(Tabulated)
37}
38
39impl SoapRadialBasis {
40    /// Get the size (number of basis function) for the current basis
41    pub fn size(&self) -> usize {
42        match self {
43            SoapRadialBasis::Gto { max_radial, .. } => max_radial + 1,
44            SoapRadialBasis::Tabulated(tabulated) => tabulated.size(),
45        }
46    }
47}

50/// The different kinds of radial basis supported LODE calculators

57    ///
