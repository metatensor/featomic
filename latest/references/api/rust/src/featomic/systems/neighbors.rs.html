7/// Maximal number of cells, we need to use this to prevent having too many
8/// cells with a small unit cell and a large cutoff
9const MAX_NUMBER_OF_CELLS: f64 = 1e5;

11/// A cell shift represents the displacement along cell axis between the actual
12/// position of an atom and a periodic image of this atom.
13///
14/// The cell shift can be used to reconstruct the vector between two points,
15/// wrapped inside the unit cell.
16#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
17pub struct CellShift([i32; 3]);
18
19impl std::ops::Add<CellShift> for CellShift {
20    type Output = CellShift;
21
22    fn add(mut self, rhs: CellShift) -> Self::Output {
23        self.0[0] += rhs[0];
24        self.0[1] += rhs[1];
25        self.0[2] += rhs[2];
26        return self;
27    }
28}
29
30impl std::ops::Sub<CellShift> for CellShift {
31    type Output = CellShift;
32
33    fn sub(mut self, rhs: CellShift) -> Self::Output {
34        self.0[0] -= rhs[0];
35        self.0[1] -= rhs[1];
36        self.0[2] -= rhs[2];
37        return self;
38    }
39}

41impl std::ops::Index<usize> for CellShift {
42    type Output = i32;
43
44    fn index(&self, index: usize) -> &Self::Output {
45        &self.0[index]
46    }

49impl CellShift {
50    /// Compute the shift vector in cartesian coordinates, using the given cell
51    /// matrix (stored in row major order).
52    pub fn cartesian(&self, cell: &Matrix3) -> Vector3D {
53        let x = cell[0][0] * self[0] as f64 + cell[1][0] * self[1] as f64 + cell[2][0] * self[2] as f64;
54        let y = cell[0][1] * self[0] as f64 + cell[1][1] * self[1] as f64 + cell[2][1] * self[2] as f64;
55        let z = cell[0][2] * self[0] as f64 + cell[1][2] * self[1] as f64 + cell[2][2] * self[2] as f64;
56        Vector3D::new(x, y, z)
57    }
58}

60/// Pair produced by the cell list. The vector between the atoms can be
61/// constructed as `position[second] - position[first] + shift.cartesian(unit_cell)`
62#[derive(Debug, Clone)]
63pub struct CellPair {
64    /// index of the first atom in the pair
65    pub first: usize,
66    /// index of the second atom in the pair
67    pub second: usize,
68    /// number of shifts along the cell for this pair
69    pub shift: CellShift,

82/// The cell list is used to sort atoms inside bins/cells.
83///
84/// The list of potential pairs is then constructed by looking through all
85/// neighboring cells (the number of cells to search depends on the cutoff and
86/// the size of the cells) for each atom to create pair candidates.
87#[derive(Debug, Clone)]
88pub struct CellList {
89    /// How many cells do we need to look at when searching neighbors to include
90    /// all neighbors below cutoff
91    n_search: [i32; 3],
92    /// the cells themselves
93    cells: ndarray::Array3<Vec<AtomData>>,
94    /// Unit cell defining periodic boundary conditions
95    unit_cell: UnitCell,
96}
97
98impl CellList {
99    /// Create a new `CellList` for the given unit cell and cutoff, determining
100    /// all required parameters.
101    pub fn new(unit_cell: UnitCell, cutoff: f64) -> CellList {
102        let distances_between_faces = if unit_cell.is_infinite() {
103            // use a pseudo orthorhombic cell with size 1, `n_search` below will
104            // make sure we look to every cell up to the cutoff
105            Vector3D::new(1.0, 1.0, 1.0)
106        } else {
107            unit_cell.distances_between_faces()
108        };
109
110        let mut n_cells = [
111            f64::clamp(f64::trunc(distances_between_faces[0] / cutoff), 1.0, f64::INFINITY),
112            f64::clamp(f64::trunc(distances_between_faces[1] / cutoff), 1.0, f64::INFINITY),
113            f64::clamp(f64::trunc(distances_between_faces[2] / cutoff), 1.0, f64::INFINITY),
114        ];
115
116        assert!(n_cells[0].is_finite() && n_cells[1].is_finite() && n_cells[2].is_finite());
117
118        // limit memory consumption by ensuring we have less than `MAX_N_CELLS`
119        // cells to look though
120        let n_cells_total = n_cells[0] * n_cells[1] * n_cells[2];
121        if n_cells_total > MAX_NUMBER_OF_CELLS {
122            // set the total number of cells close to MAX_N_CELLS, while keeping
123            // roughly the ratio of cells in each direction
124            let ratio_x_y = n_cells[0] / n_cells[1];
125            let ratio_y_z = n_cells[1] / n_cells[2];
126
127            n_cells[2] = f64::trunc(f64::cbrt(MAX_NUMBER_OF_CELLS / (ratio_x_y * ratio_y_z * ratio_y_z)));
128            n_cells[1] = f64::trunc(ratio_y_z * n_cells[2]);
129            n_cells[0] = f64::trunc(ratio_x_y * n_cells[1]);
130        }
131
132        // number of cells to search in each direction to make sure all possible
133        // pairs below the cutoff are accounted for.
134        let mut n_search = [
135            f64::ceil(cutoff * n_cells[0] / distances_between_faces[0]) as i32,
136            f64::ceil(cutoff * n_cells[1] / distances_between_faces[1]) as i32,
137            f64::ceil(cutoff * n_cells[2] / distances_between_faces[2]) as i32,
138        ];
139
140        let n_cells = [
141            n_cells[0] as usize,
142            n_cells[1] as usize,
143            n_cells[2] as usize,
144        ];
145
146        for xyz in 0..3 {
147            if n_search[xyz] < 1 {
148                n_search[xyz] = 1;
149            }
150
151            // don't look for neighboring cells if we have only one cell and no
152            // periodic boundary condition
153            if n_cells[xyz] == 1 && unit_cell.is_infinite() {
154                n_search[xyz] = 0;
155            }
156        }
157
158        CellList {
159            n_search: n_search,
160            cells: Array3::from_elem(n_cells, Default::default()),
161            unit_cell: unit_cell,
162        }
163    }

165    /// Add a single atom to the cell list at the given `position`. The atom is
166    /// uniquely identified by its `index`.
167    pub fn add_atom(&mut self, index: usize, position: Vector3D) {
168        let fractional = if self.unit_cell.is_infinite() {
169            position
170        } else {
171            self.unit_cell.fractional(position)
172        };
173
174        let n_cells = self.cells.shape();
175        let n_cells = [n_cells[0], n_cells[1], n_cells[2]];
176
177        // find the subcell in which this atom 'should go'
178        let cell_index = [
179            f64::floor(fractional[0] * n_cells[0] as f64) as i32,
180            f64::floor(fractional[1] * n_cells[1] as f64) as i32,
181            f64::floor(fractional[2] * n_cells[2] as f64) as i32,
182        ];
183
184        // deal with pbc by wrapping the atom inside if it was outside of the
185        // cell
186        let (shift, cell_index) = if self.unit_cell.is_infinite() {
187            let cell_index = [
188                usize::clamp(cell_index[0] as usize, 0, n_cells[0] - 1),
189                usize::clamp(cell_index[1] as usize, 0, n_cells[1] - 1),
190                usize::clamp(cell_index[2] as usize, 0, n_cells[2] - 1),
191            ];
192            ([0, 0, 0], cell_index)
193        } else {
194            divmod_vec(cell_index, n_cells)
195        };
196
197        self.cells[cell_index].push(AtomData {
198            index: index,
199            shift: CellShift(shift),
200        });
201    }
202
203    /// Get the list of candidate pair. Some pairs might be separated by more
204    /// than `cutoff`, so additional filtering of the pairs might be required
205    /// later.
206    ///
207    /// This function produces a so-called "half" neighbors list, where each
208    /// pair is only included once. For example, if atoms 33 and 64 are in range
209    /// of each other, the output will only contain pairs in the order 33-64,
210    /// and not 64-33.
211    ///
212    /// If two atoms are neighbors of one another more than once (this can
213    /// happen when not using minimal image convention), all pairs at different
214    /// distances/directions are still included. Using the example above and
215    /// with a cutoff of 5 Å, we can have a pair between atoms 33-64 at 2.6 Å
216    /// and another pair between atoms 33-64 at 4.8 Å.
217    pub fn pairs(&self) -> Vec<CellPair> {
218        let mut pairs = Vec::new();
219
220        let n_cells = self.cells.shape();
221        let n_cells = [n_cells[0], n_cells[1], n_cells[2]];
222
223        let search_x = -self.n_search[0]..=self.n_search[0];
224        let search_y = -self.n_search[1]..=self.n_search[1];
225        let search_z = -self.n_search[2]..=self.n_search[2];
226
227        // for each cell in the cell list
228        for ((cell_i_x, cell_i_y, cell_i_z), current_cell) in self.cells.indexed_iter() {
229            // look through each neighboring cell
230            for delta_x in search_x.clone() {
231                for delta_y in search_y.clone() {
232                    for delta_z in search_z.clone() {
233                        let cell_i = [
234                            cell_i_x as i32 + delta_x,
235                            cell_i_y as i32 + delta_y,
236                            cell_i_z as i32 + delta_z,
237                        ];
238
239                        // shift vector from one cell to the other and index of
240                        // the neighboring cell
241                        let (cell_shift, neighbor_cell_i) = divmod_vec(cell_i, n_cells);
242
243                        for atom_i in current_cell {
244                            for atom_j in &self.cells[neighbor_cell_i] {
245                                // create a half neighbor list
246                                if atom_i.index > atom_j.index {
247                                    continue;
248                                }
249
250                                let shift = CellShift(cell_shift) + atom_i.shift - atom_j.shift;
251                                let shift_is_zero = shift[0] == 0 && shift[1] == 0 && shift[2] == 0;
252
253                                if atom_i.index == atom_j.index {
254                                    if shift_is_zero {
255                                        // only create pairs with the same atom twice
256                                        // if the pair spans more than one unit cell
257                                        continue;
258                                    }
259
260                                    // When creating pairs between an atom and one of its
261                                    // periodic images, the code generate multiple redundant
262                                    // pairs (e.g. with shifts 0 1 1 and 0 -1 -1); and we
263                                    // want to only keep one of these.
264                                    if shift[0] + shift[1] + shift[2] < 0 {
265                                        // drop shifts on the negative half-space
266                                        continue;
267                                    }
268
269                                    if (shift[0] + shift[1] + shift[2] == 0)
270                                        && (shift[2] < 0 || (shift[2] == 0 && shift[1] < 0)) {
271
272                                        // drop shifts in the negative half plane or the
273                                        // negative shift[1] axis. See below for a graphical
274                                        // representation: we are keeping the shifts indicated
275                                        // with `O` and dropping the ones indicated with `X`
276                                        //
277                                        //  O O O │ O O O
278                                        //  O O O │ O O O
279                                        //  O O O │ O O O
280                                        // ─X─X─X─┼─O─O─O─
281                                        //  X X X │ X X X
282                                        //  X X X │ X X X
283                                        //  X X X │ X X X
284                                        continue;
285                                    }
286                                }
287
288                                if self.unit_cell.is_infinite() && !shift_is_zero {
289                                    // do not create pairs crossing the periodic
290                                    // boundaries in an infinite cell
291                                    continue;
292                                }
293
294                                pairs.push(CellPair {
295                                    first: atom_i.index,
296                                    second: atom_j.index,
297                                    shift: shift,
298                                });
299                            }
300                        } // loop over atoms in current neighbor cells
301
302                    }
303                }
304            } // loop over neighboring cells
305
306        }
307
308        return pairs;
309    }
310}

313/// Function to compute both quotient and remainder of the division of a by b.
314/// This function follows Python convention, making sure the remainder have the
315/// same sign as `b`.
316fn divmod(a: i32, b: usize) -> (i32, usize) {
317    debug_assert!(b < (i32::MAX as usize));
318    let b = b as i32;
319    let mut quotient = a / b;
320    let mut remainder = a % b;
321    if remainder < 0 {
322        remainder += b;
323        quotient -= 1;
324    }
325    return (quotient, remainder as usize);
326}
327
328/// Apply the [`divmod`] function to three components at the time
329fn divmod_vec(a: [i32; 3], b: [usize; 3]) -> ([i32; 3], [usize; 3]) {
330    let (qx, rx) = divmod(a[0], b[0]);
331    let (qy, ry) = divmod(a[1], b[1]);
332    let (qz, rz) = divmod(a[2], b[2]);
333    return ([qx, qy, qz], [rx, ry, rz]);
334}

336/// A neighbor list implementation usable with any system
337#[derive(Clone, Debug)]
338pub struct NeighborsList {
339    /// the cutoff used to create this neighbor list
340    pub cutoff: f64,
341    /// all pairs in the system
342    pub pairs: Vec<Pair>,
343    /// all pairs associated with a given atom
344    pub pairs_by_atom: Vec<Vec<Pair>>,
345}
346
347impl NeighborsList {
348    #[time_graph::instrument(name = "NeighborsList")]
349    pub fn new(positions: &[Vector3D], unit_cell: UnitCell, cutoff: f64) -> NeighborsList {
350        let mut cell_list = CellList::new(unit_cell, cutoff);
351
352        for (index, &position) in positions.iter().enumerate() {
353            cell_list.add_atom(index, position);
354        }
355
356        let cell_matrix = unit_cell.matrix();
357        let cutoff2 = cutoff * cutoff;
358
359        // the cell list creates too many pairs, we only need to keep the one where
360        // the distance is actually below the cutoff
361        let mut pairs = Vec::new();
362        let mut pairs_by_center = vec![Vec::new(); positions.len()];
363
364        for pair in cell_list.pairs() {
365            let mut vector = positions[pair.second] - positions[pair.first];
366            vector += pair.shift.cartesian(&cell_matrix);
367
368            let distance2 = vector * vector;
369            if distance2 < cutoff2 {
370                if distance2 < 1e-3 {
371                    warn!(
372                        "atoms {} and {} are very close to one another ({} A)",
373                        pair.first, pair.second, distance2.sqrt()
374                    );
375                }
376
377                let pair = Pair {
378                    first: pair.first,
379                    second: pair.second,
380                    distance: distance2.sqrt(),
381                    vector: vector,
382                    cell_shift_indices: pair.shift.0
383                };
384
385                pairs.push(pair);
386                pairs_by_center[pair.first].push(pair);
387                pairs_by_center[pair.second].push(pair);
388            }
389        }
390
391        // sort the pairs to make sure the final output of featomic is ordered
392        // naturally
393        pairs.sort_unstable_by_key(|pair| (pair.first, pair.second));
394        for pairs in &mut pairs_by_center {
395            pairs.sort_unstable_by_key(|pair| (pair.first, pair.second));
396        }
397
398        return NeighborsList {
399            cutoff: cutoff,
400            pairs: pairs,
401            pairs_by_atom: pairs_by_center,
402        };
403    }
404}
