11/// Compute the 1F1 confluent hypergeometric function.
12///
13/// The implementation is translated from scipy, and distributed under the
14/// CSD-3-Clauses license, Copyright (c) 2001-2002 Enthought, Inc. 2003-2022,
15/// scipy Developers.
16#[allow(clippy::float_cmp)]
17pub fn hyp1f1(a: f64, b: f64, x: f64) -> f64 {
18    assert!(a.is_finite() && b.is_finite() && x.is_finite());
19    assert!(b > 0.0, "b must be positive");
20
21    if a == 0.0 || x == 0.0 {
22        return 1.0;
23    } else if a == -1.0 {
24        return 1.0 - x / b;
25    } else if a == b {
26        return f64::exp(x);
27    } else if a - b == 1.0 {
28        return (1.0 + x / b) * f64::exp(x);
29    } else if a <= 0.0 && a == f64::floor(a) {
30        // The geometric series is finite in this case, but it could
31        // still suffer from cancellation.
32        return hyp1f1_series_track_convergence(a, b, x)
33    }
34
35    if b > 0.0 && (f64::abs(a) + 1.0) * f64::abs(x) < 0.9 * b {
36        // For the kth term of the series we are multiplying by
37        //
38        // t_k = (a + k) * x / ((b + k) * (k + 1))
39        //
40        // We have that
41        //
42        // |t_k| < (|a| + 1) * |x| / |b|,
43        //
44        // which means that in this branch we get geometric
45        // convergence.
46        return hyp1f1_series(a, b, x);
47    }
48
49    return chgm_fortran(a, b, x);
50}

53fn hyp1f1_series_track_convergence(a: f64, b: f64, x: f64) -> f64 {
54    // The hypergeometric series can suffer from cancellation or take a
55    // prohibitive number of terms to converge. This function computes
56    // the series while monitoring those conditions.
57    let mut converged = false;
58    let mut n_steps_to_converge = 0.0;
59
60    let mut a_p_k;
61    let mut b_p_k;
62    let mut term = 1.0;
63    let mut result = 1.0;
64    let mut abssum = result;
65
66    for k in (0..1000).map(|k| k as f64) {
67        a_p_k = a + k;
68        b_p_k = b + k;
69        if b_p_k != 0.0 {
70            term *= a_p_k * x / b_p_k / (k + 1.0);
71        } else if a_p_k == 0.0 {
72            // The Pochammer symbol in the denominator has become zero,
73            // but we still have the continuation formula DLMF 13.2.5.
74            term = 0.0;
75        } else {
76            // We hit a pole
77            return f64::NAN;
78        }
79
80        abssum += f64::abs(term);
81        result += term;
82        if f64::abs(term) <= f64::EPSILON * f64::abs(result) {
83            n_steps_to_converge = k;
84            converged = true;
85            break;
86        }
87    }
88
89    if !converged {
90        return f64::NAN;
91    }
92
93    if n_steps_to_converge * f64::EPSILON * abssum <= ACCEPTABLE_RTOL * f64::abs(result) {
94        return result;
95    }
96
97    return f64::NAN;
98}

101fn hyp1f1_series(a: f64, b: f64, x: f64) -> f64 {
102    let mut term = 1.0;
103    let mut result = 1.0;
104
105    let mut converged = false;
106    for k in (0..1000).map(|k| k as f64) {
107        term *= (a + k) * x / (b + k) / (k + 1.0);
108        result += term;
109        if f64::abs(term) <= f64::EPSILON * f64::abs(result) {
110            converged = true;
111            break;
112        }
113    }
114
115    if !converged {
116        return f64::NAN;
117    }
118
119    return result
120}

122/// This is a rust version of the CHGM fortran subroutine in scipy
123///
124/// The original fortran source was translated to C with f2c, then manually
125/// cleaned up and translated to rust.
126///
127/// DLMF refers to <https://dlmf.nist.gov/>
128#[allow(clippy::similar_names, clippy::too_many_lines)]
129fn chgm_fortran(mut a: f64, b: f64, mut x: f64) -> f64{
130    let mut a0;
131    let mut y0;
132    let mut y1;
133    let mut la;
134    let mut nl;
135
136    let mut hg1;
137    let mut hg2;
138    let mut ln_gamma_a;
139    let mut ln_gamma_b;
140    let mut ln_gamma_ba;
141
142    a0 = a;
143    let x0 = x;
144
145    let mut result = 0.0;
146
147    /* DLMF 13.2.39 */
148    if x < 0. {
149        a = b - a;
150        a0 = a;
151        x = f64::abs(x);
152    }
153    nl = 0;
154    la = 0;
155    if a >= 2.0 {
156        /* preparing terms for DLMF 13.3.1 */
157        nl = 1;
158        la = a as i32;
159        a = a - (la as f64) - 1.0;
160    }
161
162    y0 = 0.0;
163    y1 = 0.0;
164
165    for n in 0..=nl {
166        if a0 >= 2.0 {
167            a += 1.0;
168        }
169        if x <= f64::abs(b) + 30.0 || a < 0.0 {
170            result = 1.0;
171            let mut term = 1.0;
172
173            if x < 500.0 {
174                // use the normal definition (DLMF 13.2.2)
175                for j in (1..=500).map(|j| j as f64) {
176                    term *= (a + j - 1.) / (j * (b + j - 1.)) * x;
177                    result += term;
178                    if result != 0. && f64::abs(term / result) < 1e-15 {
179                        break;
180                    }
181                }
182
183                /* DLMF 13.2.39 (cf. above) */
184                if x0 < 0.0 {
185                    result *= f64::exp(x0);
186                }
187            } else {
188                // use the definition for x -> \infty (DLMF 13.7.1 and 13.2.3)
189                for j in (1..=50).map(|j| j as f64) {
190                    term *= (j - a) * (b - a + j - 1.) / (j * x);
191                    result += term;
192                    if result != 0. && f64::abs(term / result) < 1e-15 {
193                        break;
194                    }
195                }
196
197                result *= x.powf(a - b) * gamma(b) / gamma(a);
198                /* DLMF 13.2.39 cancels the exp(x) for x0 < 0 */
199                if x0 > 0.0 {
200                    result *= f64::exp(x);
201                }
202            }
203        } else {
204            /* DLMF 13.7.2 & 13.2.4, SUM2 corresponds to first sum */
205            ln_gamma_a = ln_gamma(a);
206            ln_gamma_b = ln_gamma(b);
207            ln_gamma_ba = ln_gamma(b - a);
208
209            let mut sum_1 = 1.0;
210            let mut sum_2 = 1.0;
211            let mut term_1 = 1.0;
212            let mut term_2 = 1.0;
213
214            // CHANGED from the original FORTRAN implementation
215            //
216            // use 30 terms instead of 8 here, to get better accuracy (accuracy
217            // on GTO goes from ~1e-6 for gradients to ~1e-11)
218            for i in (1..=30).map(|i| i as f64) {
219                term_1 = -term_1 * (a + i - 1.) * (a - b + i) / (x * i);
220                term_2 = -term_2 * (b - a + i - 1.) * (a - i) / (x * i);
221                sum_1 += term_1;
222                sum_2 += term_2;
223            }
224
225            if x0 >= 0.0 {
226                hg1 = f64::exp(ln_gamma_b - ln_gamma_ba) * f64::powf(x, -a) * f64::cos(f64::consts::PI * a) * sum_1;
227                hg2 = f64::exp(ln_gamma_b - ln_gamma_a + x) * f64::powf(x, a - b) * sum_2;
228            } else {
229                /* DLMF 13.2.39 (cf. above) */
230                hg1 = f64::exp(ln_gamma_b - ln_gamma_ba + x0) * f64::powf(x, -a) * f64::cos(f64::consts::PI * a) * sum_1;
231                hg2 = f64::exp(ln_gamma_b - ln_gamma_a) * f64::powf(x, a - b) * sum_2;
232            }
233            result = hg1 + hg2;
234        }
235
236        if n == 0 {
237            y0 = result;
238        }
239        if n == 1 {
240            y1 = result;
241        }
242    }
243
244    if a0 >= 2.0 {
245        /* DLMF 13.3.1 */
246        for _ in 1..la {
247            result = ((a * 2.0 - b + x) * y1 + (b - a) * y0) / a;
248            y0 = y1;
249            y1 = result;
250            a += 1.0;
251        }
252    }
253
254    return result;
255}
