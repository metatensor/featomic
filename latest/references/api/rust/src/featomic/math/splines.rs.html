7/// Maximal number of points in the splines
8const MAX_SPLINE_SIZE: usize = 10_000;
9
10/// [Hermit cubit spline][splines-wiki] implementation.
11///
12/// This kind of spline uses information of the values of a function on control
13/// points as well as the derivatives of the function at these points. This
14/// implementation takes a single scalar as input, and output array values, i.e.
15/// we can spline functions of the form `R -> R^n`.
16///
17/// [splines-wiki]: https://en.wikipedia.org/wiki/Cubic_Hermite_spline
18#[derive(Debug, Clone)]
19pub struct HermitCubicSpline<D: ndarray::Dimension> {
20    parameters: SplineParameters,
21    pub(crate) points: Vec<HermitSplinePoint<D>>,
22}
23
24
25/// Parameters controlling a `HermitCubicSpline`
26#[derive(Debug, Clone)]
27pub struct SplineParameters {
28    /// Beginning of the interpolation space
29    pub start: f64,
30    /// End of the interpolation space
31    pub stop: f64,
32    /// Shape of the array output
33    pub shape: Vec<usize>,
34}
35
36/// A single control point/knot in the Hermit cubic spline
37#[derive(Debug, Clone)]
38pub struct HermitSplinePoint<D: ndarray::Dimension> {
39    /// Position of the point
40    pub(crate) position: f64,
41    /// Values of the function to interpolate at the position
42    pub(crate) values: Array<f64, D>,
43    /// Derivatives of the function to interpolate at the position
44    pub(crate) derivatives: Array<f64, D>,
45}
46
47impl<D: ndarray::Dimension> HermitCubicSpline<D> {
48    #[allow(clippy::float_cmp)]
49    pub(crate) fn new(
50        parameters: SplineParameters,
51        mut points: Vec<HermitSplinePoint<D>>,
52    ) -> HermitCubicSpline<D>  {
53        assert!(points.len() >= 2, "we need at least two points to create a spline");
54        assert!(parameters.start < parameters.stop);
55
56        points.sort_unstable_by(|a, b| {
57            a.position.partial_cmp(&b.position).expect("got NaN while sorting by point position")
58        });
59
60        assert_eq!(points.first().unwrap().position, parameters.start);
61        assert!(points.last().unwrap().position >= parameters.stop);
62
63        Self {
64            parameters: parameters,
65            points: points,
66        }
67    }
68
69    /// Create a new `HermitCubicSpline` from the given function, trying to
70    /// reach the given accuracy on average.
71    ///
72    /// When called, the `function` should return a tuple of `(values, derivatives)`
73    /// at the input position, where `values` and `derivatives` are arrays with the
74    /// same shape as `parameter.shape`.
75    ///
76    /// Points are added to the spline until the requested accuracy is reached.
77    /// We consider that the accuracy is reached when either the mean absolute
78    /// error or the mean relative error gets below the `accuracy` threshold.
79    pub fn with_accuracy<F>(
80        accuracy: f64,
81        parameters: SplineParameters,
82        function: F,
83    ) -> Result<HermitCubicSpline<D>, Error> where
84            F: Fn(f64) -> (Array<f64, D>, Array<f64, D>),
85    {
86        if accuracy < 0.0 {
87            return Err(Error::InvalidParameter(format!(
88                "got invalid accuracy in spline ({}), it must be positive", accuracy
89            )));
90        }
91
92        let interpolated = Array::from_elem(parameters.shape.clone(), 0.0);
93        let mut interpolated = match interpolated.into_dimensionality::<D>() {
94            Ok(array) => array,
95            Err(e) => {
96                return Err(Error::InvalidParameter(format!(
97                    "wrong shape parameter: {}", e
98                )));
99            }
100        };
101
102        let initial_grid_size = 11;
103        let grid_step = (parameters.stop - parameters.start) / (initial_grid_size - 1) as f64;
104
105        let mut points = Vec::new();
106        for k in 0..(initial_grid_size - 1) {
107            let position = parameters.start + k as f64 * grid_step;
108            let (values, derivatives) = function(position);
109
110            if values.shape() != parameters.shape || derivatives.shape() != parameters.shape  {
111                return Err(Error::InvalidParameter(format!(
112                    "function ({:?}) or gradient of the function ({:?}) returned a different shape than expected ({:?})",
113                    values.shape(), derivatives.shape(), parameters.shape
114                )));
115            }
116
117            points.push(HermitSplinePoint { position, values, derivatives });
118        }
119
120        // Add a point exactly at `parameters.stop`
121        let position = parameters.stop;
122        let (values, derivatives) = function(position);
123        if values.shape() != parameters.shape || derivatives.shape() != parameters.shape  {
124            return Err(Error::InvalidParameter(format!(
125                "function ({:?}) or gradient of the function ({:?}) returned a different shape than expected ({:?})",
126                values.shape(), derivatives.shape(), parameters.shape
127            )));
128        }
129        points.push(HermitSplinePoint { position, values, derivatives });
130
131        let mut spline = HermitCubicSpline::new(parameters, points);
132
133        // add more points as required to reach the requested accuracy
134        loop {
135            let mut max_absolute_error = 0.0;
136            let mut mean_absolute_error = 0.0;
137            let mut mean_relative_error = 0.0;
138            let mut error_count = 0;
139
140            let positions = spline.positions();
141
142            // evaluate the error at points in between grid points, since these
143            // should have the highest error in average.
144            let mut new_points = Vec::new();
145            for k in 0..(spline.len() - 1) {
146                let position = (positions[k] + positions[k + 1]) / 2.0;
147
148                let (values, derivatives) = function(position);
149
150                interpolated.fill(0.0);
151                spline.compute(position, interpolated.view_mut(), None);
152
153                // get the error across all values in the arrays
154                azip!((interpolated in &interpolated, values in &values) {
155                    let absolute_error = f64::abs(interpolated - values);
156                    if absolute_error > max_absolute_error {
157                        max_absolute_error = absolute_error;
158                    }
159
160                    mean_absolute_error += absolute_error;
161                    mean_relative_error += f64::abs((interpolated - values) / values);
162                    error_count += 1;
163                });
164
165                new_points.push(HermitSplinePoint { position, values, derivatives });
166            }
167            mean_absolute_error /= error_count as f64;
168            mean_relative_error /= error_count as f64;
169
170            if mean_absolute_error < accuracy || mean_relative_error < accuracy {
171                debug!(
172                    "spline reached requested accuracy ({:.3e}) with {} reference points (max absolute error is {:.3e})",
173                    accuracy, spline.len(), max_absolute_error,
174                );
175                break;
176            }
177
178            if spline.len() + new_points.len() > MAX_SPLINE_SIZE {
179                return Err(Error::Internal(format!(
180                    "failed to reach requested accuracy ({:e}) in spline interpolation, \
181                    mean absolute error is {:e} and mean relative error is {:e}",
182                    accuracy, mean_absolute_error, mean_relative_error
183                )));
184            }
185
186            // add more points and continue
187            for point in new_points {
188                spline.add_point(point);
189            }
190        }
191
192        return Ok(spline);
193    }

195    /// Add a new control points to this spline. The new point must be between
196    /// `self.start` and `self.stop`.
197    fn add_point(&mut self, point: HermitSplinePoint<D>) {
198        debug_assert!(point.position > self.parameters.start);
199        debug_assert!(point.position < self.parameters.stop);
200        match self.points.binary_search_by(
201            |v| v.position.partial_cmp(&point.position).expect("got NaN")
202        ) {
203            Ok(_) => panic!("trying to add the same point twice to the spline"),
204            Err(k) => self.points.insert(k, point)
205        }
206    }
207
208    /// Get the number of control points in this spline
209    pub fn len(&self) -> usize {
210        self.points.len()
211    }
212
213    /// Get the shape of the arrays returned by the splined function
214    pub fn shape(&self) -> &[usize] {
215        &self.parameters.shape
216    }
217
218    /// Get the position of the control points for this spline
219    fn positions(&self) -> Vec<f64> {
220        self.points.iter().map(|p| p.position).collect()
221    }

223    /// Compute the spline at point `x`, storing the results in `values` and
224    /// optionally `gradients`.
225    pub fn compute(&self, x: f64, values: ArrayViewMut<f64, D>, gradients: Option<ArrayViewMut<f64, D>>) {
226        debug_assert!(x.is_finite());
227        debug_assert!(x >= self.parameters.start && x <= self.parameters.stop);
228        debug_assert_eq!(values.shape(), self.parameters.shape);
229        if let Some(ref gradients) = gradients {
230            debug_assert_eq!(gradients.shape(), self.parameters.shape);
231        }
232
233        // notation in this function follows
234        // https://en.wikipedia.org/wiki/Cubic_Hermite_spline
235
236        let mut k = match self.points.binary_search_by(
237            |v| v.position.partial_cmp(&x).expect("got NaN")
238        ) {
239            Ok(k) => k,
240            Err(k) => k - 1,
241        };
242
243        // If we are evaluating at exactly the last spline point, use the
244        // previous point as a basis, and t will be 1 below.
245        if k == self.points.len() - 1 {
246            k -= 1;
247        }
248
249        let point_k = &self.points[k];
250        let point_k_1 = &self.points[k + 1];
251
252        let x_k = point_k.position;
253        let x_k_1 = point_k_1.position;
254        debug_assert!(x_k <= x && x <= x_k_1);
255
256        let delta = x_k_1 - x_k;
257        let t = (x - x_k) / delta;
258        let t_2 = t * t;
259        let t_3 = t_2 * t;
260
261        // Hermit base polynomials
262        let h00 = 2.0 * t_3 - 3.0 * t_2 + 1.0;
263        let h10 = t_3 - 2.0 * t_2 + t;
264        let h01 = -2.0 * t_3 + 3.0 * t_2;
265        let h11 = t_3 - t_2;
266
267        let p_k = &point_k.values;
268        let p_k_1 = &point_k_1.values;
269
270        let m_k = &point_k.derivatives;
271        let m_k_1 = &point_k_1.derivatives;
272
273        azip!((v in values, p_k in p_k, p_k_1 in p_k_1, m_k in m_k, m_k_1 in m_k_1) {
274            *v = h00 * p_k + h10 * delta * m_k + h01 * p_k_1 + h11 * delta * m_k_1;
275        });
276
277        if let Some(gradients) = gradients {
278            let d_h00_dt = 6.0 * (t_2 - t);
279            let d_h10_dt = 3.0 * t_2 - 4.0 * t + 1.0;
280            let d_h01_dt = -d_h00_dt;
281            let d_h11_dt = 3.0 * t_2 - 2.0 * t;
282
283            let dx_dt = 1.0 / delta;
284
285            azip!((g in gradients, p_k in p_k, p_k_1 in p_k_1, m_k in m_k, m_k_1 in m_k_1) {
286                *g = d_h00_dt * p_k * dx_dt + d_h10_dt * m_k + d_h01_dt * p_k_1 * dx_dt + d_h11_dt * m_k_1;
287            });
288        }
289    }

295    use approx::{assert_relative_eq, assert_ulps_eq};

297    #[test]
298    fn evaluate_simple_spline() {
299        let accuracy = 1e-9;
300        let parameters = SplineParameters {
301            start: -3.0,
302            stop: 6.0,
303            shape: vec![1],
304        };
305        let spline = HermitCubicSpline::with_accuracy(
306            accuracy,
307            parameters,
308            |x| (ndarray::arr1(&[f64::sin(x)]), ndarray::arr1(&[f64::cos(x)])),
309        ).unwrap();
310
311        let mut values = ndarray::Array1::from_elem((1,), 0.0);
312        let mut gradients = ndarray::Array1::from_elem((1,), 0.0);
313        for &x in &[-3.0, -2.2, -1.00242144, 0.0, 0.000000001, 2.3, 3.2, 4.7, 5.3, 5.99999999, 6.0] {
314            spline.compute(x, values.view_mut(), Some(gradients.view_mut()));
315            assert_relative_eq!(values[0], f64::sin(x), max_relative=1e-5, epsilon=1e-12);
316            assert_relative_eq!(gradients[0], f64::cos(x), max_relative=1e-5, epsilon=1e-12);
317        }
318
319        // check that the values match exactly at the control points. The only
320        // exception is the last control point were we can not compute the
321        // spline.
322        for &x in spline.positions().iter().filter(|&&x| x < spline.parameters.stop) {
323            spline.compute(x, values.view_mut(), Some(gradients.view_mut()));
324            assert_ulps_eq!(values[0], f64::sin(x));
325            assert_ulps_eq!(gradients[0], f64::cos(x));
326        }
327    }
