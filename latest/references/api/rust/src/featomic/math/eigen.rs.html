1// Eigendecomposition of symmetric matrix. Adapted from
2// https://github.com/xasmx/rust-la, which is a Rust port of the JAMA implementation
3// https://en.wikipedia.org/wiki/JAMA_(numerical_linear_algebra_library)

5use ndarray::{Array2, Array1};
6
7/// Eigendecomposition of a real symmetric matrix into eigenvalues and eigenvectors
8#[derive(Debug, Clone)]
9pub struct SymmetricEigen {
10    /// Eigenvalues of the input matrix, sorted in increasing order
11    pub eigenvalues: Array1<f64>,
12    /// Eigenvectors of the input matrix
13    pub eigenvectors: Array2<f64>,
14}
15
16impl SymmetricEigen {
17    // Symmetric Householder reduction to tridiagonal form.
18    //
19    //  This is derived from the Algol procedures tred2 by Bowdler, Martin,
20    //  Reinsch, and Wilkinson, Handbook for Auto. Comp., Vol.ii-Linear
21    //  Algebra, and the corresponding Fortran subroutine in EISPACK.
22    #[allow(clippy::needless_range_loop)]
23    fn tridiagonalize(eval: &mut Array1<f64>, evec: &mut Array2<f64>, work: &mut [f64]) {
24        let n = eval.len();
25        debug_assert_eq!(work.len(), n);
26        debug_assert_eq!(evec.len(), n * n);
27
28        for j in 0..n {
29            eval[j] = evec[[(n - 1), j]];
30        }
31
32        // Householder reduction to tridiagonal form.
33        for i in (1..n).rev() {
34            // Scale to avoid under/overflow.
35            let mut scale = 0.0;
36            let mut h = 0.0;
37            for k in 0..i {
38                scale += f64::abs(eval[k]);
39            }
40            if scale == 0.0 {
41                work[i] = eval[i - 1];
42                for j in 0..i {
43                    eval[j] = evec[[(i - 1), j]];
44                    evec[[i, j]] = 0.0;
45                    evec[[j, i]] = 0.0;
46                }
47            } else {
48                // Generate Householder vector.
49                for k in 0..i {
50                    eval[k] /= scale;
51                    h += eval[k] * eval[k];
52                }
53                let mut f = eval[i - 1];
54                let mut g = h.sqrt();
55                if f > 0.0 {
56                    g = -g;
57                }
58                work[i] = scale * g;
59                h -= f * g;
60                eval[i - 1] = f - g;
61                for j in 0..i {
62                    work[j] = 0.0;
63                }
64
65                // Apply similarity transformation to remaining columns.
66                for j in 0..i {
67                    f = eval[j];
68                    evec[[j, i]] = f;
69                    g = work[j] + evec[[j, j]] * f;
70                    for k in (j + 1)..i {
71                        g += evec[[k, j]] * eval[k];
72                        work[k] += evec[[k, j]] * f;
73                    }
74                    work[j] = g;
75                }
76                f = 0.0;
77                for j in 0..i {
78                    work[j] /= h;
79                    f += work[j] * eval[j];
80                }
81                let hh = f / (h + h);
82                for j in 0..i {
83                    work[j] -= hh * eval[j];
84                }
85                for j in 0..i {
86                    f = eval[j];
87                    g = work[j];
88                    for k in j..i {
89                        let orig_val = evec[[k, j]];
90                        evec[[k, j]] = orig_val - (f * work[k] + g * eval[k]);
91                    }
92                    eval[j] = evec[[(i - 1), j]];
93                    evec[[i, j]] = 0.0;
94                }
95            }
96            eval[i] = h;
97        }
98
99        // Accumulate transformations.
100        for i in 0..(n - 1) {
101            let orig_val = evec[[i, i]];
102            evec[[(n - 1), i]] = orig_val;
103            evec[[i, i]] = 1.0;
104            let h = eval[i + 1];
105            if h != 0.0 {
106                for k in 0..(i + 1) {
107                    eval[k] = evec[[k, (i + 1)]] / h;
108                }
109                for j in 0..(i + 1) {
110                    let mut g = 0.0;
111                    for k in 0..(i + 1) {
112                        g += evec[[k, (i + 1)]] * evec[[k, j]];
113                    }
114                    for k in 0..(i + 1) {
115                        let orig_val = evec[[k, j]];
116                        evec[[k, j]] = orig_val - g * eval[k];
117                    }
118                }
119            }
120            for k in 0..(i + 1) {
121                evec[[k, (i + 1)]] = 0.0;
122            }
123        }
124        for j in 0..n {
125            eval[j] = evec[[(n - 1), j]];
126            evec[[(n - 1), j]] = 0.0;
127        }
128        evec[[(n - 1), (n - 1)]] = 1.0;
129        work[0] = 0.0;
130    }
131
132    // Symmetric tridiagonal QL algorithm.
133    //
134    // This is derived from the Algol procedures tql2, by Bowdler, Martin,
135    // Reinsch, and Wilkinson, Handbook for Auto. Comp., Vol.ii-Linear
136    // Algebra, and the corresponding Fortran subroutine in EISPACK.
137    #[allow(clippy::many_single_char_names)]
138    fn tridiagonal_ql(eval: &mut Array1<f64>, evec: &mut Array2<f64>, work: &mut [f64]) {
139        let n = eval.len();
140        debug_assert_eq!(work.len(), n);
141        debug_assert_eq!(evec.len(), n * n);
142
143        for i in 1..n {
144            work[i - 1] = work[i];
145        }
146        work[n - 1] = 0.0;
147
148        let mut f = 0.0;
149        let mut tst1 = 0.0f64;
150        for l in 0..n {
151            // Find small subdiagonal element
152            tst1 = tst1.max(f64::abs(eval[l]) + f64::abs(work[l]));
153            let mut m = l;
154            while m < n {
155                if f64::abs(work[m]) <= (f64::EPSILON * tst1) {
156                    break;
157                }
158                m += 1;
159            }
160
161            // If m == l, d[l] is an eigenvalue, otherwise, iterate.
162            if m > l {
163                loop {
164                    // Compute implicit shift
165                    let mut g = eval[l];
166                    let tmp = 2.0;
167                    let mut p = (eval[l + 1] - g) / (tmp * work[l]);
168                    let mut r = f64::hypot(p, 1.0);
169                    if p < 0.0 {
170                        r = -r;
171                    }
172                    eval[l] = work[l] / (p + r);
173                    eval[l + 1] = work[l] * (p + r);
174                    let eval_lp1 = eval[l + 1];
175                    let mut h = g - eval[l];
176                    for i in (l + 2)..n {
177                        eval[i] -= h;
178                    }
179                    f += h;
180
181                    // Implicit QL transformation.
182                    p = eval[m];
183                    let mut c = 1.0;
184                    let mut c2 = c;
185                    let mut c3 = c;
186                    let work_lp1 = work[l + 1];
187                    let mut s = 0.0;
188                    let mut s2 = 0.0;
189                    for i in (l..m).rev() {
190                        c3 = c2;
191                        c2 = c;
192                        s2 = s;
193                        g = c * work[i];
194                        h = c * p;
195                        r = f64::hypot(p, work[i]);
196                        work[i + 1] = s * r;
197                        s = work[i] / r;
198                        c = p / r;
199                        p = c * eval[i] - s * g;
200                        eval[i + 1] = h + s * (c * g + s * eval[i]);
201
202                        // Accumulate transformation.
203                        for k in 0..n {
204                            h = evec[[k, (i + 1)]];
205                            evec[[k, (i + 1)]] = s * evec[[k, i]] + c * h;
206                            evec[[k, i]] = c * evec[[k, i]] - s * h;
207                        }
208                    }
209                    p = -s * s2 * c3 * work_lp1 * work[l] / eval_lp1;
210                    work[l] = s * p;
211                    eval[l] = c * p;
212
213                    // Check for convergence.
214                    if f64::abs(work[l]) <= (f64::EPSILON * tst1) {
215                        break;
216                    }
217                }
218            }
219            eval[l] += f;
220            work[l] = 0.0;
221        }
222
223        // Bubble sort eigenvalues and corresponding vectors.
224        for i in 0..(n - 1) {
225            let mut k = i;
226            let mut p = eval[i];
227            for j in (i + 1)..n {
228                if eval[j] < p {
229                    k = j;
230                    p = eval[j];
231                }
232            }
233            if k != i {
234                // Swap columns k and i of the diagonal and v.
235                eval[k] = eval[i];
236                eval[i] = p;
237                for j in 0..n {
238                    p = evec[[j, i]];
239                    evec[[j, i]] = evec[[j, k]];
240                    evec[[j, k]] = p;
241                }
242            }
243        }
244    }

246    /// Compute the eigendecomposition of a symmetric real matrix
247    #[allow(clippy::float_cmp)]
248    pub fn new(matrix: Array2<f64>) -> SymmetricEigen {
249        assert_eq!(matrix.nrows(), matrix.ncols());
250
251        let n = matrix.ncols();
252        for i in 0..n {
253            for j in i..n {
254                debug_assert_eq!(matrix[[i, j]], matrix[[j, i]], "matrix is not symmetric");
255            }
256        }
257
258        let mut eigenvectors = matrix;
259        let mut eigenvalues = Array1::from_elem(n, 0.0);
260
261        let mut work = vec![0.0; n];
262
263        SymmetricEigen::tridiagonalize(&mut eigenvalues, &mut eigenvectors, &mut work);
264        SymmetricEigen::tridiagonal_ql(&mut eigenvalues, &mut eigenvectors, &mut work);
265
266        SymmetricEigen {
267            eigenvalues,
268            eigenvectors,
269        }
270    }
271
272    /// Recreate the input matrix from the eigenvalues and eigenvectors
273    pub fn recompose(&self) -> Array2<f64> {
274        let result = self.eigenvectors.dot(&Array2::from_diag(&self.eigenvalues));
275        return result.dot(&self.eigenvectors.t());
276    }
277}
