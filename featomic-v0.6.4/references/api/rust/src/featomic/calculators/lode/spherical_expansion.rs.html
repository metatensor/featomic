30/// Parameters for LODE spherical expansion calculator.
31///
32/// The spherical expansion is at the core of representations in the LODE
33/// (long-distance equivariant) family. See [this
34/// article](https://aip.scitation.org/doi/10.1063/1.5128375) for more
35/// information on the LODE representation.
36#[derive(Debug, Clone)]
37#[derive(serde::Deserialize, serde::Serialize, schemars::JsonSchema)]
38pub struct LodeSphericalExpansionParameters {
39    /// Spherical reciprocal cutoff. If `k_cutoff` is `None`, a cutoff of
40    /// `1.2 Ï€ / SmearedPowerLaw.width`, which is a reasonable value for most
41    /// systems, is used.
42    pub k_cutoff: Option<f64>,
43    /// Definition of the density arising from atoms in the whole system
44    pub density: Density,
45    /// Definition of the basis functions used to expand the atomic density in
46    /// local environments
47    pub basis: SphericalExpansionBasis<LodeRadialBasis>,
48}
49
50impl LodeSphericalExpansionParameters {
51    /// Get the value of the k-space cutoff (either provided by the user or a
52    /// default).
53    pub fn get_k_cutoff(&self) -> f64 {
54        match self.density.kind {
55            SmearedPowerLaw { smearing, .. } => {
56                return self.k_cutoff.unwrap_or(1.2 * std::f64::consts::PI / smearing);
57            },
58            _ => unreachable!()
59        }
60    }
61}

80/// Compute the trigonometric functions for LODE coefficients
81struct StructureFactors {
82    /// Real part of `e^{i k r}`, the array shape is `(n_atoms, k_vector)`
83    real: Array2<f64>,
84    /// Imaginary part of `e^{i k r}`, the array shape is `(n_atoms, k_vector)`
85    imag: Array2<f64>,
86    /// Real part of `\sum_j e^{i k r_i} e^{-i k r_j}`, with one map entry for
87    /// each atomic neighbor type. The arrays shape are `(n_atoms, k_vector)`
88    real_per_center: BTreeMap<i32, Array2<f64>>,
89    /// Imaginary part of `\sum_j e^{i k r_i} e^{-i k r_j}`, with one map entry
90    /// for each atomic neighbor type. The arrays shape are `(n_atoms,
91    /// k_vector)`
92    imag_per_center: BTreeMap<i32, Array2<f64>>,
93}

95fn compute_structure_factors(positions: &[Vector3D], types: &[i32], k_vectors: &[KVector]) -> StructureFactors {
96    let n_atoms = positions.len();
97    let n_k_vectors = k_vectors.len();
98
99    let mut cosines = Array2::from_elem((n_atoms, n_k_vectors), 0.0);
100    let mut sines = Array2::from_elem((n_atoms, n_k_vectors), 0.0);
101    for (atom_i, position) in positions.iter().enumerate() {
102        for (ik, k_vector) in k_vectors.iter().enumerate() {
103            // dot product between k-vector and positions
104            let s = k_vector.norm * k_vector.direction * position;
105
106            cosines[[atom_i, ik]] = f64::cos(s);
107            sines[[atom_i, ik]] = f64::sin(s);
108        }
109    }
110
111    let all_types = types.iter().copied().collect::<BTreeSet<_>>();
112    let mut real_per_center = all_types.iter().copied()
113        .map(|s| (s, Array2::from_elem((n_atoms, n_k_vectors), 0.0)))
114        .collect::<BTreeMap<_, _>>();
115    let mut imag_per_center = all_types.iter().copied()
116        .map(|s| (s, Array2::from_elem((n_atoms, n_k_vectors), 0.0)))
117        .collect::<BTreeMap<_, _>>();
118
119    // Precompute sums of sines and cosines over neighbors (j), depending on the
120    // neighbors atomic types
121    let mut sum_j_cos = all_types.iter().copied()
122        .map(|s| (s, Array1::from_elem(n_k_vectors, 0.0)))
123        .collect::<BTreeMap<_, _>>();
124    let mut sum_j_sin = all_types.iter().copied()
125        .map(|s| (s, Array1::from_elem(n_k_vectors, 0.0)))
126        .collect::<BTreeMap<_, _>>();
127    for j in 0..n_atoms {
128        let sum_j_cos = sum_j_cos.get_mut(&types[j]).unwrap();
129        let sum_j_sin = sum_j_sin.get_mut(&types[j]).unwrap();
130        for k in 0..n_k_vectors {
131            sum_j_cos[k] += cosines[[j, k]];
132            sum_j_sin[k] += sines[[j, k]];
133        }
134    }
135
136    // Compute Sum_j cos(k*r_ij) and Sum_j sin(k*r_ij) using the subtraction theorem
137    for i in 0..n_atoms {
138        for (neighbor_type, real_per_center) in &mut real_per_center {
139            let sum_j_cos = sum_j_cos.get_mut(neighbor_type).unwrap();
140            let sum_j_sin = sum_j_sin.get_mut(neighbor_type).unwrap();
141            let imag_per_center = imag_per_center.get_mut(neighbor_type).unwrap();
142            for k in 0..n_k_vectors {
143                let real = cosines[[i, k]] * sum_j_cos[k] + sines[[i, k]] * sum_j_sin[k];
144                let imag = sines[[i, k]] * sum_j_cos[k] - cosines[[i, k]] * sum_j_sin[k];
145                real_per_center[[i, k]] += 2.0 * real;
146                imag_per_center[[i, k]] += 2.0 * imag;
147            }
148        }
149    }
150
151    return StructureFactors {
152        real: cosines,
153        imag: sines,
154        real_per_center: real_per_center,
155        imag_per_center: imag_per_center,
156    }
157}

165/// Resize a 3D ndarray while keeping existing allocation
166fn resize_array3(array: &mut ndarray::Array3<f64>, shape: (usize, usize, usize)) {
167    let tmp_array = std::mem::take(array);
168
169    let (mut data, offset) = tmp_array.into_raw_vec_and_offset();
170    debug_assert!(matches!(offset, Some(0) | None));
171
172    data.resize(shape.0 * shape.1 * shape.2, 0.0);
173    *array = Array3::from_shape_vec(shape, data).expect("wrong shape");
174}

176/// Resize a 1D ndarray while keeping existing allocation
177fn resize_array1(array: &mut ndarray::Array1<f64>, shape: usize) {
178    let tmp_array = std::mem::take(array);
179
180    let (mut data, offset) = tmp_array.into_raw_vec_and_offset();
181    debug_assert!(matches!(offset, Some(0) | None));
182
183    data.resize(shape, 0.0);
184    *array = Array1::from_shape_vec(shape, data).expect("wrong shape");
185}

187impl LodeSphericalExpansion {
188    pub fn new(parameters: LodeSphericalExpansionParameters) -> Result<LodeSphericalExpansion, Error> {
189        match parameters.density.kind {
190            SmearedPowerLaw { exponent, .. } => {
191                if exponent >= 10 {
192                    return Err(Error::InvalidParameter(
193                        "LODE is only implemented for potential_exponent < 10".into()
194                    ));
195                }
196            }
197            _ => {
198                return Err(Error::InvalidParameter(
199                    "only SmearedPowerLaw density can be used with LODE".into()
200                ));
201            }
202        }
203
204        if parameters.density.scaling.is_some() {
205            return Err(Error::InvalidParameter(
206                "LODE does not support custom density scaling".into()
207            ));
208        }
209
210        // validate the parameters once here, so we are sure we can construct
211        // more radial integrals later
212        LodeRadialIntegralCacheByAngular::new(
213            parameters.density.kind,
214            &parameters.basis,
215            parameters.get_k_cutoff()
216        )?;
217
218        return Ok(LodeSphericalExpansion {
219            parameters,
220            spherical_harmonics: ThreadLocal::new(),
221            radial_integral: ThreadLocal::new(),
222            k_vector_to_m_n: ThreadLocal::new(),
223            k_dependent_values: ThreadLocal::new(),
224        });
225    }
226
227    fn project_k_to_nlm(&self, k_vectors: &[KVector]) {
228        let mut radial_integral = self.radial_integral.get_or(|| {
229            let radial_integral = LodeRadialIntegralCacheByAngular::new(
230                self.parameters.density.kind,
231                &self.parameters.basis,
232                self.parameters.get_k_cutoff()
233            ).expect("could not create a radial integral");
234
235            return RefCell::new(radial_integral);
236        }).borrow_mut();
237
238        let mut spherical_harmonics = self.spherical_harmonics.get_or(|| {
239            let max_angular = self.parameters.basis.angular_channels().into_iter().max().unwrap_or(0);
240            RefCell::new(SphericalHarmonicsCache::new(max_angular))
241        }).borrow_mut();
242
243        let mut k_vector_to_m_n = self.k_vector_to_m_n.get_or(|| {
244            let mut k_vector_to_m_n = BTreeMap::new();
245            for o3_lambda in self.parameters.basis.angular_channels() {
246                k_vector_to_m_n.insert(o3_lambda, Array3::from_elem((0, 0, 0), 0.0));
247            }
248
249            return RefCell::new(k_vector_to_m_n);
250        }).borrow_mut();
251
252        for o3_lambda in self.parameters.basis.angular_channels() {
253            let radial_size = radial_integral.get(o3_lambda).expect("missing o3_lambda").size();
254            let shape = (2 * o3_lambda + 1, radial_size, k_vectors.len());
255            resize_array3(k_vector_to_m_n.get_mut(&o3_lambda).expect("missing o3_lambda"), shape);
256        }
257
258        for (ik, k_vector) in k_vectors.iter().enumerate() {
259            // we don't need the gradients of spherical harmonics/radial
260            // integral w.r.t. k-vectors until we implement gradients w.r.t cell
261            radial_integral.compute(k_vector.norm, false);
262            spherical_harmonics.compute(k_vector.direction, false);
263
264            for o3_lambda in self.parameters.basis.angular_channels() {
265                let spherical_harmonics = spherical_harmonics.values.angular_slice(o3_lambda);
266                let radial_integral = radial_integral.get(o3_lambda).expect("missing o3_lambda");
267                let radial_integral = &radial_integral.values;
268                let array = k_vector_to_m_n.get_mut(&o3_lambda).expect("missing o3_lambda");
269
270                for (m, sph_value) in spherical_harmonics.iter().enumerate() {
271                    for (n, ri_value) in radial_integral.iter().enumerate() {
272                        array[[m, n, ik]] = ri_value * sph_value;
273                    }
274                }
275            }
276        }
277    }

279    #[allow(clippy::float_cmp)]
280    fn compute_density_fourier(&self, k_vectors: &[KVector]) -> Array1<f64> {
281        let (potential_exponent, smearing_squared) = match self.parameters.density.kind {
282            SmearedPowerLaw { smearing, exponent } => {
283                (exponent as f64, smearing * smearing)
284            },
285            _ => unreachable!()
286        };
287
288        let mut fourier = Vec::with_capacity(k_vectors.len());
289        if potential_exponent == 0.0 {
290            let factor = (4.0 * std::f64::consts::PI * smearing_squared).powf(0.75);
291
292            for k_vector in k_vectors {
293                let value = f64::exp(-0.5 * k_vector.norm * k_vector.norm * smearing_squared);
294                fourier.push(factor * value);
295            }
296        } else if potential_exponent == 1.0 {
297            let factor = 4.0 * std::f64::consts::PI;
298
299            for k_vector in k_vectors {
300                let k_norm_squared = k_vector.norm * k_vector.norm;
301                let value = f64::exp(-0.5 * k_norm_squared * smearing_squared) / k_norm_squared;
302                fourier.push(factor * value);
303            }
304        } else {
305            let p_eff = 3.0 - potential_exponent;
306            let factor = std::f64::consts::PI.powf(1.5) * (2.0 * smearing_squared).powf(0.5 * p_eff) / gamma(0.5 * potential_exponent);
307
308            for k_vector in k_vectors {
309                let k_norm_squared = k_vector.norm * k_vector.norm;
310                let x = 0.5 * k_norm_squared * smearing_squared;
311
312                // Compute the gamma_ui over a power law using analytical
313                // expressions for a more stable Fourier transform of the
314                // density
315                let value = if potential_exponent == 2.0 {
316                    f64::sqrt(std::f64::consts::PI / x) * erfc(f64::sqrt(x))
317                } else if potential_exponent == 3.0 {
318                    -expi(-x)
319                } else if potential_exponent == 4.0 {
320                    2.0 * (f64::exp(-x) - f64::sqrt(std::f64::consts::PI*x) * erfc(f64::sqrt(x)))
321                } else if potential_exponent == 5.0 {
322                    f64::exp(-x) + x * expi(-x)
323                } else if potential_exponent == 6.0 {
324                    ((2.0 - 4.0 * x) * f64::exp(-x)
325                        + 4.0 * f64::sqrt(std::f64::consts::PI) * x.powf(1.5) * erfc(f64::sqrt(x))) / 3.0
326                } else if potential_exponent == 7.0 {
327                    (1.0 - x) * f64::exp(-x) / 2.0 - x.powi(2)/2.0 * expi(-x)
328                } else if potential_exponent == 8.0 {
329                    - 2.0 / 15.0 * ((-3.0 + 2.0 * x - 4.0 * x.powi(2)) * f64::exp(-x)
330                        + 4.0 * f64::sqrt(std::f64::consts::PI) * x.powf(2.5) * erfc(f64::sqrt(x)))
331                } else if potential_exponent == 9.0 {
332                    (x.powi(2) - x + 2.0) * f64::exp(-x) / 6.0 + x.powi(3)/6.0 * expi(-x)
333                } else {
334                    panic!("potential_exponent = {} is not implemented", potential_exponent);
335                };
336
337                fourier.push(factor * value);
338            }
339        }
340
341        return fourier.into();
342    }

344    /// Compute k = 0 contributions.
345    ///
346    /// Values are only non zero for `exponent` = 0 and > 3.
347    fn compute_k0_contributions(&self) -> Array1<f64> {
348        let (exponent, smearing) = match self.parameters.density.kind {
349            SmearedPowerLaw { exponent, smearing } => {
350                (exponent, smearing)
351            },
352            _ => unreachable!()
353        };
354
355        let factor = if exponent == 0 {
356            let smearing_squared = smearing * smearing;
357
358            (2.0 * std::f64::consts::PI * smearing_squared).powf(1.5)
359                / (std::f64::consts::PI * smearing_squared).powf(0.75)
360                / f64::sqrt(4.0 * std::f64::consts::PI)
361
362        } else if exponent > 3 {
363            let p_eff = 3.0 - exponent as f64;
364
365            0.5 * std::f64::consts::PI * 2.0_f64.powf(p_eff)
366                / gamma(0.5 * exponent as f64)
367                * 2.0_f64.powf((exponent as f64 - 1.0) / 2.0) / -p_eff
368                * smearing.powf(-p_eff)
369                / smearing.powf(2.0 * exponent as f64 - 6.0)
370        } else {
371            0.0
372        };
373
374         let mut radial_integral = self.radial_integral.get_or(|| {
375            let radial_integral = LodeRadialIntegralCacheByAngular::new(
376                self.parameters.density.kind,
377                &self.parameters.basis,
378                self.parameters.get_k_cutoff()
379            ).expect("could not create a radial integral");
380
381            return RefCell::new(radial_integral);
382        }).borrow_mut();
383
384        let radial_integral = radial_integral.get_mut(0)
385            .expect("k0 contributions can't be done when o3_lambda=0 is missing");
386
387        radial_integral.compute(0.0, false);
388
389        return factor * radial_integral.values.clone();
390    }

392    /// Compute center atom contribution.
393    ///
394    /// By symmetry, this only affects the (l, m) = (0, 0) components of the
395    /// projection coefficients and only the neighbor type blocks that agrees
396    /// with the center atom.
397    fn do_center_contribution(&mut self, systems: &mut[Box<dyn System>], descriptor: &mut TensorMap) -> Result<(), Error> {
398        if !self.parameters.basis.angular_channels().contains(&0) {
399            // o3_lambda is not part of the output, skip self contributions
400            return Ok(());
401        }
402
403        let radial_integral = self.radial_integral.get_or(|| {
404            let radial_integral = LodeRadialIntegralCacheByAngular::new(
405                self.parameters.density.kind, &self.parameters.basis, self.parameters.get_k_cutoff()
406            ).expect("could not create a radial integral");
407
408            return RefCell::new(radial_integral);
409        }).borrow_mut();
410
411        let central_atom_contrib = &radial_integral.get(0)
412            .expect("missing o3_lambda")
413            .get_center_contribution(self.parameters.density.kind)?;
414
415        for (system_i, system) in systems.iter_mut().enumerate() {
416            let types = system.types()?;
417
418            for center_i in 0..system.size()? {
419                let block_i = descriptor.keys().position(&[
420                    0.into(),
421                    1.into(),
422                    types[center_i].into(),
423                    types[center_i].into(),
424                ]);
425
426                if block_i.is_none() {
427                    continue;
428                }
429                let block_i = block_i.expect("we just checked");
430
431                let mut block = descriptor.block_mut_by_id(block_i);
432                let block = block.data_mut();
433                let array = block.values.to_array_mut();
434
435                let sample = [system_i.into(), center_i.into()];
436                let sample_i = match block.samples.position(&sample) {
437                    Some(s) => s,
438                    None => continue
439                };
440
441                for (property_i, [n]) in block.properties.iter_fixed_size().enumerate() {
442                    let n = n.usize();
443                    array[[sample_i, 0, property_i]] -= (1.0 - self.parameters.density.center_atom_weight) * central_atom_contrib[n];
444                }
445            }
446        }
447
448        return Ok(());
449    }

465    fn keys(&self, systems: &mut [Box<dyn System>]) -> Result<Labels, Error> {
466        let builder = AllTypesPairsKeys {};
467        let keys = builder.keys(systems)?;
468
469        let mut builder = LabelsBuilder::new(vec!["o3_lambda", "o3_sigma", "center_type", "neighbor_type"]);
470        for &[center_type, neighbor_type] in keys.iter_fixed_size() {

483    fn samples(&self, keys: &Labels, systems: &mut [Box<dyn System>]) -> Result<Vec<Labels>, Error> {
484        assert_eq!(keys.names(), ["o3_lambda", "o3_sigma", "center_type", "neighbor_type"]);
485
486        // only compute the samples once for each `center_type, neighbor_type`,
487        // and re-use the results across `o3_lambda`.
488        let mut samples_per_types = BTreeMap::new();
489        for [_, _, center_type, neighbor_type] in keys.iter_fixed_size() {
490            if samples_per_types.contains_key(&(center_type, neighbor_type)) {
491                continue;
492            }
493
494            let builder = LongRangeSamplesPerAtom {
495                center_type: AtomicTypeFilter::Single(center_type.i32()),
496                neighbor_type: AtomicTypeFilter::Single(neighbor_type.i32()),
497                self_pairs: true,
498            };
499
500            samples_per_types.insert((center_type, neighbor_type), builder.samples(systems)?);
501        }
502
503        let mut result = Vec::new();
504        for [_, _, center_type, neighbor_type] in keys.iter_fixed_size() {
505            let samples = samples_per_types.get(
506                &(center_type, neighbor_type)
507            ).expect("missing samples");
508
509            result.push(samples.clone());
510        }
511
512        return Ok(result);
513    }

540    fn components(&self, keys: &Labels) -> Vec<Vec<Labels>> {
541        assert_eq!(keys.names(), ["o3_lambda", "o3_sigma", "center_type", "neighbor_type"]);
542
543        // only compute the components once for each `o3_lambda`,
544        // and re-use the results across `center_type, neighbor_type`.
545        let mut component_by_l = BTreeMap::new();
546        for [o3_lambda, _, _, _] in keys.iter_fixed_size() {
547            if component_by_l.contains_key(o3_lambda) {
548                continue;
549            }
550
551            let mut component = LabelsBuilder::new(vec!["o3_mu"]);
552            for m in -o3_lambda.i32()..=o3_lambda.i32() {
553                component.add(&[LabelValue::new(m)]);
554            }
555
556            let components = vec![component.finish_assume_unique()];
557            component_by_l.insert(*o3_lambda, components);
558        }
559
560        let mut result = Vec::new();
561        for [o3_lambda, _, _, _] in keys.iter_fixed_size() {
562            let components = component_by_l.get(o3_lambda).expect("missing samples");
563            result.push(components.clone());
564        }
565        return result;
566    }

568    fn property_names(&self) -> Vec<&str> {
569        vec!["n"]
570    }
571
572    fn properties(&self, keys: &Labels) -> Vec<Labels> {
573        assert_eq!(keys.names(), ["o3_lambda", "o3_sigma", "center_type", "neighbor_type"]);
574
575        match self.parameters.basis {
576            SphericalExpansionBasis::TensorProduct(ref basis) => {
577                let mut properties = LabelsBuilder::new(self.property_names());
578                for n in 0..basis.radial.size() {
579                    properties.add(&[n]);
580                }
581
582                return vec![properties.finish_assume_unique(); keys.count()];
583            }
584            SphericalExpansionBasis::Explicit(ref basis) => {
585                let mut result = Vec::new();
586                for [o3_lambda, _, _, _] in keys.iter_fixed_size() {
587                    let mut properties = LabelsBuilder::new(self.property_names());
588
589                    let radial = basis.by_angular.get(&o3_lambda.usize()).expect("missing o3_lambda");
590                    for n in 0..radial.size() {
591                        properties.add(&[n]);
592                    }
593
594                    result.push(properties.finish_assume_unique());
595                }
596                return result;
597            }
598        }
599    }

601    #[time_graph::instrument(name = "LodeSphericalExpansion::compute")]
602    fn compute(&mut self, systems: &mut [Box<dyn System>], descriptor: &mut TensorMap) -> Result<(), Error> {
603        assert_eq!(descriptor.keys().names(), ["o3_lambda", "o3_sigma", "center_type", "neighbor_type"]);
604
605        self.do_center_contribution(systems, descriptor)?;
606
607        let mut descriptors_by_system = split_tensor_map_by_system(descriptor, systems.len());
608
609        // pick either parallelization over systems (if we have a lot of
610        // systems) or parallelization over samples.
611        //
612        // The parallelization over samples has a slightly higher overhead, so
613        // if we can avoid it we try to!
614        //
615        // Trying to use both at the same time would require additional
616        // synchronisation to ensure the right threads access per-system
617        // pre-computed data (i.e. anything that depends on k-vectors).
618        let (parallel_systems, parallel_samples) = if systems.len() > rayon::current_num_threads() {
619            (true, false)
620        } else {
621            (false, true)
622        };
623        let n_systems = systems.len();
624
625        let potential_exponent = match self.parameters.density.kind {
626            SmearedPowerLaw { exponent, .. } => exponent,
627            _ => unreachable!()
628        };
629
630        systems.par_iter_mut()
631            .zip_eq(&mut descriptors_by_system)
632            .with_min_len(if parallel_systems {1} else {n_systems})
633            .enumerate()
634            .try_for_each(|(system_i, (system, descriptor))| {
635                let types = system.types()?;
636                let cell = system.cell()?;
637                if cell.shape() == UnitCell::infinite().shape() {
638                    return Err(Error::InvalidParameter("LODE can only be used with periodic systems".into()));
639                }
640
641                let k_vectors = compute_k_vectors(&cell, self.parameters.get_k_cutoff());
642                if k_vectors.is_empty() {
643                    return Err(Error::InvalidParameter("No k-vectors for current combination of hyper parameters.".into()));
644                }
645
646                self.project_k_to_nlm(&k_vectors);
647
648                let structure_factors = compute_structure_factors(
649                    system.positions()?,
650                    system.types()?,
651                    &k_vectors
652                );
653
654                let density_fourier = self.compute_density_fourier(&k_vectors);
655
656                let global_factor = 4.0 * std::f64::consts::PI / cell.volume();
657
658                // Add k = 0 contributions for (m, l) = (0, 0)
659                let has_lambda_0 = self.parameters.basis.angular_channels().contains(&0);
660                if has_lambda_0 && (potential_exponent == 0 || potential_exponent > 3) {
661                    let k0_contrib = &self.compute_k0_contributions();
662                    for &neighbor_type in types {
663                        for center_i in 0..system.size()? {
664                            let block_i = descriptor.keys().position(&[
665                                0.into(),
666                                1.into(),
667                                types[center_i].into(),
668                                neighbor_type.into(),
669                            ]).expect("missing block");
670
671                            let mut block = descriptor.block_mut_by_id(block_i);
672                            let data = block.data_mut();
673                            let mut array = array_mut_for_system(data.values);
674
675                            let sample = [system_i.into(), center_i.into()];
676                            let sample_i = match data.samples.position(&sample) {
677                                Some(s) => s,
678                                None => continue
679                            };
680
681                            for (_property_i, [n]) in data.properties.iter_fixed_size().enumerate() {
682                                let n = n.usize();
683                                array[[sample_i, 0, _property_i]] += global_factor * k0_contrib[[n]];
684                            }
685                        }
686                    }
687                }
688
689                let k_vector_to_m_n = self.k_vector_to_m_n.get()
690                    .expect("k_vector_to_m_n should have been created")
691                    .borrow();
692
693                // Main loop: Iterate over all blocks, and then all samples in
694                // each block (in parallel) to evaluate the projection
695                // coefficients
696                for (key, mut block) in descriptor.iter_mut() {
697                    if block.samples().is_empty() {
698                        continue;
699                    }
700
701                    let o3_lambda = key[0].usize();
702                    let center_type = key[2].i32();
703                    let neighbor_type = key[3].i32();
704
705                    let phase = if o3_lambda % 2 == 0 {
706                        (-1.0_f64).powi(o3_lambda as i32 / 2)
707                    } else {
708                        (-1.0_f64).powi((o3_lambda as i32 + 1) / 2)
709                    };
710
711                    let sf_per_center_real = &structure_factors.real_per_center[&neighbor_type];
712                    let sf_per_center_imag = &structure_factors.imag_per_center[&neighbor_type];
713
714                    let sf_per_center = if o3_lambda % 2 == 0 {
715                        sf_per_center_real
716                    } else {
717                        sf_per_center_imag
718                    };
719
720                    let k_vector_to_m_n = k_vector_to_m_n.get(&o3_lambda).expect("missing o3_lambda");
721
722                    let data = block.data_mut();
723                    let samples = &*data.samples;
724                    let properties = &*data.properties;
725
726                    array_mut_for_system(data.values)
727                        .axis_iter_mut(ndarray::Axis(0))
728                        .into_par_iter()
729                        .with_min_len(if parallel_samples {1} else {samples.count()})
730                        .zip_eq(samples.par_iter())
731                        .for_each(|(mut row, sample)| {
732                            let center_i = sample[1].usize();
733
734                            if types[center_i] != center_type {
735                                // this can happen with sample selection if the
736                                // user adds extra samples
737                                return;
738                            }
739
740                            let sf_center = &sf_per_center.slice(s![center_i, ..]);
741
742                            let mut k_dependent_values = self.k_dependent_values
743                                .get_or(|| RefCell::new(Array1::zeros(0)))
744                                .borrow_mut();
745
746                            resize_array1(&mut k_dependent_values, k_vectors.len());
747                            ndarray::azip!((value in &mut *k_dependent_values, &df in &density_fourier, &sf in sf_center) {
748                                *value = global_factor * phase * df * sf;
749                            });
750
751                            for m in 0..(2 * o3_lambda + 1) {
752                                for (property_i, [n]) in properties.iter_fixed_size().enumerate() {
753                                    let n = n.usize();
754
755                                    let mut value = 0.0;
756                                    for ik in 0..k_vectors.len() {
757                                        // Use unsafe to remove bound checking in release mode with `uget`
758                                        // (everything is still bound checked in debug mode).
759                                        //
760                                        // This divides the calculation time by two for values.
761                                        unsafe {
762                                            value += k_dependent_values.uget(ik)
763                                                   * k_vector_to_m_n.uget([m, n, ik]);
764                                        }
765                                    }
766                                    row[[m, property_i]] += value;
767                                }
768                            }
769                        });
770
771                    if let Some(mut gradient) = block.gradient_mut("positions") {
772                        let gradient = gradient.data_mut();
773                        let gradient_samples = &*gradient.samples;
774
775                        array_mut_for_system(gradient.values)
776                            .axis_iter_mut(ndarray::Axis(0))
777                            .into_par_iter()
778                            .with_min_len(if parallel_samples {1} else {gradient_samples.count()})
779                            .zip_eq(gradient_samples.par_iter())
780                            .for_each(|(mut grad_row, grad_sample)| {
781                                let sample_i = grad_sample[0].usize();
782                                let neighbor_i = grad_sample[2].usize();
783
784                                let sample = &samples[sample_i];
785                                let center_i = sample[1].usize();
786
787                                if center_i != neighbor_i {
788                                    assert!(types[neighbor_i] == neighbor_type);
789                                }
790                                assert!(types[center_i] == center_type);
791
792                                let cosines = &structure_factors.real;
793                                let sines = &structure_factors.imag;
794
795                                let mut sf_grad_pair = self.k_dependent_values
796                                    .get_or(|| RefCell::new(Array1::zeros(0)))
797                                    .borrow_mut();
798
799                                resize_array1(&mut sf_grad_pair, k_vectors.len());
800
801                                if o3_lambda % 2 == 0 {
802                                    // real part of i*e^{i k (ri - rj)}
803                                    for ik in 0..k_vectors.len() {
804                                        let factor = sines[[center_i, ik]] * cosines[[neighbor_i, ik]]
805                                                   - cosines[[center_i, ik]] * sines[[neighbor_i, ik]];
806                                        sf_grad_pair[ik] = 2.0 * factor;
807                                    }
808                                } else {
809                                    // imaginary part of i*e^{i k (ri - rj)}
810                                    for ik in 0..k_vectors.len() {
811                                        let factor = cosines[[center_i, ik]] * cosines[[neighbor_i, ik]]
812                                                   + sines[[center_i, ik]] * sines[[neighbor_i, ik]];
813                                        sf_grad_pair[ik] = -2.0 * factor;
814                                    }
815                                }
816
817                                if center_i == neighbor_i {
818                                    // We want the real/imaginary part of `-\sum_{j != i} i*e^{i k (rj - ri)}`
819                                    // for which we try to reuse the structure factor per center from
820                                    // the values calculation
821                                    let (sf_phase, sf_grad) = if o3_lambda % 2 == 0 {
822                                        (1.0, sf_per_center_imag)
823                                    } else {
824                                        (-1.0, sf_per_center_real)
825                                    };
826
827                                    let sf_center = &sf_grad.slice(s![center_i, ..]);
828
829                                    // #[allow(clippy::assign_op_pattern)]
830                                    if neighbor_type == center_type {
831                                        // we need to remove the contribution from the i-i pair
832                                        ndarray::azip!((pair in &mut *sf_grad_pair, &center in sf_center) {
833                                            *pair = -(sf_phase * center - *pair);
834                                        });
835                                    } else {
836                                        ndarray::azip!((pair in &mut *sf_grad_pair, &center in sf_center) {
837                                            *pair = -sf_phase * center;
838                                        });
839                                    }
840                                }
841
842                                // pre-combine everything that only depends on k
843                                let mut k_dependent_values = sf_grad_pair;
844                                ndarray::azip!((value in &mut *k_dependent_values, &df in &density_fourier) {
845                                    *value *= global_factor * phase * df;
846                                });
847
848                                for m in 0..(2 * o3_lambda + 1) {
849                                    for (property_i, [n]) in properties.iter_fixed_size().enumerate() {
850                                        let n = n.usize();
851
852                                        let mut grad = Vector3D::zero();
853                                        for (ik, k_vector) in k_vectors.iter().enumerate() {
854                                            // Use unsafe to remove bound checking in release mode with `uget`
855                                            // (everything is still bound checked in debug mode).
856                                            //
857                                            // This divides the calculation time by ten for gradients.
858                                            unsafe {
859                                                grad += k_dependent_values.uget(ik)
860                                                      * k_vector_to_m_n.uget([m, n, ik])
861                                                      * k_vector.norm
862                                                      * k_vector.direction;
863                                            }
864                                        }
865
866                                        grad_row[[0, m, property_i]] = grad[0];
867                                        grad_row[[1, m, property_i]] = grad[1];
868                                        grad_row[[2, m, property_i]] = grad[2];
869                                    }
870                                }
871                            });
872                    }
873                }
874
875                return Ok(());
876            }
877        )?;
878
879        Ok(())
880    }
881}
