13/// `erf` calculates the error function at `x`.
14pub fn erf(x: f64) -> f64 {
15    if x.is_nan() {
16        f64::NAN
17    } else if x >= 0.0 && x.is_infinite() {
18        1.0
19    } else if x <= 0.0 && x.is_infinite() {
20        -1.0
21    } else if x == 0.0 {
22        0.0
23    } else {
24        erf_impl(x, false)
25    }
26}

43/// `erfc` calculates the complementary error function at `x`.
44pub fn erfc(x: f64) -> f64 {
45    if x.is_nan() {
46        f64::NAN
47    } else if x == f64::INFINITY {
48        0.0
49    } else if x == f64::NEG_INFINITY {
50        2.0
51    } else {
52        erf_impl(x, true)
53    }
54}

382/// `erf_impl` computes the error function at `z`.
383/// If `inv` is true, `1 - erf` is calculated as opposed to `erf`
384#[allow(clippy::too_many_lines)]
385fn erf_impl(z: f64, inv: bool) -> f64 {
386    if z < 0.0 {
387        if !inv {
388            return -erf_impl(-z, false);
389        }
390        if z < -0.5 {
391            return 2.0 - erf_impl(-z, true);
392        }
393        return 1.0 + erf_impl(-z, false);
394    }
395
396    let result = if z < 0.5 {
397        if z < 1e-10 {
398            z * 1.125 + z * 0.003379167095512573896158903121545171688
399        } else {
400            z * 1.125
401                + z * evaluate_polynomial(z, ERF_IMPL_AN) / evaluate_polynomial(z, ERF_IMPL_AD)
402        }
403    } else if z < 110.0 {
404        let (r, b) = if z < 0.75 {
405            (
406                evaluate_polynomial(z - 0.5, ERF_IMPL_BN)
407                    / evaluate_polynomial(z - 0.5, ERF_IMPL_BD),
408                0.3440242112,
409            )
410        } else if z < 1.25 {
411            (
412                evaluate_polynomial(z - 0.75, ERF_IMPL_CN)
413                    / evaluate_polynomial(z - 0.75, ERF_IMPL_CD),
414                0.419990927,
415            )
416        } else if z < 2.25 {
417            (
418                evaluate_polynomial(z - 1.25, ERF_IMPL_DN)
419                    / evaluate_polynomial(z - 1.25, ERF_IMPL_DD),
420                0.4898625016,
421            )
422        } else if z < 3.5 {
423            (
424                evaluate_polynomial(z - 2.25, ERF_IMPL_EN)
425                    / evaluate_polynomial(z - 2.25, ERF_IMPL_ED),
426                0.5317370892,
427            )
428        } else if z < 5.25 {
429            (
430                evaluate_polynomial(z - 3.5, ERF_IMPL_FN)
431                    / evaluate_polynomial(z - 3.5, ERF_IMPL_FD),
432                0.5489973426,
433            )
434        } else if z < 8.0 {
435            (
436                evaluate_polynomial(z - 5.25, ERF_IMPL_GN)
437                    / evaluate_polynomial(z - 5.25, ERF_IMPL_GD),
438                0.5571740866,
439            )
440        } else if z < 11.5 {
441            (
442                evaluate_polynomial(z - 8.0, ERF_IMPL_HN)
443                    / evaluate_polynomial(z - 8.0, ERF_IMPL_HD),
444                0.5609807968,
445            )
446        } else if z < 17.0 {
447            (
448                evaluate_polynomial(z - 11.5, ERF_IMPL_IN)
449                    / evaluate_polynomial(z - 11.5, ERF_IMPL_ID),
450                0.5626493692,
451            )
452        } else if z < 24.0 {
453            (
454                evaluate_polynomial(z - 17.0, ERF_IMPL_JN)
455                    / evaluate_polynomial(z - 17.0, ERF_IMPL_JD),
456                0.5634598136,
457            )
458        } else if z < 38.0 {
459            (
460                evaluate_polynomial(z - 24.0, ERF_IMPL_KN)
461                    / evaluate_polynomial(z - 24.0, ERF_IMPL_KD),
462                0.5638477802,
463            )
464        } else if z < 60.0 {
465            (
466                evaluate_polynomial(z - 38.0, ERF_IMPL_LN)
467                    / evaluate_polynomial(z - 38.0, ERF_IMPL_LD),
468                0.5640528202,
469            )
470        } else if z < 85.0 {
471            (
472                evaluate_polynomial(z - 60.0, ERF_IMPL_MN)
473                    / evaluate_polynomial(z - 60.0, ERF_IMPL_MD),
474                0.5641309023,
475            )
476        } else {
477            (
478                evaluate_polynomial(z - 85.0, ERF_IMPL_NN)
479                    / evaluate_polynomial(z - 85.0, ERF_IMPL_ND),
480                0.5641584396,
481            )
482        };
483        let g = (-z * z).exp() / z;
484        g * b + g * r
485    } else {
486        0.0
487    };
488
489    if inv && z >= 0.5 {
490        result
491    } else if z >= 0.5 || inv {
492        1.0 - result
493    } else {
494        result
495    }
496}

743/// Evaluates a polynomial at `z` where `coeff` are the coefficients to a
744/// polynomial of order `k` where `k` is the length of `coeff` and the
745/// coefficient to the `k`th power is the `k`th element in coeff. E.g. [3,-1,2]
746/// equates to `2z^2 - z + 3`
747///
748/// # Remarks
749///
750/// Returns 0 for a 0 length coefficient slice
751pub fn evaluate_polynomial(z: f64, coeff: &[f64]) -> f64 {
752    let n = coeff.len();
753    if n == 0 {
754        return 0.0;
755    }
756
757    let mut sum = *coeff.last().unwrap();
758    for c in coeff[0..n - 1].iter().rev() {
759        sum = *c + z * sum;
760    }
761    sum
762}

770    #[test]
771    fn test_polynomial() {
772        let empty: [f64; 0] = [];
773        assert_eq!(evaluate_polynomial(2.0, &empty), 0.0);
774
775        let zero = [0.0];
776        assert_eq!(evaluate_polynomial(2.0, &zero), 0.0);
777
778        let mut coeff = [1.0, 0.0, 5.0];
779        assert_eq!(evaluate_polynomial(2.0, &coeff), 21.0);
780
781        coeff = [-5.0, -2.0, 3.0];
782        assert_eq!(evaluate_polynomial(2.0, &coeff), 3.0);
783        assert_eq!(evaluate_polynomial(-2.0, &coeff), 11.0);
784
785        let large_coeff = [-1.35e3, 2.5e2, 8.0, -4.0, 1e2, 3.0];
786        assert_eq!(evaluate_polynomial(5.0, &large_coeff), 71475.0);
787        assert_eq!(evaluate_polynomial(-5.0, &large_coeff), 51225.0);
788
789        coeff = [f64::INFINITY, -2.0, 3.0];
790        assert_eq!(evaluate_polynomial(2.0, &coeff), f64::INFINITY);
791        assert_eq!(evaluate_polynomial(-2.0, &coeff), f64::INFINITY);
792
793        coeff = [f64::NEG_INFINITY, -2.0, 3.0];
794        assert_eq!(evaluate_polynomial(2.0, &coeff), f64::NEG_INFINITY);
795        assert_eq!(evaluate_polynomial(-2.0, &coeff), f64::NEG_INFINITY);
796
797        coeff = [f64::NAN, -2.0, 3.0];
798        assert!(evaluate_polynomial(2.0, &coeff).is_nan());
799        assert!(evaluate_polynomial(-2.0, &coeff).is_nan());
