8/// A `LodeRadialIntegral` computes the LODE radial integral  for all radial
9/// basis functions and a single spherical harmonic `l` channel
10///
11/// See equations 5 to 8 of [this paper](https://doi.org/10.1063/5.0044689) for
12/// mor information on the radial integral.
13///
14/// `std::panic::RefUnwindSafe` is a required super-trait to enable passing
15/// radial integrals across the C API. `Send` is a required super-trait to
16/// enable passing radial integrals between threads.
17pub trait LodeRadialIntegral: std::panic::RefUnwindSafe + Send {
18    /// Compute the LODE radial integral for a single k-vector `norm` and store
19    /// the resulting data in the `values` array. If `gradients` is `Some`, also
20    /// compute and store gradients there.
21    fn compute(&self, k_norm: f64, values: ArrayViewMut1<f64>, gradients: Option<ArrayViewMut1<f64>>);
22
23    /// Get how many basis functions are part of this integral. This is the
24    /// shape to use for the `values` and `gradients` parameters to `compute`.
25    fn size(&self) -> usize;
26
27    /// Compute the contribution of the central atom to the final `<n l m>`
28    /// coefficients. By symmetry, only l=0 is non-zero, so this function
29    /// returns a 1-D array containing the different `<n 0 0>` coefficients.
30    ///
31    /// This function differs from the rest of LODE calculation because it goes
32    /// straight from atom => n l m, without using k-space projection in the
33    /// middle.
34    fn get_center_contribution(&self, density: DensityKind) -> Result<Array1<f64>, Error>;
35}

43/// Store together a Radial integral implementation and cached allocation for
44/// values/gradients.
45pub struct LodeRadialIntegralCache {
46    /// Implementations of the radial integral
47    implementation: Box<dyn LodeRadialIntegral>,
48    /// Cache for the radial integral values
49    pub(crate) values: Array1<f64>,
50    /// Cache for the radial integral gradient
51    pub(crate) gradients: Array1<f64>,
52}
53
54impl LodeRadialIntegralCache {
55    fn new(
56        o3_lambda: usize,
57        radial: &LodeRadialBasis,
58        density: DensityKind,
59        k_cutoff: f64,
60        spline_accuracy: Option<f64>,
61    ) -> Result<LodeRadialIntegralCache, Error> {
62        let implementation = match radial {
63            LodeRadialBasis::Gto { .. } => {
64                let gto = LodeRadialIntegralGto::new(radial, o3_lambda)?;
65
66                if let Some(accuracy) = spline_accuracy {
67                    let do_center_contribution = o3_lambda == 0;
68                    Box::new(LodeRadialIntegralSpline::with_accuracy(
69                        gto, density, k_cutoff, accuracy, do_center_contribution
70                    )?)
71                } else {
72                    Box::new(gto) as Box<dyn LodeRadialIntegral>
73                }
74            },
75            LodeRadialBasis::Tabulated(ref tabulated) => {
76                Box::new(LodeRadialIntegralSpline::from_tabulated(
77                    tabulated.clone(),
78                    density,
79                )) as Box<dyn LodeRadialIntegral>
80            }
81        };
82
83        let size = implementation.size();
84        let values = Array1::from_elem(size, 0.0);
85        let gradients = Array1::from_elem(size, 0.0);
86
87        return Ok(LodeRadialIntegralCache {
88            implementation,
89            values,
90            gradients,
91        });
92    }
93
94    /// Run the calculation, the results are stored inside `self.values` and
95    /// `self.gradients`
96    pub fn compute(&mut self, k_norm: f64, do_gradients: bool) {
97        let gradient_view = if do_gradients {
98            Some(self.gradients.view_mut())
99        } else {
100            None
101        };
102
103        self.implementation.compute(k_norm, self.values.view_mut(), gradient_view);
104    }
105
106    /// Get the size of this radial integral (i.e. number of radial basis)
107    pub fn size(&self) -> usize {
108        return self.implementation.size()
109    }
110
111    /// Get the size of this radial integral (i.e. number of radial basis)
112    pub fn get_center_contribution(&self, density: DensityKind) -> Result<Array1<f64>, Error> {
113        return self.implementation.get_center_contribution(density);
114    }
115}

117/// Store all `LodeRadialIntegralCache` for different angular channels
118pub struct LodeRadialIntegralCacheByAngular {
119    pub(crate) by_angular: BTreeMap<usize, LodeRadialIntegralCache>,
120}
121
122impl LodeRadialIntegralCacheByAngular {
123    /// Create a new `LodeRadialIntegralCacheByAngular` for the given radial basis & parameters
124    pub fn new(
125        density: DensityKind,
126        basis: &SphericalExpansionBasis<LodeRadialBasis>,
127        k_cutoff: f64
128    ) -> Result<Self, Error> {
129        match basis {
130            SphericalExpansionBasis::TensorProduct(basis) => {
131                let mut by_angular = BTreeMap::new();
132                for o3_lambda in 0..=basis.max_angular {
133                    let cache = LodeRadialIntegralCache::new(
134                        o3_lambda,
135                        &basis.radial,
136                        density,
137                        k_cutoff,
138                        basis.spline_accuracy
139                    )?;
140                    by_angular.insert(o3_lambda, cache);
141                }
142
143                return Ok(LodeRadialIntegralCacheByAngular {
144                    by_angular
145                });
146            }
147            SphericalExpansionBasis::Explicit(basis) => {
148                let mut by_angular = BTreeMap::new();
149                for (&o3_lambda, radial) in &*basis.by_angular {
150                    let cache = LodeRadialIntegralCache::new(
151                        o3_lambda,
152                        radial,
153                        density,
154                        k_cutoff,
155                        basis.spline_accuracy
156                    )?;
157                    by_angular.insert(o3_lambda, cache);
158                }
159                return Ok(LodeRadialIntegralCacheByAngular {
160                    by_angular
161                });
162            }
163        }
164    }
165
166    /// Run the calculation, the results are accessible with `get`
167    pub fn compute(&mut self, distance: f64, do_gradients: bool) {
168        self.by_angular.iter_mut().for_each(|(_, cache)| cache.compute(distance, do_gradients));
169    }
170
171    /// Get one of the individual cache, corresponding to the `o3_lambda`
172    /// angular channel
173    pub fn get(&self, o3_lambda: usize) -> Option<&LodeRadialIntegralCache> {
174        self.by_angular.get(&o3_lambda)
175    }
176
177    pub(crate) fn get_mut(&mut self, o3_lambda: usize) -> Option<&mut LodeRadialIntegralCache> {
178        self.by_angular.get_mut(&o3_lambda)
179    }
